1//! Minimal shell scripting: variable expansion, `if`, `for`, `while`.
2//!
3//! # Supported constructs
4//!
5//! - **Variables**: `set VAR=VALUE`, `$VAR` expansion, `$?` for last exit code.
6//! - **For loops**: `for VAR in A B C ; do COMMAND ; done`
7//! - **While loops**: `while COMMAND ; do BODY ; done`
8//! - **If/then**: `if COMMAND ; then BODY ; fi` or `if COMMAND ; then A ; else B ; fi`
9
10use alloc::{collections::BTreeMap, string::{String, ToString}, vec::Vec};
11use crate::sync::SpinLock;
12
13static SHELL_VARS: SpinLock<BTreeMap<String, String>> = SpinLock::new(BTreeMap::new());
14static LAST_EXIT: core::sync::atomic::AtomicI32 = core::sync::atomic::AtomicI32::new(0);
15
16/// Set the exit code of the last executed command.
17pub fn set_last_exit(code: i32) {
18    LAST_EXIT.store(code, core::sync::atomic::Ordering::Relaxed);
19}
20
21/// Get the exit code of the last executed command.
22pub fn last_exit() -> i32 {
23    LAST_EXIT.load(core::sync::atomic::Ordering::Relaxed)
24}
25
26/// Set a shell variable.
27pub fn set_var(key: &str, val: &str) {
28    SHELL_VARS.lock().insert(String::from(key), String::from(val));
29}
30
31/// Get a shell variable.
32pub fn get_var(key: &str) -> Option<String> {
33    SHELL_VARS.lock().get(key).cloned()
34}
35
36/// Remove a shell variable.
37pub fn unset_var(key: &str) {
38    SHELL_VARS.lock().remove(key);
39}

41/// Expand `$VAR` and `$?` references in a string.
42pub fn expand_vars(input: &str) -> String {
43    let mut result = String::new();
44    let bytes = input.as_bytes();
45    let mut i = 0;
46
47    while i < bytes.len() {
48        if bytes[i] == b'$' {
49            i += 1;
50            if i < bytes.len() && bytes[i] == b'?' {
51                result.push_str(&alloc::format!("{}", last_exit()));
52                i += 1;
53            } else {
54                let start = i;
55                while i < bytes.len() && (bytes[i].is_ascii_alphanumeric() || bytes[i] == b'_') {
56                    i += 1;
57                }
58                if i == start {
59                    // Keep literal '$' when no variable name follows.
60                    result.push('$');
61                    continue;
62                }
63                let var_name = core::str::from_utf8(&bytes[start..i]).unwrap_or("");
64                if let Some(val) = get_var(var_name) {
65                    result.push_str(&val);
66                } else if let Some(env_val) = super::commands::util::shell_getenv(var_name) {
67                    result.push_str(&env_val);
68                }
69            }
70        } else {
71            result.push(bytes[i] as char);
72            i += 1;
73        }
74    }
75
76    result
77}

79/// Script construct types recognized by the parser.
80pub enum ScriptConstruct {
81    /// Simple command line (possibly with pipes).
82    Simple(String),
83    /// `set VAR=VALUE`
84    SetVar { key: String, val: String },
85    /// `unset VAR`
86    UnsetVar(String),
87    /// `for VAR in ITEMS ; do BODY ; done`
88    ForLoop { var: String, items: Vec<String>, body: Vec<String> },
89    /// `while COND ; do BODY ; done`
90    WhileLoop { cond: String, body: Vec<String> },
91    /// `if COND ; then THEN_BODY [; else ELSE_BODY] ; fi`
92    IfElse { cond: String, then_body: Vec<String>, else_body: Vec<String> },
93}
94
95/// Parse a full line into a script construct.
96///
97/// The parser splits on `;` to find keywords. For simple commands
98/// (no scripting keywords), returns [`ScriptConstruct::Simple`].
99pub fn parse_script(line: &str) -> ScriptConstruct {
100    let trimmed = line.trim();
101
102    if trimmed.starts_with("set ") {
103        let rest = &trimmed[4..];
104        if let Some(eq) = rest.find('=') {
105            return ScriptConstruct::SetVar {
106                key: String::from(rest[..eq].trim()),
107                val: String::from(rest[eq + 1..].trim()),
108            };
109        }
110        return ScriptConstruct::Simple(String::from(trimmed));
111    }
112
113    if trimmed.starts_with("unset ") {
114        return ScriptConstruct::UnsetVar(String::from(trimmed[6..].trim()));
115    }
116
117    let parts: Vec<&str> = trimmed.split(';').map(|s| s.trim()).collect();
118
119    if parts.first().map(|p| p.starts_with("for ")) == Some(true) {
120        return parse_for_loop(&parts);
121    }
122    if parts.first().map(|p| p.starts_with("while ")) == Some(true) {
123        return parse_while_loop(&parts);
124    }
125    if parts.first().map(|p| p.starts_with("if ")) == Some(true) {
126        return parse_if_else(&parts);
127    }
128
129    ScriptConstruct::Simple(String::from(trimmed))
130}

132/// `for VAR in A B C ; do cmd1 ; cmd2 ; done`
133fn parse_for_loop(parts: &[&str]) -> ScriptConstruct {
134    if parts.is_empty() {
135        return ScriptConstruct::Simple(String::new());
136    }
137    let header = parts[0];
138    let tokens: Vec<&str> = header.split_whitespace().collect();
139    if tokens.len() < 4 || tokens[0] != "for" || tokens[2] != "in" {
140        return ScriptConstruct::Simple(String::from(header));
141    }
142
143    let var = tokens.get(1).unwrap_or(&"_").to_string();
144    let items: Vec<String> = tokens.iter()
145        .skip(3) // skip "for VAR in"
146        .map(|s| String::from(*s))
147        .collect();
148
149    let Some(do_idx) = parts.iter().position(|p| *p == "do") else {
150        return ScriptConstruct::Simple(String::from(header));
151    };
152    let Some(done_idx) = parts.iter().position(|p| *p == "done") else {
153        return ScriptConstruct::Simple(String::from(header));
154    };
155    if done_idx <= do_idx {
156        return ScriptConstruct::Simple(String::from(header));
157    }
158
159    let body: Vec<String> = parts[do_idx + 1..done_idx]
160        .iter()
161        .filter(|s| !s.is_empty())
162        .map(|s| String::from(*s))
163        .collect();
164
165    ScriptConstruct::ForLoop { var: String::from(var), items, body }
166}

168/// `while cond ; do body ; done`
169fn parse_while_loop(parts: &[&str]) -> ScriptConstruct {
170    if parts.is_empty() {
171        return ScriptConstruct::Simple(String::new());
172    }
173    let header = parts[0];
174    let Some(cond) = header.strip_prefix("while ") else {
175        return ScriptConstruct::Simple(String::from(header));
176    };
177    let Some(do_idx) = parts.iter().position(|p| *p == "do") else {
178        return ScriptConstruct::Simple(String::from(header));
179    };
180    let Some(done_idx) = parts.iter().position(|p| *p == "done") else {
181        return ScriptConstruct::Simple(String::from(header));
182    };
183    if done_idx <= do_idx {
184        return ScriptConstruct::Simple(String::from(header));
185    }
186
187    let body: Vec<String> = parts[do_idx + 1..done_idx]
188        .iter()
189        .filter(|s| !s.is_empty())
190        .map(|s| String::from(*s))
191        .collect();
192
193    ScriptConstruct::WhileLoop {
194        cond: String::from(cond),
195        body,
196    }
197}

199/// `if cond ; then body ; [else body ;] fi`
200fn parse_if_else(parts: &[&str]) -> ScriptConstruct {
201    if parts.is_empty() {
202        return ScriptConstruct::Simple(String::new());
203    }
204    let header = parts[0];
205    let Some(cond) = header.strip_prefix("if ") else {
206        return ScriptConstruct::Simple(String::from(header));
207    };
208    let Some(then_idx) = parts.iter().position(|p| *p == "then") else {
209        return ScriptConstruct::Simple(String::from(header));
210    };
211    let else_idx = parts.iter().position(|p| *p == "else");
212    let Some(fi_idx) = parts.iter().position(|p| *p == "fi") else {
213        return ScriptConstruct::Simple(String::from(header));
214    };
215    if fi_idx <= then_idx {
216        return ScriptConstruct::Simple(String::from(header));
217    }
218
219    let then_end = else_idx.unwrap_or(fi_idx);
220    let then_body: Vec<String> = parts[then_idx + 1..then_end]
221        .iter()
222        .filter(|s| !s.is_empty())
223        .map(|s| String::from(*s))
224        .collect();
225
226    let else_body: Vec<String> = if let Some(ei) = else_idx {
227        parts[ei + 1..fi_idx]
228            .iter()
229            .filter(|s| !s.is_empty())
230            .map(|s| String::from(*s))
231            .collect()
232    } else {
233        Vec::new()
234    };
235
236    ScriptConstruct::IfElse {
237        cond: String::from(cond),
238        then_body,
239        else_body,
240    }
