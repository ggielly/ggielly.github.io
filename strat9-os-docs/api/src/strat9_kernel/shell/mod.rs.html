40/// Global flag set by Ctrl+C. Long-running commands should poll this
41/// via [`is_interrupted`] and abort early when it returns `true`.
42pub static SHELL_INTERRUPTED: AtomicBool = AtomicBool::new(false);
43
44/// Returns `true` if Ctrl+C was pressed, and clears the flag.
45///
46/// Commands that loop (e.g. `top`, `watch`) should call this each
47/// iteration to support cancellation.
48pub fn is_interrupted() -> bool {
49    SHELL_INTERRUPTED.swap(false, Ordering::Relaxed)
50}

61/// Returns whether continuation byte.
62#[inline]
63fn is_continuation_byte(b: u8) -> bool {
64    (b & 0b1100_0000) == 0b1000_0000
65}

67/// Performs the prev char boundary operation.
68fn prev_char_boundary(input: &[u8], mut idx: usize) -> usize {
69    if idx == 0 {
70        return 0;
71    }
72    idx -= 1;
73    while idx > 0 && is_continuation_byte(input[idx]) {
74        idx -= 1;
75    }
76    idx
77}
78
79/// Performs the next char boundary operation.
80fn next_char_boundary(input: &[u8], mut idx: usize) -> usize {
81    if idx >= input.len() {
82        return input.len();
83    }
84    idx += 1;
85    while idx < input.len() && is_continuation_byte(input[idx]) {
86        idx += 1;
87    }
88    idx
89}

91/// Performs the char count operation.
92fn char_count(input: &[u8]) -> usize {
93    core::str::from_utf8(input)
94        .map(|s| s.chars().count())
95        .unwrap_or(input.len())
96}

98/// Performs the print bytes operation.
99fn print_bytes(input: &[u8]) {
100    if let Ok(s) = core::str::from_utf8(input) {
101        for ch in s.chars() {
102            print_char(ch);
103        }
104    } else {
105        for &b in input {
106            print_char(if b.is_ascii() { b as char } else { '?' });
107        }
108    }
109}

111/// Performs the move cursor left chars operation.
112fn move_cursor_left_chars(n: usize) {
113    for _ in 0..n {
114        print_char('\x08');
115    }
116}

118/// Performs the clear visible line operation.
119fn clear_visible_line(line: &[u8]) {
120    let n = char_count(line);
121    move_cursor_left_chars(n);
122    for _ in 0..n {
123        print_char(' ');
124    }
125    move_cursor_left_chars(n);
126}

128/// Redraw the current shell input line after the prompt
129fn redraw_line(input: &[u8], cursor_pos: usize) {
130    print_bytes(input);
131
132    // Print a trailing space to clear any leftover char from a longer previous line
133    print_char(' ');
134    print_char('\x08');
135
136    // Move visual cursor back to its logical position
137    let back_moves = if cursor_pos <= input.len() {
138        if let (Ok(full), Ok(prefix)) = (
139            core::str::from_utf8(input),
140            core::str::from_utf8(&input[..cursor_pos]),
141        ) {
142            full.chars().count().saturating_sub(prefix.chars().count())
143        } else {
144            input.len().saturating_sub(cursor_pos)
145        }
146    } else {
147        0
148    };
149    for _ in 0..back_moves {
150        print_char('\x08');
151    }
152}

154/// Performs the redraw full line operation.
155fn redraw_full_line(input: &[u8], cursor_pos: usize) {
156    clear_visible_line(input);
157    print_bytes(input);
158    if cursor_pos <= input.len() {
159        if let Ok(sfx) = core::str::from_utf8(&input[cursor_pos..]) {
160            move_cursor_left_chars(sfx.chars().count());
161        } else {
162            move_cursor_left_chars(input.len().saturating_sub(cursor_pos));
163        }
164    }
165}

167/// Performs the insert bytes at cursor operation.
168fn insert_bytes_at_cursor(
169    input_buf: &mut [u8],
170    input_len: &mut usize,
171    cursor_pos: &mut usize,
172    bytes: &[u8],
173) -> bool {
174    if bytes.is_empty() {
175        return true;
176    }
177    if *input_len + bytes.len() > input_buf.len() {
178        return false;
179    }
180    let old_cursor = *cursor_pos;
181    if old_cursor < *input_len {
182        for i in (old_cursor..*input_len).rev() {
183            input_buf[i + bytes.len()] = input_buf[i];
184        }
185    }
186    input_buf[old_cursor..old_cursor + bytes.len()].copy_from_slice(bytes);
187    *input_len += bytes.len();
188    *cursor_pos += bytes.len();
189    redraw_line(&input_buf[old_cursor..*input_len], bytes.len());
190    true
191}

193/// Performs the delete prev char at cursor operation.
194fn delete_prev_char_at_cursor(
195    input_buf: &mut [u8],
196    input_len: &mut usize,
197    cursor_pos: &mut usize,
198) -> bool {
199    if *cursor_pos == 0 {
200        return false;
201    }
202
203    let prev = prev_char_boundary(&input_buf[..*input_len], *cursor_pos);
204    let removed = *cursor_pos - prev;
205    for i in *cursor_pos..*input_len {
206        input_buf[i - removed] = input_buf[i];
207    }
208    *input_len -= removed;
209    *cursor_pos = prev;
210
211    // Backspace behavior: visual cursor moves left by one character first.
212    move_cursor_left_chars(1);
213    redraw_line(&input_buf[*cursor_pos..*input_len], 0);
214    true
215}

217/// Performs the delete next char at cursor operation.
218fn delete_next_char_at_cursor(
219    input_buf: &mut [u8],
220    input_len: &mut usize,
221    cursor_pos: &mut usize,
222) -> bool {
223    if *cursor_pos >= *input_len {
224        return false;
225    }
226
227    let next = next_char_boundary(&input_buf[..*input_len], *cursor_pos);
228    let removed = next - *cursor_pos;
229    for i in next..*input_len {
230        input_buf[i - removed] = input_buf[i];
231    }
232    *input_len -= removed;
233
234    // Delete behavior: cursor stays at the same logical position.
235    redraw_line(&input_buf[*cursor_pos..*input_len], 0);
236    true
237}

239/// Main shell loop
240///
241/// This function never returns. It continuously reads keyboard input,
242/// parses commands, and executes them.
243pub extern "C" fn shell_main() -> ! {
244    let registry = CommandRegistry::new();
245    commands::util::init_shell_env();
246    let mut input_buf = [0u8; 256];
247    let mut input_len = 0;
248    let mut cursor_pos = 0;
249
250    // Command history
251    let mut history = VecDeque::new();
252    let mut history_idx: isize = -1;
253    let mut current_input_saved = String::new();
254    let mut utf8_pending = [0u8; 4];
255    let mut utf8_pending_len = 0usize;
256    let mut in_escape_seq = false;
257
258    // Mouse state
259    let mut prev_left = false;
260    let mut selecting = false;
261    let mut scrollbar_dragging = false;
262    let mut last_scrollbar_drag_tick = 0u64;
263    let mut pending_scrollbar_drag_y: Option<usize> = None;
264    let mut mouse_x: i32 = 0;
265    let mut mouse_y: i32 = 0;
266
267    // Display welcome message using ASCII for robust terminal rendering.
268    shell_println!("");
269    shell_println!("+--------------------------------------------------------------+");
270    shell_println!("|         Strat9-OS chevron shell v0.1.0 (Bedrock)            |");
271    shell_println!("|         Type 'help' for available commands                  |");
272    shell_println!("+--------------------------------------------------------------+");
273    shell_println!("");
274
275    print_prompt();
276
277    let mut last_blink_tick = 0;
278    let mut cursor_visible = false;
279    const MAX_MOUSE_EVENTS_PER_TURN: usize = 64;
280    const SCROLLBAR_DRAG_MIN_TICKS: u64 = 1;
281
282    loop {
283        // Handle cursor blinking (graphics only)
284        let ticks = crate::process::scheduler::ticks();
285
286        if ticks / 50 != last_blink_tick {
287            last_blink_tick = ticks / 50;
288            cursor_visible = !cursor_visible;
289
290            if crate::arch::x86_64::vga::is_available() {
291                let color = if cursor_visible {
292                    crate::arch::x86_64::vga::RgbColor::new(0x4F, 0xB3, 0xB3) // Cyan
293                } else {
294                    crate::arch::x86_64::vga::RgbColor::new(0x12, 0x16, 0x1E) // Background
295                };
296                crate::arch::x86_64::vga::draw_text_cursor(color);
297            }
298        }
299
300        // Read from keyboard buffer
301        if let Some(ch) = crate::arch::x86_64::keyboard::read_char() {
302            // Any keypress returns the view to live output.
303            if crate::arch::x86_64::vga::is_available() {
304                crate::arch::x86_64::vga::scroll_to_live();
305            }
306
307            // Hide cursor before any action
308            if crate::arch::x86_64::vga::is_available() {
309                crate::arch::x86_64::vga::draw_text_cursor(
310                    crate::arch::x86_64::vga::RgbColor::new(0x12, 0x16, 0x1E),
311                );
312            }
313
314            match ch {
315                b'\r' | b'\n' => {
316                    in_escape_seq = false;
317                    utf8_pending_len = 0;
318                    shell_println!();
319
320                    if input_len > 0 {
321                        let line = core::str::from_utf8(&input_buf[..input_len]).unwrap_or("");
322
323                        if !line.is_empty() {
324                            if history.is_empty()
325                                || history.back().map(|s: &String| s.as_str()) != Some(line)
326                            {
327                                history.push_back(line.to_string());
328                                if history.len() > 50 {
329                                    history.pop_front();
330                                }
331                            }
332                        }
333
334                        execute_line(line, &registry);
335                        input_len = 0;
336                        cursor_pos = 0;
337                        history_idx = -1;
338                    }
339
340                    print_prompt();
341                }
342                b'\x08' | b'\x7f' => {
343                    in_escape_seq = false;
344                    utf8_pending_len = 0;
345                    let _ = delete_prev_char_at_cursor(
346                        &mut input_buf,
347                        &mut input_len,
348                        &mut cursor_pos,
349                    );
350                }
351                b'\x03' => {
352                    in_escape_seq = false;
353                    utf8_pending_len = 0;
354                    shell_println!("^C");
355                    input_len = 0;
356                    cursor_pos = 0;
357                    history_idx = -1;
358                    SHELL_INTERRUPTED.store(false, Ordering::Relaxed);
359                    print_prompt();
360                }
361                b'\t' => {
362                    in_escape_seq = false;
363                    utf8_pending_len = 0;
364                    tab_complete(
365                        &mut input_buf,
366                        &mut input_len,
367                        &mut cursor_pos,
368                        &registry,
369                    );
370                }
371                b'\x04' => {
372                    in_escape_seq = false;
373                    utf8_pending_len = 0;
374                    let _ = delete_next_char_at_cursor(
375                        &mut input_buf,
376                        &mut input_len,
377                        &mut cursor_pos,
378                    );
379                }
380                KEY_LEFT => {
381                    in_escape_seq = false;
382                    utf8_pending_len = 0;
383                    if cursor_pos > 0 {
384                        cursor_pos = prev_char_boundary(&input_buf[..input_len], cursor_pos);
385                        print_char('\x08');
386                    }
387                }
388                KEY_RIGHT => {
389                    in_escape_seq = false;
390                    utf8_pending_len = 0;
391                    if cursor_pos < input_len {
392                        let next = next_char_boundary(&input_buf[..input_len], cursor_pos);
393                        print_bytes(&input_buf[cursor_pos..next]);
394                        cursor_pos = next;
395                    }
396                }
397                KEY_HOME => {
398                    in_escape_seq = false;
399                    utf8_pending_len = 0;
400                    while cursor_pos > 0 {
401                        cursor_pos = prev_char_boundary(&input_buf[..input_len], cursor_pos);
402                        print_char('\x08');
403                    }
404                }
405                KEY_END => {
406                    in_escape_seq = false;
407                    utf8_pending_len = 0;
408                    while cursor_pos < input_len {
409                        let next = next_char_boundary(&input_buf[..input_len], cursor_pos);
410                        print_bytes(&input_buf[cursor_pos..next]);
411                        cursor_pos = next;
412                    }
413                }
414                KEY_UP => {
415                    in_escape_seq = false;
416                    utf8_pending_len = 0;
417                    if !history.is_empty() && history_idx < (history.len() as isize - 1) {
418                        if history_idx == -1 {
419                            current_input_saved = core::str::from_utf8(&input_buf[..input_len])
420                                .unwrap_or("")
421                                .to_string();
422                        }
423
424                        while cursor_pos < input_len {
425                            let next = next_char_boundary(&input_buf[..input_len], cursor_pos);
426                            print_bytes(&input_buf[cursor_pos..next]);
427                            cursor_pos = next;
428                        }
429                        clear_visible_line(&input_buf[..input_len]);
430
431                        history_idx += 1;
432                        let hist_str = &history[history.len() - 1 - history_idx as usize];
433                        let bytes = hist_str.as_bytes();
434                        let copy_len = bytes.len().min(input_buf.len());
435                        input_buf[..copy_len].copy_from_slice(&bytes[..copy_len]);
436                        input_len = copy_len;
437                        cursor_pos = input_len;
438
439                        redraw_full_line(&input_buf[..input_len], cursor_pos);
440                    }
441                }
442                KEY_DOWN => {
443                    in_escape_seq = false;
444                    utf8_pending_len = 0;
445                    if history_idx >= 0 {
446                        while cursor_pos < input_len {
447                            let next = next_char_boundary(&input_buf[..input_len], cursor_pos);
448                            print_bytes(&input_buf[cursor_pos..next]);
449                            cursor_pos = next;
450                        }
451                        clear_visible_line(&input_buf[..input_len]);
452
453                        history_idx -= 1;
454                        if history_idx == -1 {
455                            let bytes = current_input_saved.as_bytes();
456                            let copy_len = bytes.len().min(input_buf.len());
457                            input_buf[..copy_len].copy_from_slice(&bytes[..copy_len]);
458                            input_len = copy_len;
459                        } else {
460                            let hist_str = &history[history.len() - 1 - history_idx as usize];
461                            let bytes = hist_str.as_bytes();
462                            let copy_len = bytes.len().min(input_buf.len());
463                            input_buf[..copy_len].copy_from_slice(&bytes[..copy_len]);
464                            input_len = copy_len;
465                        }
466                        cursor_pos = input_len;
467
468                        redraw_full_line(&input_buf[..input_len], cursor_pos);
469                    }
470                }
471                b'\x1b' => {
472                    utf8_pending_len = 0;
473                    in_escape_seq = true;
474                }
475                _ if in_escape_seq => {
476                    if (0x40..=0x7E).contains(&ch) {
477                        in_escape_seq = false;
478                    } else if ch == b'[' || ch == b';' || ch == b'?' || ch.is_ascii_digit() {
479                        // stay in escape sequence
480                    } else {
481                        in_escape_seq = false;
482                    }
483                }
484                _ if ch >= 0x20 => {
485                    in_escape_seq = false;
486                    if ch < 0x80 {
487                        utf8_pending_len = 0;
488                        if insert_bytes_at_cursor(
489                            &mut input_buf,
490                            &mut input_len,
491                            &mut cursor_pos,
492                            core::slice::from_ref(&ch),
493                        ) {
494                            history_idx = -1;
495                        }
496                    } else {
497                        if utf8_pending_len >= utf8_pending.len() {
498                            utf8_pending_len = 0;
499                        }
500                        utf8_pending[utf8_pending_len] = ch;
501                        utf8_pending_len += 1;
502                        match core::str::from_utf8(&utf8_pending[..utf8_pending_len]) {
503                            Ok(s) => {
504                                if insert_bytes_at_cursor(
505                                    &mut input_buf,
506                                    &mut input_len,
507                                    &mut cursor_pos,
508                                    s.as_bytes(),
509                                ) {
510                                    history_idx = -1;
511                                }
512                                utf8_pending_len = 0;
513                            }
514                            Err(err) => {
515                                if err.error_len().is_some() {
516                                    utf8_pending_len = 0;
517                                }
518                            }
519                        }
520                    }
521                }
522                _ => {
523                    in_escape_seq = false;
524                    utf8_pending_len = 0;
525                }
526            }
527            // Reset blink state on input
528            last_blink_tick = ticks / 50;
529            cursor_visible = true;
530        } else {
531            if crate::arch::x86_64::mouse::MOUSE_READY.load(core::sync::atomic::Ordering::Relaxed) {
532                let mut scroll_delta: i32 = 0;
533                let mut left_pressed = false;
534                let mut left_released = false;
535                let mut left_held = false;
536                let mut had_events = false;
537
538                let mut mouse_events_seen = 0usize;
539                while let Some(ev) = crate::arch::x86_64::mouse::read_event() {
540                    had_events = true;
541                    scroll_delta += ev.dz as i32;
542                    if ev.left && !prev_left {
543                        left_pressed = true;
544                    }
545                    if !ev.left && prev_left {
546                        left_released = true;
547                    }
548                    if ev.left && prev_left {
549                        left_held = true;
550                    }
551                    prev_left = ev.left;
552                    mouse_events_seen += 1;
553                    if mouse_events_seen >= MAX_MOUSE_EVENTS_PER_TURN {
554                        // Prevent monopolizing the CPU under heavy mouse input
555                        // (e.g. rapid drag on scrollbar). Remaining events are
556                        // processed on next loop iteration after yield_task().
557                        break;
558                    }
559                }
560
561                if had_events || left_held {
562                    let (new_mx, new_my) = crate::arch::x86_64::mouse::mouse_pos();
563                    let moved = new_mx != mouse_x || new_my != mouse_y;
564                    mouse_x = new_mx;
565                    mouse_y = new_my;
566
567                    if crate::arch::x86_64::vga::is_available() {
568                        // Inverted wheel: wheel up (dz>0) → scroll down (history forward)
569                        if scroll_delta > 0 {
570                            crate::arch::x86_64::vga::scroll_view_down((scroll_delta as usize) * 3);
571                        } else if scroll_delta < 0 {
572                            crate::arch::x86_64::vga::scroll_view_up((-scroll_delta as usize) * 3);
573                        }
574
575                        if left_pressed {
576                            let (mx, my) = (new_mx as usize, new_my as usize);
577                            if crate::arch::x86_64::vga::scrollbar_hit_test(mx, my) {
578                                crate::arch::x86_64::vga::scrollbar_click(mx, my);
579                                crate::arch::x86_64::vga::clear_selection();
580                                selecting = false;
581                                scrollbar_dragging = true;
582                            } else {
583                                crate::arch::x86_64::vga::start_selection(mx, my);
584                                selecting = true;
585                                scrollbar_dragging = false;
586                            }
587                        } else if left_held && scrollbar_dragging && moved {
588                            pending_scrollbar_drag_y = Some(new_my as usize);
589                            if ticks.saturating_sub(last_scrollbar_drag_tick)
590                                >= SCROLLBAR_DRAG_MIN_TICKS
591                            {
592                                if let Some(py) = pending_scrollbar_drag_y.take() {
593                                    crate::arch::x86_64::vga::scrollbar_drag_to(py);
594                                    last_scrollbar_drag_tick = ticks;
595                                }
596                            }
597                        } else if left_held && selecting && moved {
598                            crate::arch::x86_64::vga::update_selection(
599                                new_mx as usize,
600                                new_my as usize,
601                            );
602                        } else if left_released {
603                            if selecting {
604                                crate::arch::x86_64::vga::end_selection();
605                                selecting = false;
606                            }
607                            if scrollbar_dragging {
608                                if let Some(py) = pending_scrollbar_drag_y.take() {
609                                    crate::arch::x86_64::vga::scrollbar_drag_to(py);
610                                }
611                            }
612                            scrollbar_dragging = false;
613                        }
614
615                        if moved {
616                            crate::arch::x86_64::vga::update_mouse_cursor(new_mx, new_my);
617                        }
618                    }
619                }
620            }
621            crate::process::yield_task();
622        }
623    }
624}

626/// Execute a command line, handling scripting, pipes and redirections.
627fn execute_line(line: &str, registry: &CommandRegistry) {
628    let expanded = scripting::expand_vars(line);
629
630    match scripting::parse_script(&expanded) {
631        scripting::ScriptConstruct::SetVar { key, val } => {
632            let expanded_val = scripting::expand_vars(&val);
633            scripting::set_var(&key, &expanded_val);
634            scripting::set_last_exit(0);
635            return;
636        }
637        scripting::ScriptConstruct::UnsetVar(key) => {
638            scripting::unset_var(&key);
639            scripting::set_last_exit(0);
640            return;
641        }
642        scripting::ScriptConstruct::ForLoop { var, items, body } => {
643            for item in &items {
644                scripting::set_var(&var, item);
645                for cmd in &body {
646                    let exp = scripting::expand_vars(cmd);
647                    execute_pipeline(&exp, registry);
648                }
649            }
650            return;
651        }
652        scripting::ScriptConstruct::WhileLoop { cond, body } => {
653            let mut iters = 0u32;
654            loop {
655                if iters > 10000 || is_interrupted() {
656                    break;
657                }
658                let cond_expanded = scripting::expand_vars(&cond);
659                execute_pipeline(&cond_expanded, registry);
660                if scripting::last_exit() != 0 {
661                    break;
662                }
663                for cmd in &body {
664                    let exp = scripting::expand_vars(cmd);
665                    execute_pipeline(&exp, registry);
666                }
667                iters += 1;
668            }
669            return;
670        }
671        scripting::ScriptConstruct::IfElse { cond, then_body, else_body } => {
672            let cond_expanded = scripting::expand_vars(&cond);
673            execute_pipeline(&cond_expanded, registry);
674            let branch = if scripting::last_exit() == 0 {
675                &then_body
676            } else {
677                &else_body
678            };
679            for cmd in branch {
680                let exp = scripting::expand_vars(cmd);
681                execute_pipeline(&exp, registry);
682            }
683            return;
684        }
685        scripting::ScriptConstruct::Simple(s) => {
686            execute_pipeline(&s, registry);
687        }
688    }
689}

691/// Execute a single pipeline (no scripting).
692fn execute_pipeline(line: &str, registry: &CommandRegistry) {
693    // Ensure stale pipe input from a previous command cannot leak.
694    output::clear_pipe_input();
695
696    let pipeline = match parse_pipeline(line) {
697        Some(p) => p,
698        None => return,
699    };
700
701    let stage_count = pipeline.stages.len();
702    let mut pipe_data: Option<alloc::vec::Vec<u8>> = None;
703
704    for (i, stage) in pipeline.stages.iter().enumerate() {
705        let is_last = i == stage_count - 1;
706        let needs_capture = !is_last || stage.stdout_redirect.is_some();
707
708        if let Some(ref stdin_path) = stage.stdin_redirect {
709            match vfs::open(stdin_path, vfs::OpenFlags::READ) {
710                Ok(fd) => {
711                    let data = vfs::read_all(fd).unwrap_or_default();
712                    let _ = vfs::close(fd);
713                    output::set_pipe_input(data);
714                }
715                Err(e) => {
716                    shell_println!("shell: cannot open '{}': {:?}", stdin_path, e);
717                    return;
718                }
719            }
720        } else if let Some(data) = pipe_data.take() {
721            output::set_pipe_input(data);
722        }
723
724        if needs_capture {
725            output::start_capture();
726        }
727
728        let result = registry.execute(&stage.command);
729
730        let captured = if needs_capture {
731            output::take_capture()
732        } else {
733            alloc::vec::Vec::new()
734        };
735
736        match result {
737            Ok(()) => {
738                scripting::set_last_exit(0);
739            }
740            Err(ShellError::UnknownCommand) => {
741                scripting::set_last_exit(127);
742                shell_println!("Error: unknown command '{}'", stage.command.name);
743                return;
744            }
745            Err(ShellError::InvalidArguments) => {
746                scripting::set_last_exit(2);
747                shell_println!("Error: invalid arguments for '{}'", stage.command.name);
748                return;
749            }
750            Err(ShellError::ExecutionFailed) => {
751                scripting::set_last_exit(1);
752                shell_println!("Error: '{}' execution failed", stage.command.name);
753                return;
754            }
755        }
756
757        // A stage may ignore stdin pipe input; clear any leftovers before next stage.
758        output::clear_pipe_input();
759
760        if let Some(ref redirect) = stage.stdout_redirect {
761            apply_redirect(redirect, &captured);
762        }
763
764        if !is_last {
765            pipe_data = Some(captured);
766        }
767    }
768}

770/// Tab completion for command names and VFS paths.
771///
772/// If the cursor is on the first token, completes against registered commands.
773/// Otherwise completes against VFS directory entries.
774fn tab_complete(
775    input_buf: &mut [u8],
776    input_len: &mut usize,
777    cursor_pos: &mut usize,
778    registry: &CommandRegistry,
779) {
780    let text = match core::str::from_utf8(&input_buf[..*input_len]) {
781        Ok(s) => s,
782        Err(_) => return,
783    };
784
785    let before_cursor = &text[..*cursor_pos];
786    let has_space = before_cursor.contains(' ');
787
788    if !has_space {
789        let prefix = before_cursor;
790        let names = registry.command_names();
791        let matches: alloc::vec::Vec<&str> =
792            names.iter().copied().filter(|n| n.starts_with(prefix)).collect();
793
794        if matches.len() == 1 {
795            complete_replace_word(input_buf, input_len, cursor_pos, 0, matches[0], true);
796        } else if matches.len() > 1 {
797            let common = longest_common_prefix(&matches);
798            if common.len() > prefix.len() {
799                complete_replace_word(input_buf, input_len, cursor_pos, 0, &common, false);
800            } else {
801                shell_println!();
802                for m in &matches {
803                    shell_print!("{}  ", m);
804                }
805                shell_println!();
806                output::print_prompt();
807                print_bytes(&input_buf[..*input_len]);
808                let back = char_count(&input_buf[*cursor_pos..*input_len]);
809                move_cursor_left_chars(back);
810            }
811        }
812    } else {
813        let last_space = before_cursor.rfind(' ').unwrap_or(0);
814        let partial = &before_cursor[last_space + 1..];
815        let (dir, file_prefix) = if let Some(slash_pos) = partial.rfind('/') {
816            (&partial[..=slash_pos], &partial[slash_pos + 1..])
817        } else {
818            ("/", partial)
819        };
820
821        if let Ok(fd) = vfs::open(dir, OpenFlags::READ | OpenFlags::DIRECTORY) {
822            let entries = vfs::getdents(fd).unwrap_or_default();
823            let _ = vfs::close(fd);
824
825            let matches: alloc::vec::Vec<alloc::string::String> = entries
826                .iter()
827                .filter(|e| e.name != "." && e.name != ".." && e.name.starts_with(file_prefix))
828                .map(|e| {
829                    let mut s = alloc::string::String::from(dir);
830                    s.push_str(&e.name);
831                    if e.file_type == strat9_abi::data::DT_DIR {
832                        s.push('/');
833                    }
834                    s
835                })
836                .collect();
837
838            if matches.len() == 1 {
839                let add_space = !matches[0].ends_with('/');
840                complete_replace_word(
841                    input_buf,
842                    input_len,
843                    cursor_pos,
844                    last_space + 1,
845                    &matches[0],
846                    add_space,
847                );
848            } else if matches.len() > 1 {
849                let refs: alloc::vec::Vec<&str> = matches.iter().map(|s| s.as_str()).collect();
850                let common = longest_common_prefix(&refs);
851                if common.len() > partial.len() {
852                    complete_replace_word(
853                        input_buf,
854                        input_len,
855                        cursor_pos,
856                        last_space + 1,
857                        &common,
858                        false,
859                    );
860                } else {
861                    shell_println!();
862                    for m in &matches {
863                        let name = m.rsplit('/').next().unwrap_or(m);
864                        shell_print!("{}  ", name);
865                    }
866                    shell_println!();
867                    output::print_prompt();
868                    print_bytes(&input_buf[..*input_len]);
869                    let back = char_count(&input_buf[*cursor_pos..*input_len]);
870                    move_cursor_left_chars(back);
871                }
872            }
873        }
874    }
875}

877/// Replace the word starting at `word_start` (byte offset) with `replacement`.
878fn complete_replace_word(
879    buf: &mut [u8],
880    len: &mut usize,
881    cursor: &mut usize,
882    word_start: usize,
883    replacement: &str,
884    add_trailing_space: bool,
885) {
886    let mut new_line = alloc::string::String::new();
887    if let Ok(prefix) = core::str::from_utf8(&buf[..word_start]) {
888        new_line.push_str(prefix);
889    }
890    new_line.push_str(replacement);
891    if add_trailing_space {
892        new_line.push(' ');
893    }
894    let new_cursor = new_line.len();
895    if let Ok(suffix) = core::str::from_utf8(&buf[*cursor..*len]) {
896        new_line.push_str(suffix);
897    }
898
899    let bytes = new_line.as_bytes();
900    if bytes.len() > buf.len() {
901        return;
902    }
903
904    let old_visible = char_count(&buf[..*len]);
905    move_cursor_left_chars(char_count(&buf[..*cursor]));
906
907    buf[..bytes.len()].copy_from_slice(bytes);
908    *len = bytes.len();
909    *cursor = new_cursor;
910
911    for _ in 0..old_visible {
912        print_char(' ');
913    }
914    move_cursor_left_chars(old_visible);
915    print_bytes(&buf[..*len]);
916    let back = char_count(&buf[*cursor..*len]);
917    move_cursor_left_chars(back);
918}

920/// Find the longest common prefix of a set of strings.
921fn longest_common_prefix(strings: &[&str]) -> alloc::string::String {
922    if strings.is_empty() {
923        return alloc::string::String::new();
924    }
925    let first = strings[0];
926    let mut end = first.len();
927    for s in &strings[1..] {
928        end = end.min(s.len());
929        for (i, (a, b)) in first.bytes().zip(s.bytes()).enumerate() {
930            if a != b {
931                end = end.min(i);
932                break;
933            }
934        }
935    }
936    alloc::string::String::from(&first[..end])
937}

939/// Write captured output to a file (truncate or append).
940fn apply_redirect(redirect: &Redirect, data: &[u8]) {
941    match redirect {
942        Redirect::Truncate(path) => {
943            let flags = OpenFlags::WRITE | OpenFlags::CREATE | OpenFlags::TRUNCATE;
944            match vfs::open(path, flags) {
945                Ok(fd) => {
946                    let _ = vfs::write(fd, data);
947                    let _ = vfs::close(fd);
948                }
949                Err(e) => shell_println!("shell: cannot write '{}': {:?}", path, e),
950            }
951        }
952        Redirect::Append(path) => {
953            let flags = OpenFlags::WRITE | OpenFlags::CREATE | OpenFlags::APPEND;
954            match vfs::open(path, flags) {
955                Ok(fd) => {
956                    let _ = vfs::write(fd, data);
957                    let _ = vfs::close(fd);
958                }
959                Err(e) => shell_println!("shell: cannot append '{}': {:?}", path, e),
960            }
961        }
962    }
