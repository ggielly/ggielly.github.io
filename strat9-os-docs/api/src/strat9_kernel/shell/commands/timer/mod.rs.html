1//! Timer debug commands
2//!
3//! Provides shell commands for debugging and verifying timer accuracy.
4
5use crate::{arch::x86_64::timer::TIMER_HZ, shell::ShellError, shell_println};
6use alloc::string::String;
7
8/// Timer debug command
9///
10/// Usage: timer test <seconds>
11///
12/// Measures elapsed real time vs kernel ticks.
13pub fn cmd_timer(args: &[String]) -> Result<(), ShellError> {
14    if args.is_empty() {
15        print_usage();
16        return Ok(());
17    }
18
19    match args[0].as_str() {
20        "test" => {
21            let seconds = if args.len() > 1 {
22                args[1].parse::<u32>().unwrap_or(1)
23            } else {
24                1 // Default: 1 second
25            };
26
27            shell_println!("Testing timer accuracy for {} seconds...", seconds);
28            shell_println!("Please compare with a real clock/stopwatch.");
29            shell_println!("Serial output will show detailed timing info.");
30
31            // Call the debug function
32            crate::arch::x86_64::timer::debug_measure_time(seconds);
33
34            Ok(())
35        }
36        "info" => {
37            shell_println!("=== Timer Information ===");
38            shell_println!("Scheduler ticks: {}", crate::process::scheduler::ticks());
39
40            if crate::arch::x86_64::timer::is_apic_timer_active() {
41                let ticks_per_10ms = crate::arch::x86_64::timer::apic_ticks_per_10ms();
42                shell_println!("Timer mode: APIC");
43                shell_println!("Ticks per 10ms: {}", ticks_per_10ms);
44                shell_println!(
45                    "Estimated CPU: {} MHz",
46                    (ticks_per_10ms as u64) * 16 * 100 / 1_000_000
47                );
48            } else {
49                shell_println!("Timer mode: PIT (fallback)");
50                shell_println!("Frequency: {} Hz", TIMER_HZ);
51            }
52
53            Ok(())
54        }
55        _ => {
56            print_usage();
57            Ok(())
58        }
59    }
60}
61
62fn print_usage() {
63    shell_println!("Timer debug commands:");
64    shell_println!("  timer test [seconds]  - Measure timer accuracy (default: 1s)");
65    shell_println!("  timer info            - Show timer configuration");
66    shell_println!("");
67    shell_println!("Examples:");
68    shell_println!("  timer test          - Test for 1 second");
69    shell_println!("  timer test 5        - Test for 5 seconds");
70    shell_println!("  timer info          - Show current timer info");
71}
