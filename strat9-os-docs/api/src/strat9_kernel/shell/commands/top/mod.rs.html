73/// Performs the collect snapshot operation.
74fn collect_snapshot() -> TopSnapshot {
75    let cpu_count = crate::arch::x86_64::percpu::cpu_count();
76    let (total_pages, used_pages) = {
77        let guard = crate::memory::buddy::get_allocator().lock();
78        guard.as_ref().map(|a| a.page_totals()).unwrap_or((0, 0))
79    };
80
81    let mut tasks = Vec::new();
82    if let Some(all_tasks) = crate::process::get_all_tasks() {
83        for task in all_tasks {
84            let state = unsafe { *task.state.get() };
85            let state_str = match state {
86                crate::process::TaskState::Ready => "Ready",
87                crate::process::TaskState::Running => "Running",
88                crate::process::TaskState::Blocked => "Blocked",
89                crate::process::TaskState::Dead => "Dead",
90            };
91            tasks.push(TaskRowData {
92                pid: format!("{}", task.pid),
93                name: String::from(task.name),
94                state: state_str,
95                priority: format!("{:?}", task.priority),
96                ticks: task.ticks.load(Ordering::Relaxed),
97            });
98        }
99    }
100
101    // Top-like behavior: most CPU-consumed tasks first.
102    tasks.sort_by(|a, b| b.ticks.cmp(&a.ticks));
103
104    let mut silos = Vec::new();
105    let mut strate_index: Vec<(String, Vec<String>)> = Vec::new();
106    let mut silo_snapshots = crate::silo::list_silos_snapshot();
107    silo_snapshots.sort_by_key(|s| s.id);
108    for s in silo_snapshots {
109        let label = s.strate_label.unwrap_or_default();
110        let strate_name = if !label.is_empty() {
111            label.clone()
112        } else {
113            s.name.clone()
114        };
115        if let Some((_, belongs)) = strate_index
116            .iter_mut()
117            .find(|(name, _)| *name == strate_name)
118        {
119            if !belongs.iter().any(|x| x == &s.name) {
120                belongs.push(s.name.clone());
121            }
122        } else {
123            strate_index.push((strate_name, vec![s.name.clone()]));
124        }
125        silos.push(SiloRowData {
126            sid: format!("{}", s.id),
127            name: s.name,
128            state: format!("{:?}", s.state),
129            tasks: format!("{}", s.task_count),
130            label: if label.is_empty() {
131                String::from("-")
132            } else {
133                label
134            },
135        });
136    }
137
138    strate_index.sort_by(|a, b| a.0.cmp(&b.0));
139    let mut strates = Vec::with_capacity(strate_index.len());
140    for (name, belongs) in strate_index {
141        let mut silos_csv = String::new();
142        for (i, silo_name) in belongs.iter().enumerate() {
143            if i != 0 {
144                silos_csv.push_str(", ");
145            }
146            silos_csv.push_str(silo_name);
147        }
148        strates.push(StrateRowData {
149            name,
150            silos: silos_csv,
151        });
152    }
153
154    TopSnapshot {
155        cpu_count,
156        total_pages,
157        used_pages,
158        tasks,
159        silos,
160        strates,
161        scheduler: crate::process::scheduler_state_snapshot(),
162    }
163}

165/// Performs the compute cpu usage window operation.
166fn compute_cpu_usage_window(
167    prev: &crate::process::CpuUsageSnapshot,
168    now: &crate::process::CpuUsageSnapshot,
169) -> CpuUsageWindow {
170    let cpu_count = now.cpu_count.min(crate::arch::x86_64::percpu::MAX_CPUS);
171    let mut ratios = [0.0f64; crate::arch::x86_64::percpu::MAX_CPUS];
172    let mut sum = 0.0;
173
174    for i in 0..cpu_count {
175        let delta_total = now.total_ticks[i].saturating_sub(prev.total_ticks[i]);
176        let delta_idle = now.idle_ticks[i].saturating_sub(prev.idle_ticks[i]);
177        let ratio = if delta_total == 0 {
178            0.0
179        } else {
180            let busy = delta_total.saturating_sub(delta_idle);
181            (busy as f64 / delta_total as f64).clamp(0.0, 1.0)
182        };
183        ratios[i] = ratio;
184        sum += ratio;
185    }
186
187    CpuUsageWindow {
188        per_cpu_ratio: ratios,
189        avg_ratio: if cpu_count == 0 {
190            0.0
191        } else {
192            (sum / cpu_count as f64).clamp(0.0, 1.0)
193        },
194    }
195}

197/// Performs the compute scheduler metrics window operation.
198fn compute_scheduler_metrics_window(
199    prev: &crate::process::SchedulerMetricsSnapshot,
200    now: &crate::process::SchedulerMetricsSnapshot,
201) -> SchedulerMetricsWindow {
202    let cpu_count = now.cpu_count.min(crate::arch::x86_64::percpu::MAX_CPUS);
203    let mut rt_delta = 0u64;
204    let mut fair_delta = 0u64;
205    let mut idle_delta = 0u64;
206    let mut switch_delta = 0u64;
207    let mut preempt_delta = 0u64;
208    let mut steal_in_delta = 0u64;
209    let mut steal_out_delta = 0u64;
210    for i in 0..cpu_count {
211        rt_delta = rt_delta
212            .saturating_add(now.rt_runtime_ticks[i].saturating_sub(prev.rt_runtime_ticks[i]));
213        fair_delta = fair_delta
214            .saturating_add(now.fair_runtime_ticks[i].saturating_sub(prev.fair_runtime_ticks[i]));
215        idle_delta = idle_delta
216            .saturating_add(now.idle_runtime_ticks[i].saturating_sub(prev.idle_runtime_ticks[i]));
217        switch_delta =
218            switch_delta.saturating_add(now.switch_count[i].saturating_sub(prev.switch_count[i]));
219        preempt_delta = preempt_delta
220            .saturating_add(now.preempt_count[i].saturating_sub(prev.preempt_count[i]));
221        steal_in_delta = steal_in_delta
222            .saturating_add(now.steal_in_count[i].saturating_sub(prev.steal_in_count[i]));
223        steal_out_delta = steal_out_delta
224            .saturating_add(now.steal_out_count[i].saturating_sub(prev.steal_out_count[i]));
225    }
226    let total = rt_delta
227        .saturating_add(fair_delta)
228        .saturating_add(idle_delta);
229    let to_ratio = |v: u64| {
230        if total == 0 {
231            0.0
232        } else {
233            (v as f64 / total as f64).clamp(0.0, 1.0)
234        }
235    };
236    SchedulerMetricsWindow {
237        rt_ratio: to_ratio(rt_delta),
238        fair_ratio: to_ratio(fair_delta),
239        idle_ratio: to_ratio(idle_delta),
240        switch_delta,
241        preempt_delta,
242        steal_in_delta,
243        steal_out_delta,
244    }
245}

247/// Performs the scheduler runtime lines operation.
248fn scheduler_runtime_lines(
249    s: &crate::process::SchedulerStateSnapshot,
250    w: &SchedulerMetricsWindow,
251) -> (String, String, String) {
252    let line1 = format!(
253        "Win: RT {:>3}% | FAIR {:>3}% | IDLE {:>3}% | sw {} | pre {} | st+ {} | st- {}",
254        (w.rt_ratio * 100.0) as u16,
255        (w.fair_ratio * 100.0) as u16,
256        (w.idle_ratio * 100.0) as u16,
257        w.switch_delta,
258        w.preempt_delta,
259        w.steal_in_delta,
260        w.steal_out_delta
261    );
262    let line2 = format!(
263        "Cfg: init={} blocked={} pick=[{},{},{}] steal=[{},{}]",
264        s.initialized,
265        s.blocked_tasks,
266        s.pick_order[0].as_str(),
267        s.pick_order[1].as_str(),
268        s.pick_order[2].as_str(),
269        s.steal_order[0].as_str(),
270        s.steal_order[1].as_str()
271    );
272    let cpu_count = s.cpu_count.min(crate::arch::x86_64::percpu::MAX_CPUS);
273    let line3 = if cpu_count == 0 {
274        String::from("CPU: n/a")
275    } else {
276        let c0 = format!(
277            "cpu0 cur={} rq={}/{}/{} nr={}",
278            s.current_task[0], s.rq_rt[0], s.rq_fair[0], s.rq_idle[0], s.need_resched[0]
279        );
280        if cpu_count == 1 {
281            format!("CPU: {}", c0)
282        } else {
283            let c1 = format!(
284                "cpu1 cur={} rq={}/{}/{} nr={}",
285                s.current_task[1], s.rq_rt[1], s.rq_fair[1], s.rq_idle[1], s.need_resched[1]
286            );
287            format!("CPU: {} | {}", c0, c1)
288        }
289    };
290    (line1, line2, line3)
291}

293/// Top command main loop
294pub fn cmd_top(_args: &[alloc::string::String]) -> Result<(), ShellError> {
295    if !vga::is_available() {
296        shell_println!("Error: 'top' requires a graphical framebuffer console.");
297        return Ok(());
298    }
299
300    // Switch to double buffering for flicker-free updates.
301    let was_db = vga::double_buffer_mode();
302    vga::set_double_buffer_mode(true);
303    let backend = Strat9RatatuiBackend::new().map_err(|_| ShellError::ExecutionFailed)?;
304    let mut terminal = Terminal::new(backend).map_err(|_| ShellError::ExecutionFailed)?;
305    terminal.clear().map_err(|_| ShellError::ExecutionFailed)?;
306
307    let mut last_refresh_tick = crate::process::scheduler::ticks();
308    let boot_tick = last_refresh_tick;
309    let mut prev_cpu_sample = crate::process::cpu_usage_snapshot();
310    let mut prev_sched_sample = crate::process::scheduler_metrics_snapshot();
311    let mut selected_task: usize = 0;
312
313    loop {
314        let ticks = crate::process::scheduler::ticks();
315
316        // Keep input responsive even between render ticks.
317        if let Some(ch) = crate::arch::x86_64::keyboard::read_char() {
318            match ch {
319                b'q' | 0x1B | 0x03 => break,
320                crate::arch::x86_64::keyboard::KEY_UP => {
321                    selected_task = selected_task.saturating_sub(1);
322                }
323                crate::arch::x86_64::keyboard::KEY_DOWN => {
324                    selected_task = selected_task.saturating_add(1);
325                }
326                _ => {}
327            }
328        }
329
330        // Refresh every 100ms with a 100Hz timer.
331        if ticks.saturating_sub(last_refresh_tick) < TOP_REFRESH_TICKS {
332            crate::process::yield_task();
333            continue;
334        }
335        last_refresh_tick = ticks;
336        let snapshot = collect_snapshot();
337        let cpu_sample = crate::process::cpu_usage_snapshot();
338        let cpu_window = compute_cpu_usage_window(&prev_cpu_sample, &cpu_sample);
339        prev_cpu_sample = cpu_sample;
340        let sched_sample = crate::process::scheduler_metrics_snapshot();
341        let sched_window = compute_scheduler_metrics_window(&prev_sched_sample, &sched_sample);
342        prev_sched_sample = sched_sample;
343        let mem_ratio = if snapshot.total_pages > 0 {
344            (snapshot.used_pages as f64) / (snapshot.total_pages as f64)
345        } else {
346            0.0
347        };
348
349        let rows: Vec<Row> = snapshot
350            .tasks
351            .iter()
352            .map(|task| {
353                Row::new(vec![
354                    Cell::from(task.pid.as_str()),
355                    Cell::from(task.name.as_str()),
356                    Cell::from(task.state),
357                    Cell::from(task.priority.as_str()),
358                    Cell::from(format!("{}", task.ticks)),
359                ])
360            })
361            .collect();
362        let row_count = rows.len();
363        if row_count == 0 {
364            selected_task = 0;
365        } else if selected_task >= row_count {
366            selected_task = row_count - 1;
367        }
368        let mut table_state = TableState::default();
369        if row_count > 0 {
370            table_state.select(Some(selected_task));
371        }
372
373        let uptime_secs = ticks.saturating_sub(boot_tick) / 100;
374
375        let frame_started = vga::begin_frame();
376        terminal
377            .draw(|frame| {
378                let title_style = Style::default().fg(Color::Cyan).add_modifier(Modifier::BOLD);
379                let primary_text = Style::default().fg(Color::White);
380                let muted_text = Style::default().fg(Color::Gray);
381                let header_style = Style::default().fg(Color::Yellow).add_modifier(Modifier::BOLD);
382
383                let area = frame.area();
384                let vertical = Layout::default()
385                    .direction(Direction::Vertical)
386                    .constraints([
387                        Constraint::Length(2),
388                        Constraint::Length(3),
389                        Constraint::Length(5),
390                        Constraint::Length(6),
391                        Constraint::Min(8),
392                        Constraint::Length(1),
393                    ])
394                    .split(area);
395
396                let title = Paragraph::new("Strat9 System Monitor (Ratatui no_std)")
397                    .style(title_style)
398                    .block(Block::default().borders(Borders::BOTTOM).title("Top"));
399                frame.render_widget(title, vertical[0]);
400
401                let stats_line = Paragraph::new(format!(
402                    "CPUs: {} | Tasks: {} | Silos: {} | Strates: {} | CPU(avg): {:>3}% | Uptime: {}s",
403                    snapshot.cpu_count,
404                    snapshot.tasks.len(),
405                    snapshot.silos.len(),
406                    snapshot.strates.len(),
407                    (cpu_window.avg_ratio * 100.0) as u16,
408                    uptime_secs
409                ))
410                .style(primary_text)
411                .block(Block::default().borders(Borders::BOTTOM).title("Stats"));
412                frame.render_widget(stats_line, vertical[1]);
413
414                let (sched_line1, sched_line2, sched_line3) =
415                    scheduler_runtime_lines(&snapshot.scheduler, &sched_window);
416                let sched_line = Paragraph::new(format!(
417                    "{}\n{}\n{}",
418                    sched_line1, sched_line2, sched_line3
419                ))
420                .style(primary_text)
421                .block(Block::default().borders(Borders::BOTTOM).title("Scheduler"));
422                frame.render_widget(sched_line, vertical[2]);
423
424                let cpu_split = Layout::default()
425                    .direction(Direction::Horizontal)
426                    .constraints([Constraint::Percentage(50), Constraint::Percentage(50)])
427                    .split(vertical[3]);
428
429                let mem_gauge = Gauge::default()
430                    .block(
431                        Block::default()
432                            .borders(Borders::TOP | Borders::BOTTOM)
433                            .title(format!(
434                                "Memory {} / {} pages",
435                                snapshot.used_pages, snapshot.total_pages
436                            )),
437                    )
438                    .gauge_style(Style::default().fg(Color::Blue))
439                    .use_unicode(false)
440                    .ratio(mem_ratio.clamp(0.0, 1.0))
441                    .label(format!("{:.1}%", mem_ratio * 100.0));
442                frame.render_widget(mem_gauge, cpu_split[0]);
443
444                let cpu_gauge_count = snapshot.cpu_count.min(MAX_CPU_GAUGES);
445                if cpu_gauge_count > 0 {
446                    let mut constraints = Vec::with_capacity(cpu_gauge_count);
447                    for _ in 0..cpu_gauge_count {
448                        constraints.push(Constraint::Length(1));
449                    }
450                    let cpu_rows = Layout::default()
451                        .direction(Direction::Vertical)
452                        .constraints(constraints)
453                        .split(cpu_split[1]);
454
455                    for i in 0..cpu_gauge_count {
456                        let ratio = cpu_window.per_cpu_ratio[i];
457                        let gauge = Gauge::default()
458                            .block(Block::default().title(format!("CPU{}", i)).borders(Borders::NONE))
459                            .gauge_style(Style::default().fg(Color::Green))
460                            .use_unicode(false)
461                            .ratio(ratio)
462                            .label(format!("{:>3}%", (ratio * 100.0) as u16));
463                        frame.render_widget(gauge, cpu_rows[i]);
464                    }
465                }
466
467                let main_split = Layout::default()
468                    .direction(Direction::Horizontal)
469                    .constraints([Constraint::Percentage(64), Constraint::Percentage(36)])
470                    .split(vertical[4]);
471
472                let task_table = Table::new(
473                    rows.iter().cloned(),
474                    [
475                        Constraint::Length(5),  // PID
476                        Constraint::Min(18),    // Name (takes remaining width)
477                        Constraint::Length(9),  // State
478                        Constraint::Length(8),  // Prio
479                        Constraint::Length(10), // Ticks
480                    ],
481                )
482                .header(
483                    Row::new(vec!["PID", "Name", "State", "Prio", "Ticks"]).style(header_style),
484                )
485                .column_spacing(1)
486                .style(primary_text)
487                .row_highlight_style(
488                    Style::default()
489                        .bg(Color::White)
490                        .fg(Color::Black)
491                        .add_modifier(Modifier::BOLD),
492                )
493                .block(
494                    Block::default()
495                        .borders(Borders::TOP)
496                        .title("Tasks (sorted by ticks)"),
497                );
498                frame.render_stateful_widget(task_table, main_split[0], &mut table_state);
499
500                let right_split = Layout::default()
501                    .direction(Direction::Vertical)
502                    .constraints([Constraint::Percentage(52), Constraint::Percentage(48)])
503                    .split(main_split[1]);
504
505                let silo_rows: Vec<Row> = snapshot
506                    .silos
507                    .iter()
508                    .map(|s| {
509                        Row::new(vec![
510                            Cell::from(s.sid.as_str()),
511                            Cell::from(s.name.as_str()),
512                            Cell::from(s.state.as_str()),
513                            Cell::from(s.tasks.as_str()),
514                            Cell::from(s.label.as_str()),
515                        ])
516                    })
517                    .collect();
518                let silo_table = Table::new(
519                    silo_rows,
520                    [
521                        Constraint::Length(5),
522                        Constraint::Length(10),
523                        Constraint::Length(8),
524                        Constraint::Length(5),
525                        Constraint::Min(8),
526                    ],
527                )
528                .header(
529                    Row::new(vec!["SID", "Name", "State", "T", "Label"]).style(
530                        Style::default().fg(Color::LightGreen).add_modifier(Modifier::BOLD),
531                    ),
532                )
533                .column_spacing(1)
534                .style(primary_text)
535                .block(Block::default().borders(Borders::TOP).title("Silos"));
536                frame.render_widget(silo_table, right_split[0]);
537
538                let strate_rows: Vec<Row> = snapshot
539                    .strates
540                    .iter()
541                    .map(|s| Row::new(vec![Cell::from(s.name.as_str()), Cell::from(s.silos.as_str())]))
542                    .collect();
543                let strate_table = Table::new(
544                    strate_rows,
545                    [Constraint::Length(12), Constraint::Min(10)],
546                )
547                .header(
548                    Row::new(vec!["Strate", "BelongsTo"]).style(
549                        Style::default().fg(Color::LightCyan).add_modifier(Modifier::BOLD),
550                    ),
551                )
552                .column_spacing(1)
553                .style(primary_text)
554                .block(Block::default().borders(Borders::TOP).title("Strates"));
555                frame.render_widget(strate_table, right_split[1]);
556
557                let footer = Paragraph::new("[Up/Down] Select process | [q|Esc] Exit")
558                    .style(muted_text)
559                    .block(Block::default().borders(Borders::TOP));
560                frame.render_widget(footer, vertical[5]);
561            })
562            .map_err(|_| ShellError::ExecutionFailed)?;
563
564        if frame_started {
565            vga::end_frame();
566        } else {
567            vga::present();
568        }
569
570        crate::process::yield_task();
571    }
572
573    // Clean exit.
574    vga::set_double_buffer_mode(was_db);
575    crate::shell::output::clear_screen();
576    vga::set_text_cursor(0, 0);
577    shell_println!("Top exited.");
578    Ok(())
