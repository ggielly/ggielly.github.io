1//! Procfs - process filesystem (similar to Linux & BSD /proc)
2//!
3//! Provides information about running processes and system state.
4//!
5//! # Structure
6//!
7//! ```text
8//! /proc/
9//!   self/       -> symlink to current process
10//!   <pid>/      -> process directory
11//!     status    -> process status
12//!     cmdline   -> command line
13//!     fd/       -> file descriptors
14//!   cpuinfo     -> CPU information
15//!   meminfo     -> memory information
16//!   version     -> kernel version
17//! ```
18

41    /// Get procfs entry content
42    fn get_entry(&self, path: &str) -> Result<ProcEntry, SyscallError> {
43        // Handle /proc/self
44        if path == "self" || path == "self/" {
45            if let Some(pid) = current_pid() {
46                return Ok(ProcEntry::File(format!("{}\n", pid)));
47            }
48            return Err(SyscallError::NotFound);
49        }
50
51        // Handle /proc/self/status
52        if path.starts_with("self/") {
53            let subpath = &path[5..];
54            if let Some(pid) = current_pid() {
55                return self.get_process_entry(pid as u64, subpath);
56            }
57            return Err(SyscallError::NotFound);
58        }
59
60        // Handle /proc/<pid>/...
61        if let Some(pid_end) = path.find('/') {
62            let pid_str = &path[..pid_end];
63            let subpath = &path[pid_end + 1..];
64            if let Ok(pid) = pid_str.parse::<u64>() {
65                return self.get_process_entry(pid, subpath);
66            }
67        }
68
69        // Handle /proc/<pid> (directory listing)
70        if path.parse::<u64>().is_ok() {
71            return Ok(ProcEntry::Directory);
72        }
73
74        // Handle root entries
75        match path {
76            "" | "/" => Ok(ProcEntry::Directory),
77            "cpuinfo" => Ok(ProcEntry::File(self.get_cpuinfo())),
78            "meminfo" => Ok(ProcEntry::File(self.get_meminfo())),
79            "version" => Ok(ProcEntry::File(self.get_version())),
80            "silos" => Ok(ProcEntry::File(self.get_silos())),
81            _ => Err(SyscallError::NotFound),
82        }
83    }

85    /// Get process-specific entry
86    fn get_process_entry(&self, pid: u64, subpath: &str) -> Result<ProcEntry, SyscallError> {
87        let tasks = get_all_tasks().ok_or(SyscallError::NotFound)?;
88        let task = tasks
89            .iter()
90            .find(|t| t.pid as u64 == pid)
91            .ok_or(SyscallError::NotFound)?;
92
93        match subpath {
94            "" | "/" => Ok(ProcEntry::Directory),
95            "status" => Ok(ProcEntry::File(self.get_process_status(task))),
96            "cmdline" => Ok(ProcEntry::File(format!("{}\n", task.name))),
97            _ => Err(SyscallError::NotFound),
98        }
99    }

156    /// Generate /proc/silos content
157    fn get_silos(&self) -> String {
158        let mut output = String::new();
159        let mut silos = silo::list_silos_snapshot();
160        silos.sort_by_key(|s| s.id);
161        let _ = writeln!(
162            output,
163            "id\tstate\ttasks\tmem_used\tmem_min\tmem_max\tlabel\tname"
164        );
165        for s in silos {
166            let label = s.strate_label.unwrap_or_else(|| String::from("-"));
167            let max = if s.mem_max_bytes == 0 {
168                String::from("unlimited")
169            } else {
170                format!("{}", s.mem_max_bytes)
171            };
172            let _ = writeln!(
173                output,
174                "{}\t{:?}\t{}\t{}\t{}\t{}\t{}\t{}",
175                s.id,
176                s.state,
177                s.task_count,
178                s.mem_usage_bytes,
179                s.mem_min_bytes,
180                max,
181                label,
182                s.name
183            );
184        }
185        output
186    }

188    /// Generate process status
189    fn get_process_status(&self, task: &Arc<crate::process::task::Task>) -> String {
190        let mut output = String::new();
191
192        let _ = writeln!(output, "Name:\t{}", task.name);
193        let _ = writeln!(output, "State:\tR (running)");
194        let _ = writeln!(output, "Tgid:\t{}", task.tgid);
195        let _ = writeln!(output, "Pid:\t{}", task.pid);
196        let _ = writeln!(
197            output,
198            "PPid:\t{}",
199            get_parent_pid(task.id).map(|p| p as u64).unwrap_or(0)
200        );
201        let _ = writeln!(
202            output,
203            "Pgid:\t{}",
204            task.pgid.load(core::sync::atomic::Ordering::Relaxed)
205        );
206        let _ = writeln!(
207            output,
208            "Sid:\t{}",
209            task.sid.load(core::sync::atomic::Ordering::Relaxed)
210        );
211        let uid = task.uid.load(core::sync::atomic::Ordering::Relaxed);
212        let euid = task.euid.load(core::sync::atomic::Ordering::Relaxed);
213        let gid = task.gid.load(core::sync::atomic::Ordering::Relaxed);
214        let egid = task.egid.load(core::sync::atomic::Ordering::Relaxed);
215        let _ = writeln!(output, "Uid:\t{}\t{}\t{}\t{}", uid, euid, euid, euid);
216        let _ = writeln!(output, "Gid:\t{}\t{}\t{}\t{}", gid, egid, egid, egid);
217        let _ = writeln!(output, "Threads:\t1");
218        if let Some((silo_id, label, mem_used, mem_min, mem_max)) = silo::silo_info_for_task(task.id) {
219            let label = label.unwrap_or_else(|| String::from("-"));
220            let _ = writeln!(output, "SiloId:\t{}", silo_id);
221            let _ = writeln!(output, "SiloLabel:\t{}", label);
222            let _ = writeln!(output, "SiloMemUsed:\t{}", mem_used);
223            let _ = writeln!(output, "SiloMemMin:\t{}", mem_min);
224            if mem_max == 0 {
225                let _ = writeln!(output, "SiloMemMax:\tunlimited");
226            } else {
227                let _ = writeln!(output, "SiloMemMax:\t{}", mem_max);
228            }
229        }
230
231        output
232    }

235/// Procfs entry type
236enum ProcEntry {
237    File(String),
238    Directory,
239}
240
241/// Kind constants for procfs file-id encoding.
242/// Fixed IDs (0, 1, 10, 11, 12) use kind = 0 and are stored as-is.
243const KIND_PROC_DIR: u64 = 1; // /proc/<pid>  directory
244const KIND_PROC_STATUS: u64 = 2; // /proc/<pid>/status
245const KIND_PROC_CMDLINE: u64 = 3; // /proc/<pid>/cmdline
246
247impl ProcScheme {
248    /// Encode a (kind, pid) pair into a file_id.
249    ///
250    /// The high 32 bits hold the kind; the low 32 bits hold the pid.
251    /// This guarantees no collision between entries regardless of PID value.
252    fn encode_id(kind: u64, pid: u64) -> u64 {
253        (kind << 32) | (pid & 0xFFFF_FFFF)
254    }
255
256    /// Decode a file_id back into (kind, pid).
257    fn decode_id(id: u64) -> (u64, u64) {
258        (id >> 32, id & 0xFFFF_FFFF)
259    }
260}

262impl Scheme for ProcScheme {
263    fn open(&self, path: &str, _flags: OpenFlags) -> Result<OpenResult, SyscallError> {
264        let entry = self.get_entry(path)?;
265
266        let (flags, file_id) = match entry {
267            ProcEntry::File(_) => {
268                let id = match path {
269                    "cpuinfo" => 10,
270                    "meminfo" => 11,
271                    "version" => 12,
272                    "silos" => 13,
273                    "self/status" => Self::encode_id(
274                        KIND_PROC_STATUS,
275                        current_pid().map(|p| p as u64).unwrap_or(0),
276                    ),
277                    "self/cmdline" => Self::encode_id(
278                        KIND_PROC_CMDLINE,
279                        current_pid().map(|p| p as u64).unwrap_or(0),
280                    ),
281                    _ if path.starts_with("self") => 1,
282                    _ => {
283                        if let Some(slash_idx) = path.find('/') {
284                            let pid_str = &path[..slash_idx];
285                            let sub = &path[slash_idx + 1..];
286                            if let Ok(pid) = pid_str.parse::<u64>() {
287                                match sub {
288                                    "status" => Self::encode_id(KIND_PROC_STATUS, pid),
289                                    "cmdline" => Self::encode_id(KIND_PROC_CMDLINE, pid),
290                                    _ => 0xFFFF,
291                                }
292                            } else {
293                                0xFFFF
294                            }
295                        } else {
296                            0xFFFF
297                        }
298                    }
299                };
300                (FileFlags::empty(), id)
301            }
302            ProcEntry::Directory => {
303                let id = if path.is_empty() || path == "/" {
304                    0
305                } else if path == "self" || path == "self/" {
306                    1
307                } else if let Ok(pid) = path.parse::<u64>() {
308                    Self::encode_id(KIND_PROC_DIR, pid)
309                } else {
310                    2
311                };
312                (FileFlags::DIRECTORY, id)
313            }
314        };
315
316        if file_id == 0xFFFF {
317            return Err(SyscallError::NotFound);
318        }
319
320        Ok(OpenResult {
321            file_id,
322            size: None,
323            flags,
324        })
325    }

327    fn read(&self, file_id: u64, offset: u64, buf: &mut [u8]) -> Result<usize, SyscallError> {
328        let (kind, pid) = Self::decode_id(file_id);
329        let content = if file_id == 0 {
330            // Root directory listing
331            let mut list = String::from("self\ncpuinfo\nmeminfo\nversion\n");
332            if let Some(tasks) = get_all_tasks() {
333                for task in tasks {
334                    let _ = writeln!(list, "{}", task.pid);
335                }
336            }
337            list
338        } else if file_id == 1 || kind == KIND_PROC_DIR {
339            // Process directory listing (self dir or /proc/<pid>)
340            String::from("status\ncmdline\n")
341        } else {
342            // File content — distinguish by kind (high 32 bits) for process files,
343            // or by the literal file_id for well-known fixed files.
344            match (kind, file_id) {
345                (0, 10) => self.get_cpuinfo(),
346                (0, 11) => self.get_meminfo(),
347                (0, 12) => self.get_version(),
348                (0, 13) => self.get_silos(),
349                (KIND_PROC_STATUS, _) => {
350                    let tasks = get_all_tasks().ok_or(SyscallError::NotFound)?;
351                    let task = tasks
352                        .iter()
353                        .find(|t| t.pid as u64 == pid)
354                        .ok_or(SyscallError::NotFound)?;
355                    self.get_process_status(task)
356                }
357                (KIND_PROC_CMDLINE, _) => {
358                    let tasks = get_all_tasks().ok_or(SyscallError::NotFound)?;
359                    let task = tasks
360                        .iter()
361                        .find(|t| t.pid as u64 == pid)
362                        .ok_or(SyscallError::NotFound)?;
363                    format!("{}\n", task.name)
364                }
365                _ => return Err(SyscallError::IoError),
366            }
367        };
368
369        if offset >= content.len() as u64 {
370            return Ok(0);
371        }
372
373        let start = offset as usize;
374        let end = core::cmp::min(start + buf.len(), content.len());
375        let to_copy = end - start;
376
377        buf[..to_copy].copy_from_slice(&content.as_bytes()[start..end]);
378        Ok(to_copy)
379    }

389    fn stat(&self, file_id: u64) -> Result<FileStat, SyscallError> {
390        let (kind, _pid) = Self::decode_id(file_id);
391        let is_dir = file_id == 0 || file_id == 1 || kind == KIND_PROC_DIR;
392        let mut st = if is_dir {
393            FileStat {
394                st_ino: file_id,
395                st_mode: 0o040555,
396                st_nlink: 2,
397                st_size: 0,
398                st_blksize: 512,
399                st_blocks: 0,
400                ..FileStat::zeroed()
401            }
402        } else {
403            FileStat {
404                st_ino: file_id,
405                st_mode: 0o100444,
406                st_nlink: 1,
407                st_size: 0,
408                st_blksize: 512,
409                st_blocks: 0,
410                ..FileStat::zeroed()
411            }
412        };
413
414        let (kind, pid) = Self::decode_id(file_id);
415        let per_process_node =
416            kind == KIND_PROC_DIR || kind == KIND_PROC_STATUS || kind == KIND_PROC_CMDLINE;
417        if per_process_node && pid != 0 {
418            if let Some(tasks) = get_all_tasks() {
419                if let Some(task) = tasks.iter().find(|t| t.pid as u64 == pid) {
420                    st.st_uid = task.uid.load(core::sync::atomic::Ordering::Relaxed);
421                    st.st_gid = task.gid.load(core::sync::atomic::Ordering::Relaxed);
422                }
423            }
424        }
425        Ok(finalize_pseudo_stat(st, DEV_PROCFS, 0))
426    }

428    fn readdir(&self, file_id: u64) -> Result<Vec<DirEntry>, SyscallError> {
429        let (kind, pid) = Self::decode_id(file_id);
430        if file_id == 0 {
431            let mut entries = Vec::new();
432            entries.push(DirEntry {
433                ino: 1,
434                file_type: DT_DIR,
435                name: String::from("self"),
436            });
437            entries.push(DirEntry {
438                ino: 10,
439                file_type: DT_REG,
440                name: String::from("cpuinfo"),
441            });
442            entries.push(DirEntry {
443                ino: 11,
444                file_type: DT_REG,
445                name: String::from("meminfo"),
446            });
447            entries.push(DirEntry {
448                ino: 12,
449                file_type: DT_REG,
450                name: String::from("version"),
451            });
452            entries.push(DirEntry {
453                ino: 13,
454                file_type: DT_REG,
455                name: String::from("silos"),
456            });
457            if let Some(tasks) = get_all_tasks() {
458                for task in tasks {
459                    entries.push(DirEntry {
460                        ino: Self::encode_id(KIND_PROC_DIR, task.pid as u64),
461                        file_type: DT_DIR,
462                        name: format!("{}", task.pid),
463                    });
464                }
465            }
466            Ok(entries)
467        } else if file_id == 1 || kind == KIND_PROC_DIR {
468            let effective_pid = if file_id == 1 {
469                current_pid().map(|p| p as u64).unwrap_or(0)
470            } else {
471                pid
472            };
473            Ok(alloc::vec![
474                DirEntry {
475                    ino: Self::encode_id(KIND_PROC_STATUS, effective_pid),
476                    file_type: DT_REG,
477                    name: String::from("status"),
478                },
479                DirEntry {
480                    ino: Self::encode_id(KIND_PROC_CMDLINE, effective_pid),
481                    file_type: DT_REG,
482                    name: String::from("cmdline"),
483                },
484            ])
485        } else {
486            Err(SyscallError::InvalidArgument)
487        }
488    }
