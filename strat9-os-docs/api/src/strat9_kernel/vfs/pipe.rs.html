1//! Kernel pipe implementation.
2//!
3//! A pipe is a unidirectional byte stream between two file descriptors.
4//! The read end blocks when empty; the write end returns EPIPE when
5//! the read end is closed.
6
7use crate::{sync::{SpinLock, waitqueue::WaitQueue}, syscall::error::SyscallError};
8use alloc::sync::Arc;
9
10const PIPE_BUF_SIZE: usize = 4096;
11
12/// Shared state for one pipe instance.
13struct PipeInner {
14    buf: [u8; PIPE_BUF_SIZE],
15    read_pos: usize,
16    write_pos: usize,
17    /// Number of bytes currently buffered.
18    len: usize,
19    read_closed: bool,
20    write_closed: bool,
21    /// Number of open file-descriptions referencing the read end.
22    /// The end is marked closed only when this reaches zero.
23    read_refs: usize,
24    /// Number of open file-descriptions referencing the write end.
25    write_refs: usize,
26}
27
28impl PipeInner {
29    fn new() -> Self {
30        PipeInner {
31            buf: [0u8; PIPE_BUF_SIZE],
32            read_pos: 0,
33            write_pos: 0,
34            len: 0,
35            read_closed: false,
36            write_closed: false,
37            read_refs: 1,
38            write_refs: 1,
39        }
40    }
41
42    fn is_empty(&self) -> bool {
43        self.len == 0
44    }
45
46    fn is_full(&self) -> bool {
47        self.len >= PIPE_BUF_SIZE
48    }
49
50    fn available_read(&self) -> usize {
51        self.len
52    }
53
54    fn available_write(&self) -> usize {
55        PIPE_BUF_SIZE - self.len
56    }
57}
58
59/// Shared pipe handle.
60pub struct Pipe {
61    inner: SpinLock<PipeInner>,
62    /// Woken when data becomes available (or write end is closed).
63    readers: WaitQueue,
64    /// Woken when space becomes available (or read end is closed).
65    writers: WaitQueue,
66}
67
68impl Pipe {
69    pub fn new() -> Arc<Self> {
70        Arc::new(Pipe {
71            inner: SpinLock::new(PipeInner::new()),
72            readers: WaitQueue::new(),
73            writers: WaitQueue::new(),
74        })
75    }
76
77    /// Read from the pipe. Returns 0 on EOF (write end closed + empty).
78    pub fn read(&self, buf: &mut [u8]) -> Result<usize, SyscallError> {
79        let result = self.readers.wait_until(|| {
80            let mut inner = self.inner.lock();
81
82            if inner.available_read() > 0 {
83                let to_read = core::cmp::min(buf.len(), inner.available_read());
84                for i in 0..to_read {
85                    buf[i] = inner.buf[inner.read_pos];
86                    inner.read_pos = (inner.read_pos + 1) % PIPE_BUF_SIZE;
87                }
88                inner.len -= to_read;
89                return Some(Ok(to_read));
90            }
91
92            if inner.write_closed {
93                return Some(Ok(0)); // EOF
94            }
95
96            None // block
97        });
98        self.writers.wake_one();
99        result
100    }
101
102    /// Write to the pipe. Returns EPIPE if read end is closed.
103    pub fn write(&self, buf: &[u8]) -> Result<usize, SyscallError> {
104        if buf.is_empty() {
105            return Ok(0);
106        }
107
108        let mut total = 0;
109        while total < buf.len() {
110            let wrote_some = self.writers.wait_until(|| {
111                let mut inner = self.inner.lock();
112
113                if inner.read_closed {
114                    if total > 0 {
115                        return Some(Ok(0)); // return what we have
116                    }
117                    return Some(Err(SyscallError::Pipe));
118                }
119
120                if inner.available_write() > 0 {
121                    let to_write = core::cmp::min(buf.len() - total, inner.available_write());
122                    for i in 0..to_write {
123                        let wp = inner.write_pos;
124                        inner.buf[wp] = buf[total + i];
125                        inner.write_pos = (wp + 1) % PIPE_BUF_SIZE;
126                    }
127                    inner.len += to_write;
128                    return Some(Ok(to_write));
129                }
130
131                None // block
132            });
133            self.readers.wake_one();
134            let n = wrote_some?;
135            if n == 0 {
136                break; // read_closed mid-write, return partial
137            }
138            total += n;
139        }
140
141        Ok(total)
142    }
143
144    /// Increment the read-end refcount (called on dup/fork).
145    pub fn dup_read(&self) {
146        self.inner.lock().read_refs += 1;
147    }
148
149    /// Increment the write-end refcount (called on dup/fork).
150    pub fn dup_write(&self) {
151        self.inner.lock().write_refs += 1;
152    }
153
154    /// Decrement the read-end refcount; marks the end closed only when it
155    /// reaches zero.  Returns true if the end was actually closed.
156    pub fn close_read(&self) -> bool {
157        let mut inner = self.inner.lock();
158        if inner.read_refs == 0 {
159            return false;
160        }
161        inner.read_refs -= 1;
162        if inner.read_refs == 0 {
163            inner.read_closed = true;
164            drop(inner);
165            self.writers.wake_all();
166            true
167        } else {
168            false
169        }
170    }
171
172    /// Decrement the write-end refcount; marks the end closed only when it
173    /// reaches zero.  Returns true if the end was actually closed.
174    pub fn close_write(&self) -> bool {
175        let mut inner = self.inner.lock();
176        if inner.write_refs == 0 {
177            return false;
178        }
179        inner.write_refs -= 1;
180        if inner.write_refs == 0 {
181            inner.write_closed = true;
182            drop(inner);
183            self.readers.wake_all();
184            true
185        } else {
186            false
187        }
188    }
189}

195use super::scheme::{finalize_pseudo_stat, DirEntry, FileStat, OpenFlags, OpenResult, Scheme, DEV_PIPEFS};
196use alloc::{collections::BTreeMap, vec::Vec};
197use core::sync::atomic::{AtomicU64, Ordering};
198
199/// A scheme that manages kernel pipes.
200///
201/// Each pipe gets two file_ids: even = read end, odd = write end.
202pub struct PipeScheme {
203    pipes: SpinLock<BTreeMap<u64, Arc<Pipe>>>,
204}
205
206static NEXT_PIPE_ID: AtomicU64 = AtomicU64::new(2); // Start at 2 (even numbers)
207
208impl PipeScheme {
209    pub fn new() -> Self {
210        PipeScheme {
211            pipes: SpinLock::new(BTreeMap::new()),
212        }
213    }
214
215    /// Create a new pipe pair. Returns (read_file_id, write_file_id).
216    pub fn create_pipe(&self) -> (u64, Arc<Pipe>) {
217        let base_id = NEXT_PIPE_ID.fetch_add(2, Ordering::SeqCst);
218        let pipe = Pipe::new();
219        self.pipes.lock().insert(base_id, pipe.clone());
220        (base_id, pipe)
221    }
222
223    fn get_pipe(&self, file_id: u64) -> Result<Arc<Pipe>, SyscallError> {
224        let base = file_id & !1; // Even = base
225        self.pipes
226            .lock()
227            .get(&base)
228            .cloned()
229            .ok_or(SyscallError::BadHandle)
230    }
231
232    fn is_read_end(file_id: u64) -> bool {
233        file_id & 1 == 0
234    }
235}
236
237impl Scheme for PipeScheme {
238    fn open(&self, _path: &str, _flags: OpenFlags) -> Result<OpenResult, SyscallError> {
239        Err(SyscallError::NotSupported) // Pipes are created via sys_pipe, not open()
240    }
241
242    fn read(&self, file_id: u64, _offset: u64, buf: &mut [u8]) -> Result<usize, SyscallError> {
243        if !Self::is_read_end(file_id) {
244            return Err(SyscallError::PermissionDenied);
245        }
246        let pipe = self.get_pipe(file_id)?;
247        pipe.read(buf)
248    }
249
250    fn write(&self, file_id: u64, _offset: u64, buf: &[u8]) -> Result<usize, SyscallError> {
251        if Self::is_read_end(file_id) {
252            return Err(SyscallError::PermissionDenied);
253        }
254        let pipe = self.get_pipe(file_id)?;
255        pipe.write(buf)
256    }
257
258    fn close(&self, file_id: u64) -> Result<(), SyscallError> {
259        let pipe = self.get_pipe(file_id)?;
260        if Self::is_read_end(file_id) {
261            pipe.close_read();
262        } else {
263            pipe.close_write();
264        }
265
266        // Remove the shared Pipe entry only when both ends are fully closed
267        // (both refcounts have reached zero).
268        let base = file_id & !1;
269        let inner = pipe.inner.lock();
270        if inner.read_closed && inner.write_closed {
271            drop(inner);
272            self.pipes.lock().remove(&base);
273        }
274        Ok(())
275    }
276
277    fn stat(&self, file_id: u64) -> Result<FileStat, SyscallError> {
278        let pipe = self.get_pipe(file_id)?;
279        let inner = pipe.inner.lock();
280        Ok(finalize_pseudo_stat(FileStat {
281            st_ino: file_id,
282            st_mode: 0o010600, // S_IFIFO | rw-------
283            st_nlink: 1,
284            st_size: inner.len as u64,
285            st_blksize: PIPE_BUF_SIZE as u64,
286            st_blocks: 0,
287            ..FileStat::zeroed()
288        }, DEV_PIPEFS, 0))
289    }
290
291    fn readdir(&self, _file_id: u64) -> Result<Vec<DirEntry>, SyscallError> {
292        Err(SyscallError::InvalidArgument)
293    }
294
295    fn size(&self, file_id: u64) -> Result<u64, SyscallError> {
296        let pipe = self.get_pipe(file_id)?;
297        let len = pipe.inner.lock().len;
298        Ok(len as u64)
299    }
