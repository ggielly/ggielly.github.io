1use alloc::sync::Arc;
2use alloc::string::String;
3use core::sync::atomic::{AtomicU32, AtomicU64};
4use crate::memory::AddressSpace;
5use crate::vfs::FileDescriptorTable;
6use crate::capability::CapabilityTable;
7use crate::process::task::SyncUnsafeCell;
8use crate::process::signal::SigActionData;
9
10/// Represents a process (a group of threads sharing resources).
11pub struct Process {
12    /// Process identifier visible to userspace.
13    pub pid: crate::process::Pid,
14    
15    /// Address space for this process
16    pub address_space: SyncUnsafeCell<Arc<AddressSpace>>,
17    /// File descriptor table for this process
18    pub fd_table: SyncUnsafeCell<FileDescriptorTable>,
19    /// Capabilities granted to this process
20    pub capabilities: SyncUnsafeCell<CapabilityTable>,
21    
22    /// Signal actions (handlers) for this process
23    pub signal_actions: SyncUnsafeCell<[SigActionData; 64]>,
24    
25    /// Program break (end of heap), in bytes. 0 = not yet initialised.
26    pub brk: AtomicU64,
27    /// mmap_hint: next candidate virtual address for anonymous mmap allocations
28    pub mmap_hint: AtomicU64,
29    
30    /// Current working directory (POSIX, inherited by children).
31    pub cwd: SyncUnsafeCell<String>,
32    /// File creation mask (inherited by children, NOT reset by exec).
33    pub umask: AtomicU32,
34}
35
36impl Process {
37    pub fn new(
38        pid: crate::process::Pid,
39        address_space: Arc<AddressSpace>,
40    ) -> Self {
41        Self {
42            pid,
43            address_space: SyncUnsafeCell::new(address_space),
44            fd_table: SyncUnsafeCell::new(FileDescriptorTable::new()),
45            capabilities: SyncUnsafeCell::new(CapabilityTable::new()),
46            signal_actions: SyncUnsafeCell::new([SigActionData::default(); 64]),
47            brk: AtomicU64::new(0),
48            mmap_hint: AtomicU64::new(0x0000_0000_6000_0000),
49            cwd: SyncUnsafeCell::new(String::from("/")),
50            umask: AtomicU32::new(0o022),
51        }
52    }
53}
