1//! Interval timer support (ITIMER_REAL, ITIMER_VIRTUAL, ITIMER_PROF)
2//!
3//! POSIX interval timers allow processes to receive periodic signals.
4//! Three types are supported:
5//! - ITIMER_REAL: Real (wall clock) time, sends SIGALRM
6//! - ITIMER_VIRTUAL: Process CPU time, sends SIGVTALRM (not yet impl)
7//! - ITIMER_PROF: Process + system CPU time, sends SIGPROF (not yet impl)
8
9use core::sync::atomic::{AtomicBool, AtomicU64, Ordering};
10
11/// Interval timer types
12#[derive(Debug, Clone, Copy, PartialEq, Eq)]
13#[repr(u32)]
14pub enum ITimerWhich {
15    Real = 0,
16    Virtual = 1,
17    Prof = 2,
18}
19
20impl ITimerWhich {
21    pub fn from_u32(value: u32) -> Option<Self> {
22        match value {
23            0 => Some(ITimerWhich::Real),
24            1 => Some(ITimerWhich::Virtual),
25            2 => Some(ITimerWhich::Prof),
26            _ => None,
27        }
28    }
29
30    /// Returns the signal number to send when this timer expires
31    pub fn signal(self) -> u32 {
32        match self {
33            ITimerWhich::Real => 14,    // SIGALRM
34            ITimerWhich::Virtual => 26, // SIGVTALRM
35            ITimerWhich::Prof => 27,    // SIGPROF
36        }
37    }
38}

40/// Interval timer specification (matches POSIX itimerval)
41#[repr(C)]
42#[derive(Debug, Clone, Copy)]
43pub struct ITimerVal {
44    /// Interval for periodic timer (0 = one-shot)
45    pub it_interval: TimeVal,
46    /// Current value (time until next expiration)
47    pub it_value: TimeVal,
48}
49
50impl ITimerVal {
51    pub const fn zero() -> Self {
52        Self {
53            it_interval: TimeVal::zero(),
54            it_value: TimeVal::zero(),
55        }
56    }
57
58    /// Convert to nanoseconds
59    pub fn to_nanos(&self) -> (u64, u64) {
60        (self.it_interval.to_nanos(), self.it_value.to_nanos())
61    }
62}

64/// Time value (seconds + microseconds)
65#[repr(C)]
66#[derive(Debug, Clone, Copy)]
67pub struct TimeVal {
68    pub tv_sec: i64,
69    pub tv_usec: i64,
70}
71
72impl TimeVal {
73    pub const fn zero() -> Self {
74        Self {
75            tv_sec: 0,
76            tv_usec: 0,
77        }
78    }
79
80    /// Convert to nanoseconds
81    pub fn to_nanos(&self) -> u64 {
82        (self.tv_sec as u64)
83            .saturating_mul(1_000_000_000)
84            .saturating_add((self.tv_usec as u64).saturating_mul(1_000))
85    }
86
87    /// Create from nanoseconds
88    pub fn from_nanos(nanos: u64) -> Self {
89        let tv_sec = (nanos / 1_000_000_000) as i64;
90        let tv_usec = ((nanos % 1_000_000_000) / 1_000) as i64;
91        Self { tv_sec, tv_usec }
92    }
93}

95/// Per-task interval timer state
96pub struct ITimerState {
97    /// Next expiration time in nanoseconds (0 = disarmed)
98    next_expiration: AtomicU64,
99    /// Interval in nanoseconds (0 = one-shot)
100    interval_ns: AtomicU64,
101    /// Whether this timer is armed
102    armed: AtomicBool,
103}
104
105impl ITimerState {
106    pub const fn new() -> Self {
107        Self {
108            next_expiration: AtomicU64::new(0),
109            interval_ns: AtomicU64::new(0),
110            armed: AtomicBool::new(false),
111        }
112    }
113
114    /// Get current timer value (time until next expiration)
115    pub fn get(&self, current_time_ns: u64) -> ITimerVal {
116        let next = self.next_expiration.load(Ordering::Relaxed);
117        let interval = self.interval_ns.load(Ordering::Relaxed);
118
119        let value_ns = if next > current_time_ns {
120            next - current_time_ns
121        } else {
122            0
123        };
124
125        ITimerVal {
126            it_interval: TimeVal::from_nanos(interval),
127            it_value: TimeVal::from_nanos(value_ns),
128        }
129    }
130
131    /// Set timer value
132    pub fn set(&self, value: &ITimerVal, current_time_ns: u64) {
133        let (interval_ns, value_ns) = value.to_nanos();
134
135        if value_ns == 0 {
136            self.armed.store(false, Ordering::Release);
137            self.next_expiration.store(0, Ordering::Relaxed);
138            self.interval_ns.store(interval_ns, Ordering::Relaxed);
139        } else {
140            self.interval_ns.store(interval_ns, Ordering::Relaxed);
141            let next = current_time_ns.saturating_add(value_ns);
142            self.next_expiration.store(next, Ordering::Relaxed);
143            self.armed.store(true, Ordering::Release);
144        }
145    }
146
147    /// Check if timer has expired and needs to fire
148    pub fn check_expired(&self, current_time_ns: u64) -> bool {
149        if !self.armed.load(Ordering::Acquire) {
150            return false;
151        }
152
153        let next = self.next_expiration.load(Ordering::Relaxed);
154        if current_time_ns >= next && next != 0 {
155            let interval = self.interval_ns.load(Ordering::Relaxed);
156            if interval == 0 {
157                self.armed.store(false, Ordering::Release);
158                self.next_expiration.store(0, Ordering::Relaxed);
159            } else {
160                // Anchor to previous expiration to avoid drift accumulation.
161                let new_next = next.saturating_add(interval);
162                let new_next = if new_next <= current_time_ns {
163                    current_time_ns.saturating_add(interval)
164                } else {
165                    new_next
166                };
167                self.next_expiration.store(new_next, Ordering::Relaxed);
168            }
169            true
170        } else {
171            false
172        }
173    }
174
175    /// Disarm the timer
176    pub fn disarm(&self) {
177        self.armed.store(false, Ordering::Release);
178        self.next_expiration.store(0, Ordering::Relaxed);
179        self.interval_ns.store(0, Ordering::Relaxed);
180    }
181}

183/// Container for all three interval timers
184pub struct ITimers {
185    pub real: ITimerState,
186    pub virtual_timer: ITimerState,
187    pub prof: ITimerState,
188}
189
190impl ITimers {
191    pub const fn new() -> Self {
192        Self {
193            real: ITimerState::new(),
194            virtual_timer: ITimerState::new(),
195            prof: ITimerState::new(),
196        }
197    }
198
199    pub fn get(&self, which: ITimerWhich) -> &ITimerState {
200        match which {
201            ITimerWhich::Real => &self.real,
202            ITimerWhich::Virtual => &self.virtual_timer,
203            ITimerWhich::Prof => &self.prof,
204        }
205    }
206
207    /// Check all timers for expiration and send signals if necessary
208    /// Returns a list of (ITimerWhich, signal_number) pairs for expired timers
209    pub fn check_all(&self, current_time_ns: u64) -> alloc::vec::Vec<(ITimerWhich, u32)> {
210        use alloc::vec::Vec;
211        let mut expired = Vec::new();
212
213        if self.real.check_expired(current_time_ns) {
214            expired.push((ITimerWhich::Real, ITimerWhich::Real.signal()));
215        }
216        if self.virtual_timer.check_expired(current_time_ns) {
217            expired.push((ITimerWhich::Virtual, ITimerWhich::Virtual.signal()));
218        }
219        if self.prof.check_expired(current_time_ns) {
220            expired.push((ITimerWhich::Prof, ITimerWhich::Prof.signal()));
221        }
222
223        expired
224    }
225}

227/// Global timer tick function to be called from timer interrupt handler.
228/// Checks all tasks for expired interval timers and delivers signals.
229///
230/// Zero-allocation: iterates tasks directly under the scheduler try_lock
231/// and sets pending signals via atomic ops. This avoids Vec/heap allocation
232/// in interrupt context which would deadlock against the buddy allocator.
233pub fn tick_all_timers(current_time_ns: u64) {
234    use crate::process::{scheduler::SCHEDULER, signal::Signal};
235
236    let scheduler = match SCHEDULER.try_lock() {
237        Some(guard) => guard,
238        None => return,
239    };
240    let Some(ref sched) = *scheduler else { return };
241
242    for (_, task) in sched.all_tasks.iter() {
243        for which in [ITimerWhich::Real, ITimerWhich::Virtual, ITimerWhich::Prof] {
244            if task.itimers.get(which).check_expired(current_time_ns) {
245                if let Some(sig) = Signal::from_u32(which.signal()) {
246                    task.pending_signals.add(sig);
247                }
248            }
249        }
250    }
