4    /// Create a new scheduler instance
5    pub fn new(cpu_count: usize) -> Self {
6        crate::serial_println!(
7            "[trace][sched] Scheduler::new enter cpu_count={}",
8            cpu_count
9        );
10        let mut cpus = alloc::vec::Vec::new();
11        for cpu_idx in 0..cpu_count {
12            crate::serial_println!(
13                "[trace][sched] Scheduler::new cpu={} create idle begin",
14                cpu_idx
15            );
16            let idle_task = Task::new_kernel_task(idle_task_main, "idle", TaskPriority::Idle)
17                .expect("Failed to create idle task");
18            crate::serial_println!(
19                "[trace][sched] Scheduler::new cpu={} create idle done id={}",
20                cpu_idx,
21                idle_task.id.as_u64()
22            );
23            idle_task.set_sched_policy(crate::process::sched::SchedPolicy::Idle);
24            let mut class_rqs = PerCpuClassRqSet::new();
25            class_rqs.enqueue(crate::process::sched::SchedClassId::Idle, idle_task.clone());
26            cpus.push(SchedulerCpu {
27                class_rqs,
28                current_task: None,
29                current_runtime: crate::process::sched::CurrentRuntime::new(),
30                idle_task,
31                task_to_requeue: None,
32                task_to_drop: None,
33                need_resched: false,
34            });
35            crate::serial_println!("[trace][sched] Scheduler::new cpu={} push done", cpu_idx);
36        }
37        crate::serial_println!(
38            "[trace][sched] Scheduler::new cpus ready len={}",
39            cpus.len()
40        );
41
42        Scheduler {
43            cpus,
44            blocked_tasks: BTreeMap::new(),
45            all_tasks: BTreeMap::new(),
46            task_cpu: BTreeMap::new(),
47            pid_to_task: BTreeMap::new(),

63    pub(crate) fn member_add(
64        map: &mut BTreeMap<Pid, alloc::vec::Vec<TaskId>>,
65        key: Pid,
66        task_id: TaskId,
67    ) {
68        let members = map.entry(key).or_default();
69        if !members.iter().any(|id| *id == task_id) {
70            members.push(task_id);
71        }

74    /// Performs the member remove operation.
75    pub(crate) fn member_remove(
76        map: &mut BTreeMap<Pid, alloc::vec::Vec<TaskId>>,
77        key: Pid,
78        task_id: TaskId,
79    ) {
80        let mut clear = false;
81        if let Some(members) = map.get_mut(&key) {
82            members.retain(|id| *id != task_id);
83            clear = members.is_empty();
84        }
85        if clear {
86            map.remove(&key);
87        }
88    }

90    /// Performs the register identity locked operation.
91    pub(crate) fn register_identity_locked(&mut self, task: &Arc<Task>) {
92        if (self as *mut Self).is_null() {
93            crate::serial_println!(
94                "[sched] WARN: register_identity_locked called on NULL scheduler!"
95            );
96            return;
97        }
98        let task_ptr = Arc::as_ptr(task);
99        if task_ptr.is_null() {
100            crate::serial_println!("[sched] WARN: register_identity_locked called with NULL task!");
101            return;
102        }
103        let task_id = task.id;
104        let pid = task.pid;
105        let pgid = task.pgid.load(Ordering::Relaxed);
106        let sid = task.sid.load(Ordering::Relaxed);
107        self.pid_to_pgid.insert(pid, pgid);
108        self.pid_to_sid.insert(pid, sid);
109        Self::member_add(&mut self.pgid_members, pgid, task_id);
110        Self::member_add(&mut self.sid_members, sid, task_id);
111    }

113    /// Performs the unregister identity locked operation.
114    pub(crate) fn unregister_identity_locked(&mut self, task_id: TaskId, pid: Pid, tid: Tid) {
115        self.pid_to_task.remove(&pid);
116        self.tid_to_task.remove(&tid);
117        if let Some(pgid) = self.pid_to_pgid.remove(&pid) {
118            Self::member_remove(&mut self.pgid_members, pgid, task_id);
119        }
120        if let Some(sid) = self.pid_to_sid.remove(&pid) {
121            Self::member_remove(&mut self.sid_members, sid, task_id);
122        }
123    }

140    /// Performs the add task on cpu operation.
141    fn add_task_on_cpu(&mut self, task: Arc<Task>, cpu_index: usize) {
142        if (self as *mut Self).is_null() {
143            crate::serial_println!("[sched] ERROR: add_task_on_cpu called on NULL scheduler!");
144            return;
145        }
146        let task_id = task.id;
147        // SAFETY: We have exclusive access via the scheduler lock
148        unsafe {
149            *task.state.get() = TaskState::Ready;
150        }
151
152        self.all_tasks.insert(task_id, task.clone());
153        self.task_cpu.insert(task_id, cpu_index);
154        self.pid_to_task.insert(task.pid, task_id);
155        self.tid_to_task.insert(task.tid, task_id);
156        self.register_identity_locked(&task);
157        if let Some(cpu) = self.cpus.get_mut(cpu_index) {
158            let class = self.class_table.class_for_task(&task);
159            cpu.class_rqs.enqueue(class, task);
160            cpu.need_resched = true;
161        }
162        sched_trace(format_args!(
163            "enqueue task={} cpu={}",
164            task_id.as_u64(),
165            cpu_index
166        ));
167        if cpu_index != current_cpu_index() {
168            send_resched_ipi_to_cpu(cpu_index);
169        }
170    }

195    /// Performs the wake task locked operation.
196    pub fn wake_task_locked(&mut self, id: TaskId) -> bool {
197        self.clear_task_wake_deadline_locked(id);
198        if let Some(task) = self.blocked_tasks.remove(&id) {
199            // SAFETY: scheduler lock held.
200            unsafe {
201                *task.state.get() = TaskState::Ready;
202            }
203            let cpu_index = self.task_cpu.get(&id).copied().unwrap_or(0);
204            if let Some(cpu) = self.cpus.get_mut(cpu_index) {
205                let class = self.class_table.class_for_task(&task);
206                cpu.class_rqs.enqueue(class, task);
207                cpu.need_resched = true;
208            }
209            if cpu_index != current_cpu_index() {
210                send_resched_ipi_to_cpu(cpu_index);
211            }
212            true
213        } else if let Some(task) = self.all_tasks.get(&id) {
214            task.wake_pending
215                .store(true, core::sync::atomic::Ordering::Release);
216            true
217        } else {
218            false
219        }
220    }

222    /// Attempts to reap child locked.
223    pub fn try_reap_child_locked(
224        &mut self,
225        parent: TaskId,
226        target: Option<TaskId>,
227    ) -> WaitChildResult {
228        let Some(children_view) = self.children_of.get(&parent) else {
229            return WaitChildResult::NoChildren;
230        };
231
232        if children_view.is_empty() {
233            return WaitChildResult::NoChildren;
234        }
235
236        if let Some(target_id) = target {
237            if !children_view.iter().any(|&id| id == target_id) {
238                return WaitChildResult::NoChildren;
239            }
240        }
241
242        let zombie = children_view
243            .iter()
244            .copied()
245            .find(|id| target.map_or(true, |t| t == *id) && self.zombies.contains_key(id));
246
247        if let Some(child) = zombie {
248            let (status, child_pid) = self.zombies.remove(&child).unwrap_or((0, 0));
249            let child_tid = self.all_tasks.get(&child).map(|task| task.tid);
250            if child_pid != 0 {
251                if let Some(tid) = child_tid {
252                    self.unregister_identity_locked(child, child_pid, tid);
253                }
254            }
255            if let Some(children) = self.children_of.get_mut(&parent) {
256                children.retain(|&id| id != child);
257                if children.is_empty() {
258                    self.children_of.remove(&parent);
259                }
260            }
261            self.parent_of.remove(&child);
262            return WaitChildResult::Reaped {
263                child,
264                pid: child_pid,
265                status,
266            };
267        }
268
269        WaitChildResult::StillRunning
270    }

272    /// Pick the next task to run on `cpu_index`.
273    ///
274    /// 1. Re-queues the current Running task (round-robin).
275    ///    The **idle task is never re-queued** — it is always available
276    ///    as the last-resort fallback.  Keeping it out of the ready queue
277    ///    ensures the queue becomes truly empty when there is no real work,
278    ///    which lets work-stealing kick in on other CPUs.
279    /// 2. Pops from the local ready queue.
280    /// 3. Falls back to **work-stealing** from the busiest other CPU.
281    /// 4. Falls back to the per-CPU idle task.
282    pub fn pick_next_task(&mut self, cpu_index: usize) -> Arc<Task> {
283        // Step 1: take current task and re-queue it if still alive.
284        let current_task = self.cpus[cpu_index].current_task.take();
285        if let Some(task) = current_task {
286            // SAFETY: scheduler lock held; we have exclusive access to state.
287            let task_state = unsafe { *task.state.get() };
288            if task_state == TaskState::Running {
289                unsafe {
290                    *task.state.get() = TaskState::Ready;
291                }
292                // Never re-queue the per-CPU idle task.  It lives in
293                // `cpus[cpu_index].idle_task` and is cloned as a fallback
294                // below.  Putting it in the ready queue prevents it from
295                // ever being empty, which breaks work-stealing.
296                if !Arc::ptr_eq(&task, &self.cpus[cpu_index].idle_task) {
297                    // DO NOT push to ready_queue yet!
298                    // Another CPU could steal it before its context is saved.
299                    self.cpus[cpu_index].task_to_requeue = Some(task);
300                }
301            } else if task_state == TaskState::Dead {
302                // Task is Dead. Clean up before deferring the drop.
303                // We must remove from all_tasks NOW to prevent double-cleanup
304                // if the task is also being killed via kill_task().
305                let task_id = task.id;
306                let task_pid = task.pid;
307                let task_tid = task.tid;
308
309                // Remove from global maps - this is the canonical cleanup point
310                self.all_tasks.remove(&task_id);
311                self.task_cpu.remove(&task_id);
312                self.unregister_identity_locked(task_id, task_pid, task_tid);
313
314                // Run resource cleanup (ports, capabilities, etc.)
315                // SAFETY: scheduler lock held, task is no longer accessible
316                super::task_ops::cleanup_task_resources(&task);
317
318                // Defer the actual Arc drop to finish_switch()
319                self.cpus[cpu_index].task_to_drop = Some(task);
320            } else {
321                // Blocked task: leave bookkeeping intact; blocked_tasks holds the ref.
322                // Dropping the Arc here is fine; the blocked map keeps the task alive.
323            }
324        }
325
326        // Step 2: local queue, then work-steal, then idle.
327        let next_task = if let Some(task) =
328            self.cpus[cpu_index].class_rqs.pick_next(&self.class_table)
329        {
330            task
331        } else if let Some(task) = self.steal_task(cpu_index, TICK_COUNT.load(Ordering::Relaxed)) {
332            task
333        } else {
334            self.cpus[cpu_index].idle_task.clone()
335        };
336
337        // SAFETY: scheduler lock held.
338        unsafe {
339            *next_task.state.get() = TaskState::Running;
340        }
341        let cloned = next_task.clone();
342        let strong_after = Arc::strong_count(&cloned);
343        self.cpus[cpu_index].current_task = Some(cloned);
344        // Reset the runtime accounting for the new task
345        self.cpus[cpu_index].current_runtime = crate::process::sched::CurrentRuntime::new();
346        sched_trace(format_args!(
347            "cpu={} pick_next task={} policy={:?} strong={}",
348            cpu_index,
349            next_task.id.as_u64(),
350            next_task.sched_policy(),
351            strong_after,
352        ));
353        next_task
354    }

356    /// Try to steal one task from the most-loaded other CPU.
357    ///
358    /// We steal from the **back** of the source queue (the task added most
359    /// recently — least likely to have warm cache data on that CPU).
360    /// We only steal when the source has ≥ 2 tasks, so it keeps at least one.
361    pub fn steal_task(&mut self, dst_cpu: usize, now_tick: u64) -> Option<Arc<Task>> {
362        if !cpu_is_valid(dst_cpu) || dst_cpu >= self.cpus.len() {
363            return None;
364        }
365        let last = LAST_STEAL_TICK[dst_cpu].load(Ordering::Relaxed);
366        if now_tick.saturating_sub(last) < STEAL_COOLDOWN_TICKS {
367            return None;
368        }
369
370        let dst_load = {
371            let cpu = &self.cpus[dst_cpu];
372            cpu.class_rqs.runnable_len() + usize::from(cpu.current_task.is_some())
373        };
374
375        // Find the busiest CPU that isn't ourselves.
376        let (best_src, best_src_load) = (0..self.cpus.len())
377            .filter(|&i| i != dst_cpu)
378            .map(|i| {
379                let cpu = &self.cpus[i];
380                let load = cpu.class_rqs.runnable_len() + usize::from(cpu.current_task.is_some());
381                (i, load)
382            })
383            .max_by_key(|(_, load)| *load)?;
384
385        if best_src_load < dst_load.saturating_add(STEAL_IMBALANCE_MIN) {
386            return None;
387        }
388
389        // Only steal if source will still have work left.
390        if self.cpus[best_src].class_rqs.runnable_len() < 2 {
391            return None;
392        }
393
394        let task = self.cpus[best_src]
395            .class_rqs
396            .steal_candidate(&self.class_table)?;
397        // Update the task->CPU mapping so wake/resume route correctly.
398        self.task_cpu.insert(task.id, dst_cpu);
399        log::trace!(
400            "WS: CPU {} stole task {:?} from CPU {} (src had {} tasks)",
401            dst_cpu,
402            task.id,
403            best_src,
404            self.cpus[best_src].class_rqs.runnable_len() + 1
405        );
406        sched_trace(format_args!(
407            "work-steal dst_cpu={} src_cpu={} task={}",
408            dst_cpu,
409            best_src,
410            task.id.as_u64()
411        ));
412        if cpu_is_valid(dst_cpu) {
413            CPU_STEAL_IN_COUNT[dst_cpu].fetch_add(1, Ordering::Relaxed);
414        }
415        if cpu_is_valid(best_src) {
416            CPU_STEAL_OUT_COUNT[best_src].fetch_add(1, Ordering::Relaxed);
417        }
418        LAST_STEAL_TICK[dst_cpu].store(now_tick, Ordering::Relaxed);
419        Some(task)
420    }

422    /// Prepare a context switch: pick next task, update TSS and CR3,
423    /// return raw pointers for `switch_context()`.
424    ///
425    /// Returns `None` if there's nothing to switch to (same task selected,
426    /// or no current task).
427    pub(super) fn yield_cpu(&mut self, cpu_index: usize) -> Option<SwitchTarget> {
428        if cpu_index >= self.cpus.len() {
429            return None;
430        }
431        // Must have a current task to yield from
432        let current_ref = self.cpus[cpu_index].current_task.as_ref()?;
433        let strong_before = Arc::strong_count(current_ref);
434        if strong_before == 0 || strong_before > (isize::MAX as usize) {
435            log::error!(
436                "[sched] CORRUPT Arc in yield_cpu before clone! cpu={} strong={:#x} task={} ptr={:p}",
437                cpu_index, strong_before,
438                current_ref.id.as_u64(),
439                Arc::as_ptr(current_ref) as *const u8,
440            );
441            return None;
442        }
443        let current = current_ref.clone();
444
445        // Pick the next task
446        let next = self.pick_next_task(cpu_index);
447
448        // Don't switch to ourselves
449        if Arc::ptr_eq(&current, &next) {
450            return None;
451        }
452        if cpu_is_valid(cpu_index) {
453            CPU_SWITCH_COUNT[cpu_index].fetch_add(1, Ordering::Relaxed);
454        }
455
456        if let Err(e) = validate_task_context(&next) {
457            // Do NOT panic here — the scheduler lock is held, which would
458            // deadlock the panic hook (current_task_clone → SCHEDULER.lock()).
459            let bad_rsp = unsafe { (*next.context.get()).saved_rsp };
460            let stk_base = next.kernel_stack.virt_base.as_u64();
461            let stk_top = stk_base + next.kernel_stack.size as u64;
462            crate::serial_println!(
463                "[sched] WARN: invalid ctx for task '{}' (id={}) cpu={}: {} \
464                 rsp={:#x} stack=[{:#x}..{:#x}] — restoring current task",
465                next.name,
466                next.id.as_u64(),
467                cpu_index,
468                e,
469                bad_rsp,
470                stk_base,
471                stk_top,
472            );
473            log::error!(
474                "[sched] invalid context for task '{}' (id={:?}) cpu={}: {} \
475                 rsp={:#x} stack=[{:#x}..{:#x}]",
476                next.name,
477                next.id,
478                cpu_index,
479                e,
480                bad_rsp,
481                stk_base,
482                stk_top,
483            );
484
485            // ── Restore scheduler invariants ────────────────────────────────────────
486            // `pick_next_task` already mutated state before we reached this point:
487            //   1. Took `current_task`, set its state to Ready, placed it in
488            //      `task_to_requeue` (non-idle tasks only).
489            //   2. Set `current_task = Some(next_clone)` with next.state = Running.
490            //
491            // We are NOT performing the actual context switch. Both mutations must be
492            // undone so the truly running task stays scheduled and no Arc reference
493            // is silently lost (which would lead to corrupted saved_rsp on the next
494            // timer tick when the scheduler tries to save context for the wrong task).
495
496            // (a) Is `next` the per-CPU idle fallback (accessed via clone, never in
497            //     class_rqs)?  We need this before any borrow of self.cpus.
498            let is_idle_fallback = Arc::ptr_eq(&next, &self.cpus[cpu_index].idle_task);
499
500            // (b) Restore the old running task as current_task.
501            //     Non-idle current was placed in task_to_requeue by pick_next_task.
502            //     Idle current was not (it is accessed via idle_task.clone()), so use
503            //     the `current` Arc cloned at the start of yield_cpu.
504            if let Some(prev) = self.cpus[cpu_index].task_to_requeue.take() {
505                // SAFETY: scheduler lock held; exclusive access to task state.
506                unsafe {
507                    *prev.state.get() = TaskState::Running;
508                }
509                self.cpus[cpu_index].current_task = Some(prev);
510            } else {
511                // SAFETY: scheduler lock held.
512                unsafe {
513                    *current.state.get() = TaskState::Running;
514                }
515                self.cpus[cpu_index].current_task = Some(current.clone());
516            }
517
518            // (c) If `next` came from the class run queue (non-idle), re-enqueue it
519            //     so it can be retried on the next tick.  Its context may be
520            //     transiently invalid (e.g. stack not yet committed by a new task);
521            //     it will be validated again when next picked.
522            if !is_idle_fallback {
523                // SAFETY: scheduler lock held.
524                unsafe {
525                    *next.state.get() = TaskState::Ready;
526                }
527                let class = self.class_table.class_for_task(&next);
528                if let Some(cpu) = self.cpus.get_mut(cpu_index) {
529                    cpu.class_rqs.enqueue(class, next);
530                }
531            }
532            // For the idle fallback: state was set to Running by pick_next_task but the
533            // idle task lives in idle_task.clone() not in class_rqs, so no re-enqueue
534            // is needed.  The next pick_next_task will reset its state correctly.
535
536            return None;
537        }
538
539        // Update TSS.rsp0 for the new task (needed for Ring 3 -> Ring 0 transitions)
540        let stack_top = next.kernel_stack.virt_base.as_u64() + next.kernel_stack.size as u64;
541        crate::arch::x86_64::tss::set_kernel_stack(x86_64::VirtAddr::new(stack_top));
542
543        // Update SYSCALL kernel RSP for the new task
544        crate::arch::x86_64::syscall::set_kernel_rsp(stack_top);
545
546        // Switch CR3 if the new task has a different address space
547        // SAFETY: The new task's address space has a valid PML4 with the kernel half mapped.
548        unsafe {
549            (*next.process.address_space.get()).switch_to();
550        }
551
552        Some(SwitchTarget {
553            old_rsp_ptr: unsafe { &raw mut (*current.context.get()).saved_rsp },
554            new_rsp_ptr: unsafe { &raw const (*next.context.get()).saved_rsp },
555            old_fpu_ptr: current.fpu_state.get() as *mut u8,
556            new_fpu_ptr: next.fpu_state.get() as *const u8,
557            old_xcr0: current
558                .xcr0_mask
559                .load(core::sync::atomic::Ordering::Relaxed),
560            new_xcr0: next.xcr0_mask.load(core::sync::atomic::Ordering::Relaxed),
561        })
562    }

564    /// Performs the select cpu for task operation.
565    fn select_cpu_for_task(&self) -> usize {
566        if (self as *const Self).is_null() {
567            return 0;
568        }
569        let mut best = 0usize;
570        let mut best_load = usize::MAX;
571        for (idx, cpu) in self.cpus.iter().enumerate() {
572            let mut load = cpu.class_rqs.runnable_len();
573            if let Some(current) = cpu.current_task.as_ref() {
574                if self.class_table.class_for_task(current)
575                    != crate::process::sched::SchedClassId::Idle
576                {
577                    load += 1;
578                }
579            }
580            if load < best_load {
581                best = idx;
582                best_load = load;
583            }
584        }
585        best
586    }

588    /// Performs the migrate ready tasks for new class table operation.
589    pub fn migrate_ready_tasks_for_new_class_table(&mut self) {
590        let mut ready: Vec<(TaskId, Arc<Task>, usize)> = Vec::new();
591        for (id, task) in self.all_tasks.iter() {
592            // SAFETY: scheduler lock is held by caller; task state is synchronized by scheduler.
593            let state = unsafe { *task.state.get() };
594            if state != TaskState::Ready {
595                continue;
596            }
597            let cpu = self.task_cpu.get(id).copied().unwrap_or(0);
598            ready.push((*id, task.clone(), cpu));
599        }
600
601        for (id, task, cpu_idx) in ready {
602            let Some(cpu) = self.cpus.get_mut(cpu_idx) else {
603                continue;
604            };
605            if cpu.class_rqs.remove(id) {
606                let class = self.class_table.class_for_task(&task);
607                cpu.class_rqs.enqueue(class, task);
608                cpu.need_resched = true;
609            }
610        }
611    }
