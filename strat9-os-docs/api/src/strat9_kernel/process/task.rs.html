15/// Performs the next pid operation.
16#[inline]
17fn next_pid() -> Pid {
18    static NEXT_PID: AtomicU32 = AtomicU32::new(1);
19    NEXT_PID.fetch_add(1, Ordering::SeqCst)
20}

22/// Performs the next tid operation.
23#[inline]
24fn next_tid() -> Tid {
25    static NEXT_TID: AtomicU32 = AtomicU32::new(1);
26    NEXT_TID.fetch_add(1, Ordering::SeqCst)
27}

104/// FPU/SSE/AVX extended state, saved and restored on context switch.
105///
106/// When XSAVE is available, uses `xsave`/`xrstor` with a variable-size area.
107/// Falls back to `fxsave`/`fxrstor` (512 bytes) on older CPUs.
108#[repr(C, align(64))]
109pub struct ExtendedState {
110    pub data: [u8; Self::MAX_XSAVE_SIZE],
111    pub size: usize,
112    pub uses_xsave: bool,
113    pub xcr0_mask: u64,
114}
115
116impl core::fmt::Debug for ExtendedState {
117    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
118        f.debug_struct("ExtendedState")
119            .field("size", &self.size)
120            .field("uses_xsave", &self.uses_xsave)
121            .field("xcr0_mask", &self.xcr0_mask)
122            .finish()
123    }
124}
125
126impl ExtendedState {
127    pub const FXSAVE_SIZE: usize = 512;
128    pub const MAX_XSAVE_SIZE: usize = 2688;
129
130    /// Create a new default state using the host's maximum capabilities.
131    pub fn new() -> Self {
132        let (uses_xsave, size, max_xcr0) =
133            if crate::arch::x86_64::cpuid::host_uses_xsave() {
134                let h = crate::arch::x86_64::cpuid::host();
135                (true, h.xsave_size.min(Self::MAX_XSAVE_SIZE), h.max_xcr0)
136            } else {
137                (false, Self::FXSAVE_SIZE, 0x3)
138            };
139
140        let mut state = Self {
141            data: [0u8; Self::MAX_XSAVE_SIZE],
142            size,
143            uses_xsave,
144            xcr0_mask: max_xcr0,
145        };
146        state.set_defaults();
147        state
148    }
149
150    /// Create a state for a specific XCR0 mask (per-silo feature restriction).
151    pub fn for_xcr0(xcr0: u64) -> Self {
152        let uses_xsave = crate::arch::x86_64::cpuid::host_uses_xsave();
153        let size = if uses_xsave {
154            crate::arch::x86_64::cpuid::xsave_size_for_xcr0(xcr0)
155                .min(Self::MAX_XSAVE_SIZE)
156        } else {
157            Self::FXSAVE_SIZE
158        };
159
160        let mut state = Self {
161            data: [0u8; Self::MAX_XSAVE_SIZE],
162            size,
163            uses_xsave,
164            xcr0_mask: xcr0,
165        };
166        state.set_defaults();
167        state
168    }
169
170    fn set_defaults(&mut self) {
171        // x87 FCW = 0x037F
172        self.data[0] = 0x7F;
173        self.data[1] = 0x03;
174        // MXCSR = 0x1F80
175        self.data[24] = 0x80;
176        self.data[25] = 0x1F;
177    }
178
179    /// Copy the state from another `ExtendedState`.
180    pub fn copy_from(&mut self, other: &ExtendedState) {
181        let len = other.size.min(self.size);
182        self.data[..len].copy_from_slice(&other.data[..len]);
183    }
184}

308/// CPU context saved/restored during context switches.
309///
310/// Only stores the saved RSP. All callee-saved registers (rbx, rbp, r12-r15)
311/// are pushed onto the task's kernel stack by `switch_context()`.
312#[repr(C)]
313pub struct CpuContext {
314    /// Saved stack pointer (points into the task's kernel stack)
315    pub saved_rsp: u64,
316}
317
318impl CpuContext {
319    /// Create a new CPU context for a task starting at the given entry point.
320    ///
321    /// Sets up a fake stack frame on the kernel stack that looks like
322    /// `switch_context()` just pushed callee-saved registers. When
323    /// `switch_context()` or `restore_first_task()` pops them and does `ret`,
324    /// it will jump to `task_entry_trampoline`, which enables interrupts
325    /// and jumps to the real entry point (stored in r12).
326    ///
327    /// Stack layout (growing downward):
328    /// ```text
329    /// [stack_top]
330    ///   task_entry_trampoline   <- ret target
331    ///   0  (r15)
332    ///   0  (r14)
333    ///   0  (r13)
334    ///   entry_point (r12)      <- trampoline reads this
335    ///   0  (rbp)
336    ///   0  (rbx)
337    ///   <- saved_rsp points here
338    /// ```
339    pub fn new(entry_point: u64, kernel_stack: &KernelStack) -> Self {
340        let stack_top = kernel_stack.virt_base.as_u64() + kernel_stack.size as u64;
341
342        // We need to push 7 values (each 8 bytes) onto the stack
343        let initial_rsp = stack_top - 7 * 8;
344
345        // SAFETY: We own this stack memory and it's properly allocated and zeroed.
346        // The stack region [virt_base, virt_base + size) is valid.
347        unsafe {
348            let stack = initial_rsp as *mut u64;
349            // Push order must match switch_context pops (LIFO, but we write linearly from RSP up):
350            // [RSP+0]  = r15
351            // [RSP+8]  = r14
352            // [RSP+16] = r13
353            // [RSP+24] = r12 (entry point)
354            // [RSP+32] = rbp
355            // [RSP+40] = rbx
356            // [RSP+48] = ret (trampoline)
357            *stack.add(0) = 0; // r15
358            *stack.add(1) = 0; // r14
359            *stack.add(2) = 0; // r13
360            *stack.add(3) = entry_point; // r12 (trampoline target)
361            *stack.add(4) = 0; // rbp
362            *stack.add(5) = 0; // rbx
363            *stack.add(6) = task_entry_trampoline as *const () as u64; // ret address
364        }
365
366        CpuContext {
367            saved_rsp: initial_rsp,
368        }
369    }
370}

372/// Trampoline for newly created tasks.
373///
374/// When a new task is first scheduled, `switch_context()` pops the fake
375/// callee-saved registers and `ret`s here. We enable interrupts (the new
376/// task starts with IF=0 because the scheduler disables interrupts) and
377/// jump to the real entry point stored in r12.
378#[unsafe(naked)]
379unsafe extern "C" fn task_entry_trampoline() {
380    core::arch::naked_asm!(
381        "call {finish_switch}",
382        "sti",          // Enable interrupts for the new task
383        "call {mark_tlb_ready}",
384        "mov rdi, r13", // Bootstrap arg (seeded cap handle)
385        "jmp r12",      // Jump to real entry point (loaded from initial stack frame)
386        finish_switch = sym crate::process::scheduler::finish_switch,
387        mark_tlb_ready = sym crate::arch::x86_64::percpu::mark_tlb_ready_current,
388    );
389}

391/// Kernel stack for a task
392pub struct KernelStack {
393    /// Physical address of the stack
394    pub base: PhysAddr,
395    /// Virtual address of the stack
396    pub virt_base: VirtAddr,
397    /// Size of the stack
398    pub size: usize,
399}
400
401impl KernelStack {
402    /// Allocate a new kernel stack using the buddy allocator
403    pub fn allocate(size: usize) -> Result<Self, &'static str> {
404        use crate::memory::{get_allocator, FrameAllocator};
405
406        // Calculate number of pages needed (round up)
407        let pages = (size + 4095) / 4096;
408        let order = pages.next_power_of_two().trailing_zeros() as u8;
409
410        // Allocate physical frames from buddy allocator
411        let mut lock = get_allocator().lock();
412        let allocator = lock.as_mut().ok_or("Allocator not initialized")?;
413        let frame = allocator
414            .alloc(order)
415            .map_err(|_| "Failed to allocate kernel stack")?;
416        drop(lock);
417
418        let phys_base = frame.start_address;
419        let virt_base = VirtAddr::new(crate::memory::phys_to_virt(phys_base.as_u64()));
420
421        // Zero out the stack for safety
422        unsafe {
423            core::ptr::write_bytes(virt_base.as_mut_ptr::<u8>(), 0, size);
424        }
425
426        Ok(KernelStack {
427            base: phys_base,
428            virt_base,
429            size,
430        })
431    }
432}
433
434impl Drop for KernelStack {
435    /// Performs the drop operation.
436    fn drop(&mut self) {
437        use crate::memory::{frame::PhysFrame, get_allocator, FrameAllocator};
438
439        let pages = (self.size + 4095) / 4096;
440        let order = pages.next_power_of_two().trailing_zeros() as u8;
441        let frame = PhysFrame {
442            start_address: self.base,
443        };
444
445        if let Some(ref mut allocator) = *get_allocator().lock() {
446            allocator.free(frame, order);
447        }
448    }
449}

624/// Context switch dispatcher. Picks the xsave or fxsave path based on host
625/// capabilities, then performs the full save/swap/restore sequence.
626///
627/// # Safety
628/// Caller must ensure all pointers in `target` are valid and interrupts are disabled.
629pub(super) unsafe fn do_switch_context(target: &super::scheduler::SwitchTarget) {
630    if crate::arch::x86_64::cpuid::host_uses_xsave() {
631        switch_context_xsave(
632            target.old_rsp_ptr,
633            target.new_rsp_ptr,
634            target.old_fpu_ptr,
635            target.new_fpu_ptr,
636            target.new_xcr0,
637        );
638    } else {
639        switch_context_fxsave(
640            target.old_rsp_ptr,
641            target.new_rsp_ptr,
642            target.old_fpu_ptr,
643            target.new_fpu_ptr,
644        );
645    }
646}

648/// First-task restore dispatcher. Like `do_switch_context` but without
649/// saving old state (there is no previous task).
650///
651/// # Safety
652/// Caller must ensure pointers are valid and interrupts are disabled. Never returns.
653pub(super) unsafe fn do_restore_first_task(
654    rsp_ptr: *const u64,
655    fpu_ptr: *const u8,
656    xcr0: u64,
657) -> ! {
658    if crate::arch::x86_64::cpuid::host_uses_xsave() {
659        restore_first_task_xsave(rsp_ptr, fpu_ptr, xcr0);
660    } else {
661        restore_first_task_fxsave(rsp_ptr, fpu_ptr);
662    }
663    core::hint::unreachable_unchecked()
664}

666// ── FXSAVE path (legacy, no XSAVE support) ──
667
668/// rdi=old_rsp, rsi=new_rsp, rdx=old_fpu, rcx=new_fpu
669#[unsafe(naked)]
670unsafe extern "C" fn switch_context_fxsave(
671    _old_rsp_ptr: *mut u64,
672    _new_rsp_ptr: *const u64,
673    _old_fpu_ptr: *mut u8,
674    _new_fpu_ptr: *const u8,
675) {
676    core::arch::naked_asm!(
677        "fxsave [rdx]",
678        "push rbx",
679        "push rbp",
680        "push r12",
681        "push r13",
682        "push r14",
683        "push r15",
684        "mov [rdi], rsp",
685        "mov rsp, [rsi]",
686        "pop r15",
687        "pop r14",
688        "pop r13",
689        "pop r12",
690        "pop rbp",
691        "pop rbx",
692        "fxrstor [rcx]",
693        "ret",
694    );
695}

697/// rdi=rsp_ptr, rsi=fpu_ptr
698#[unsafe(naked)]
699unsafe extern "C" fn restore_first_task_fxsave(
700    _rsp_ptr: *const u64,
701    _fpu_ptr: *const u8,
702) {
703    core::arch::naked_asm!(
704        "mov rsp, [rdi]",
705        "pop r15",
706        "pop r14",
707        "pop r13",
708        "pop r12",
709        "pop rbp",
710        "pop rbx",
711        "fxrstor [rsi]",
712        "ret",
713    );
714}
715
716// ── XSAVE path (with XCR0 switching per-silo) ──
717
718/// rdi=old_rsp, rsi=new_rsp, rdx=old_fpu, rcx=new_fpu, r8=new_xcr0
719#[unsafe(naked)]
720unsafe extern "C" fn switch_context_xsave(
721    _old_rsp_ptr: *mut u64,
722    _new_rsp_ptr: *const u64,
723    _old_fpu_ptr: *mut u8,
724    _new_fpu_ptr: *const u8,
725    _new_xcr0: u64,
726) {
727    core::arch::naked_asm!(
728        // xsave clobbers edx — save old_fpu_ptr in r10 first
729        "mov r10, rdx",
730        "mov eax, 0xFFFFFFFF",
731        "mov edx, 0xFFFFFFFF",
732        "xsave [r10]",
733        "push rbx",
734        "push rbp",
735        "push r12",
736        "push r13",
737        "push r14",
738        "push r15",
739        "mov [rdi], rsp",
740        "mov rsp, [rsi]",
741        "pop r15",
742        "pop r14",
743        "pop r13",
744        "pop r12",
745        "pop rbp",
746        "pop rbx",
747        // Switch XCR0 to new task's silo mask (r8 = new_xcr0)
748        "push rcx",
749        "mov ecx, 0",
750        "mov eax, r8d",
751        "shr r8, 32",
752        "mov edx, r8d",
753        "xsetbv",
754        "pop rcx",
755        // Restore new FPU state
756        "mov eax, 0xFFFFFFFF",
757        "mov edx, 0xFFFFFFFF",
758        "xrstor [rcx]",
759        "ret",
760    );
761}
762
763/// rdi=rsp_ptr, rsi=fpu_ptr, rdx=xcr0
764#[unsafe(naked)]
765unsafe extern "C" fn restore_first_task_xsave(
766    _rsp_ptr: *const u64,
767    _fpu_ptr: *const u8,
768    _xcr0: u64,
769) {
770    core::arch::naked_asm!(
771        "mov rsp, [rdi]",
772        "pop r15",
773        "pop r14",
774        "pop r13",
775        "pop r12",
776        "pop rbp",
777        "pop rbx",
778        // Set XCR0 for the first task
779        "push rsi",
780        "mov ecx, 0",
781        "mov eax, edx",
782        "shr rdx, 32",
783        "xsetbv",
784        "pop rsi",
785        // Restore FPU
786        "mov eax, 0xFFFFFFFF",
787        "mov edx, 0xFFFFFFFF",
788        "xrstor [rsi]",
789        "ret",
790    );
