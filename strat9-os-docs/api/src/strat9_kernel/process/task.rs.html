27/// Unique identifier for a task

31impl TaskId {

33    pub fn new() -> Self {
34        static NEXT_ID: AtomicU64 = AtomicU64::new(0);
35        TaskId(NEXT_ID.fetch_add(1, Ordering::SeqCst))

38    /// Get the raw u64 value

101pub struct FpuState {
102    pub data: [u8; 512],
103}
104
105impl FpuState {
106    pub const fn new() -> Self {
107        let mut data = [0; 512];
108        // Default x87 FPU Control Word (FCW) = 0x037F
109        data[0] = 0x7F;
110        data[1] = 0x03;
111        // Default MXCSR = 0x1F80
112        data[24] = 0x80;
113        data[25] = 0x1F;
114        Self { data }
115    }

237/// CPU context saved/restored during context switches.
238///
239/// Only stores the saved RSP. All callee-saved registers (rbx, rbp, r12-r15)
240/// are pushed onto the task's kernel stack by `switch_context()`.
241#[repr(C)]
242pub struct CpuContext {
243    /// Saved stack pointer (points into the task's kernel stack)
244    pub saved_rsp: u64,
245}
246
247impl CpuContext {
248    /// Create a new CPU context for a task starting at the given entry point.
249    ///
250    /// Sets up a fake stack frame on the kernel stack that looks like
251    /// `switch_context()` just pushed callee-saved registers. When
252    /// `switch_context()` or `restore_first_task()` pops them and does `ret`,
253    /// it will jump to `task_entry_trampoline`, which enables interrupts
254    /// and jumps to the real entry point (stored in r12).
255    ///
256    /// Stack layout (growing downward):
257    /// ```text
258    /// [stack_top]
259    ///   task_entry_trampoline   <- ret target
260    ///   0  (r15)
261    ///   0  (r14)
262    ///   0  (r13)
263    ///   entry_point (r12)      <- trampoline reads this
264    ///   0  (rbp)
265    ///   0  (rbx)
266    ///   <- saved_rsp points here
267    /// ```
268    pub fn new(entry_point: u64, kernel_stack: &KernelStack) -> Self {
269        let stack_top = kernel_stack.virt_base.as_u64() + kernel_stack.size as u64;
270
271        // We need to push 7 values (each 8 bytes) onto the stack
272        let initial_rsp = stack_top - 7 * 8;
273
274        // SAFETY: We own this stack memory and it's properly allocated and zeroed.
275        // The stack region [virt_base, virt_base + size) is valid.
276        unsafe {
277            let stack = initial_rsp as *mut u64;
278            // Push order must match switch_context pops (LIFO, but we write linearly from RSP up):
279            // [RSP+0]  = r15
280            // [RSP+8]  = r14
281            // [RSP+16] = r13
282            // [RSP+24] = r12 (entry point)
283            // [RSP+32] = rbp
284            // [RSP+40] = rbx
285            // [RSP+48] = ret (trampoline)
286            *stack.add(0) = 0; // r15
287            *stack.add(1) = 0; // r14
288            *stack.add(2) = 0; // r13
289            *stack.add(3) = entry_point; // r12 (trampoline target)
290            *stack.add(4) = 0; // rbp
291            *stack.add(5) = 0; // rbx
292            *stack.add(6) = task_entry_trampoline as *const () as u64; // ret address
293        }
294
295        CpuContext {
296            saved_rsp: initial_rsp,
297        }
298    }
299}

301/// Trampoline for newly created tasks.
302///
303/// When a new task is first scheduled, `switch_context()` pops the fake
304/// callee-saved registers and `ret`s here. We enable interrupts (the new
305/// task starts with IF=0 because the scheduler disables interrupts) and
306/// jump to the real entry point stored in r12.
307#[unsafe(naked)]
308unsafe extern "C" fn task_entry_trampoline() {
309    core::arch::naked_asm!(
310        "call {finish_switch}",
311        "sti",          // Enable interrupts for the new task
312        "call {mark_tlb_ready}",
313        "mov rdi, r13", // Bootstrap arg (seeded cap handle)
314        "jmp r12",      // Jump to real entry point (loaded from initial stack frame)
315        finish_switch = sym crate::process::scheduler::finish_switch,
316        mark_tlb_ready = sym crate::arch::x86_64::percpu::mark_tlb_ready_current,
317    );
318}

320/// Kernel stack for a task
321pub struct KernelStack {
322    /// Physical address of the stack
323    pub base: PhysAddr,
324    /// Virtual address of the stack
325    pub virt_base: VirtAddr,
326    /// Size of the stack
327    pub size: usize,
328}
329
330impl KernelStack {
331    /// Allocate a new kernel stack using the buddy allocator
332    pub fn allocate(size: usize) -> Result<Self, &'static str> {
333        use crate::memory::{get_allocator, FrameAllocator};
334
335        // Calculate number of pages needed (round up)
336        let pages = (size + 4095) / 4096;
337        let order = pages.next_power_of_two().trailing_zeros() as u8;
338
339        // Allocate physical frames from buddy allocator
340        let mut lock = get_allocator().lock();
341        let allocator = lock.as_mut().ok_or("Allocator not initialized")?;
342        let frame = allocator
343            .alloc(order)
344            .map_err(|_| "Failed to allocate kernel stack")?;
345        drop(lock);
346
347        let phys_base = frame.start_address;
348        let virt_base = VirtAddr::new(crate::memory::phys_to_virt(phys_base.as_u64()));
349
350        // Zero out the stack for safety
351        unsafe {
352            core::ptr::write_bytes(virt_base.as_mut_ptr::<u8>(), 0, size);
353        }
354
355        Ok(KernelStack {
356            base: phys_base,
357            virt_base,
358            size,
359        })
360    }
361}
362
363impl Drop for KernelStack {
364    fn drop(&mut self) {
365        use crate::memory::{frame::PhysFrame, get_allocator, FrameAllocator};
366
367        let pages = (self.size + 4095) / 4096;
368        let order = pages.next_power_of_two().trailing_zeros() as u8;
369        let frame = PhysFrame {
370            start_address: self.base,
371        };
372
373        if let Some(ref mut allocator) = *get_allocator().lock() {
374            allocator.free(frame, order);
375        }
376    }
377}

550/// Low-level context switch: save callee-saved registers on current stack,
551/// swap RSP, restore callee-saved registers from new stack, and `ret`.
552///
553/// Arguments (x86-64 SysV ABI):
554/// - rdi = pointer to old task's `saved_rsp` field
555/// - rsi = pointer to new task's `saved_rsp` field
556/// - rdx = pointer to old task's FPU state
557/// - rcx = pointer to new task's FPU state
558///
559/// The caller (scheduler) has already handled TSS.rsp0 and CR3 switching.
560#[unsafe(naked)]
561pub(super) unsafe extern "C" fn switch_context(
562    _old_rsp_ptr: *mut u64,
563    _new_rsp_ptr: *const u64,
564    _old_fpu_ptr: *mut FpuState,
565    _new_fpu_ptr: *const FpuState,
566) {
567    core::arch::naked_asm!(
568        // Save FPU state
569        "fxsave [rdx]",
570        // Save callee-saved registers on current stack
571        "push rbx",
572        "push rbp",
573        "push r12",
574        "push r13",
575        "push r14",
576        "push r15",
577        // Save current RSP into old context
578        "mov [rdi], rsp",
579        // Load new RSP from new context
580        "mov rsp, [rsi]",
581        // Restore callee-saved registers from new stack
582        "pop r15",
583        "pop r14",
584        "pop r13",
585        "pop r12",
586        "pop rbp",
587        "pop rbx",
588        // Restore FPU state
589        "fxrstor [rcx]",
590        // Return to wherever the new task left off
591        // (or to task_entry_trampoline for new tasks)
592        "ret",
593    );
594}

596/// Restore context for the very first task (no old context to save).
597///
598/// Argument (x86-64 SysV ABI):
599/// - rdi = pointer to the first task's `saved_rsp` field
600/// - rsi = pointer to the first task's FPU state
601#[unsafe(naked)]
602pub(super) unsafe extern "C" fn restore_first_task(
603    _rsp_ptr: *const u64,
604    _fpu_ptr: *const FpuState,
605) {
606    core::arch::naked_asm!(
607        // Load RSP from context
608        "mov rsp, [rdi]",
609        // Restore callee-saved registers
610        "pop r15",
611        "pop r14",
612        "pop r13",
613        "pop r12",
614        "pop rbp",
615        "pop rbx",
616        // Restore FPU state
617        "fxrstor [rsi]",
618        // Jump to task_entry_trampoline -> real entry point
619        "ret",
620    );
