15const AT_NULL: u64 = 0;
16const AT_PHDR: u64 = 3;
17const AT_PHENT: u64 = 4;
18const AT_PHNUM: u64 = 5;
19const AT_PAGESZ: u64 = 6;
20const AT_BASE: u64 = 7;
21const AT_ENTRY: u64 = 9;
22const AT_RANDOM: u64 = 25;
23const AT_EXECFN: u64 = 31;

25/// SYS_PROC_EXECVE (301): replace current process image.
26pub fn sys_execve(
27    frame: &mut SyscallFrame,
28    path_ptr: u64,
29    argv_ptr: u64,
30    envp_ptr: u64,
31) -> Result<u64, SyscallError> {
32    let current = current_task_clone().ok_or(SyscallError::PermissionDenied)?;
33
34    let mut path_buf = [0u8; 4096];
35    let path_slice = UserSliceRead::new(path_ptr, 4096).map_err(|_| SyscallError::Fault)?;
36
37    let mut len = 0;
38
39    loop {
40        if len >= 4096 {
41            return Err(SyscallError::ArgumentListTooLong);
42        } // Reused error code
43        let b = path_slice.read_u8(len).map_err(|_| SyscallError::Fault)?;
44        if b == 0 {
45            break;
46        }
47        path_buf[len] = b;
48        len += 1;
49    }
50    let path_str =
51        core::str::from_utf8(&path_buf[..len]).map_err(|_| SyscallError::InvalidArgument)?;
52
53    let fd = vfs::open(path_str, vfs::OpenFlags::READ)?;
54
55    // Read into memory
56    const MAX_EXEC_SIZE: usize = 64 * 1024 * 1024;
57    let mut elf_data = Vec::new();
58    let mut buf = [0u8; 4096];
59    loop {
60        match vfs::read(fd, &mut buf) {
61            Ok(n) => {
62                if n == 0 {
63                    break;
64                }
65                if elf_data.len() + n > MAX_EXEC_SIZE {
66                    let _ = vfs::close(fd);
67                    return Err(SyscallError::OutOfMemory);
68                }
69                elf_data.extend_from_slice(&buf[..n]);
70            }
71            Err(e) => {
72                let _ = vfs::close(fd);
73                return Err(e);
74            }
75        }
76    }
77    let _ = vfs::close(fd);
78
79    if elf_data.len() < 4 {
80        return Err(SyscallError::ExecFormatError);
81    }
82
83    let new_as = AddressSpace::new_user().map_err(|_| SyscallError::OutOfMemory)?;
84    let new_as_arc = alloc::sync::Arc::new(new_as);
85
86    let load_info =
87        load_elf_image(&elf_data, &new_as_arc).map_err(|_| SyscallError::ExecFormatError)?;
88
89    let stack_flags = VmaFlags {
90        readable: true,
91        writable: true,
92        executable: false,
93        user_accessible: true,
94    };
95    new_as_arc
96        .map_region(
97            USER_STACK_BASE,
98            USER_STACK_PAGES,
99            stack_flags,
100            VmaType::Stack,
101            VmaPageSize::Small,
102        )
103        .map_err(|_| SyscallError::OutOfMemory)?;
104
105    let sp = setup_user_stack(
106        &new_as_arc,
107        argv_ptr,
108        envp_ptr,
109        &load_info,
110        path_str.as_bytes(),
111    )?;
112
113    // TLS setup (Variant II) if the ELF has a PT_TLS segment.
114    let mut new_fs_base = 0u64;
115    if load_info.tls_memsz > 0 {
116        let tls_align = core::cmp::max(load_info.tls_align, 8).next_power_of_two();
117        let aligned_memsz = (load_info.tls_memsz + tls_align - 1) & !(tls_align - 1);
118        let total_size = aligned_memsz + 8;
119        let n_pages = ((total_size + 4095) / 4096) as usize;
120        let tls_flags = VmaFlags {
121            readable: true,
122            writable: true,
123            executable: false,
124            user_accessible: true,
125        };
126        let tls_base = new_as_arc
127            .find_free_vma_range(0x7FFF_E000_0000, n_pages, VmaPageSize::Small)
128            .ok_or(SyscallError::OutOfMemory)?;
129        new_as_arc
130            .map_region(tls_base, n_pages, tls_flags, VmaType::Anonymous, VmaPageSize::Small)
131            .map_err(|_| SyscallError::OutOfMemory)?;
132        if load_info.tls_filesz > 0 && load_info.tls_vaddr != 0 {
133            let src_vaddr = load_info.tls_vaddr;
134            let mut off = 0u64;
135            let mut tmp = [0u8; 256];
136            while off < load_info.tls_filesz {
137                let chunk = core::cmp::min(256, (load_info.tls_filesz - off) as usize);
138                crate::process::elf::read_user_mapped_bytes_pub(&new_as_arc, src_vaddr + off, &mut tmp[..chunk])
139                    .map_err(|_| SyscallError::Fault)?;
140                crate::process::elf::write_user_mapped_bytes_pub(&new_as_arc, tls_base + off, &tmp[..chunk])
141                    .map_err(|_| SyscallError::Fault)?;
142                off += chunk as u64;
143            }
144        }
145        let tp = tls_base + aligned_memsz;
146        crate::process::elf::write_user_u64_pub(&new_as_arc, tp, tp)
147            .map_err(|_| SyscallError::Fault)?;
148        new_fs_base = tp;
149    }
150
151    // === EXECVE CLEANUP (POSIX semantics) ===
152    // Now that ELF is valid and loaded, perform cleanup before switching address space.
153
154    // 1. Close all file descriptors with CLOEXEC flag
155    unsafe {
156        let fd_table = &mut *current.process.fd_table.get();
157        fd_table.close_cloexec();
158    }
159
160    // 2. Reset all signal handlers to SIG_DFL
161    current.reset_signals();
162
163    // 3. Clear thread-local storage address and TID pointer : POSIX exec semantics.
164    current
165        .clear_child_tid
166        .store(0, core::sync::atomic::Ordering::Relaxed);
167    current
168        .user_fs_base
169        .store(new_fs_base, core::sync::atomic::Ordering::Relaxed);
170
171    // 4. Reset memory layout: brk and mmap_hint belong to the old image.
172    current
173        .process
174        .brk
175        .store(0, core::sync::atomic::Ordering::Relaxed);
176    current
177        .process
178        .mmap_hint
179        .store(0x0000_0000_6000_0000, core::sync::atomic::Ordering::Relaxed);
180    // Set FS.base MSR for the new image TLS (or 0 if no PT_TLS).
181    unsafe {
182        let lo = new_fs_base as u32;
183        let hi = (new_fs_base >> 32) as u32;
184        core::arch::asm!(
185            "mov ecx, 0xC0000100", // MSR_FS_BASE
186            "wrmsr",
187            in("eax") lo,
188            in("edx") hi,
189            options(nostack, preserves_flags),
190        );
191    }
192
193    let old_as =
194        unsafe { core::mem::replace(&mut *current.process.address_space.get(), new_as_arc.clone()) };
195
196    unsafe {
197        (&*current.process.address_space.get()).switch_to();
198    }
199
200    frame.iret_rip = load_info.runtime_entry;
201    frame.iret_rsp = sp;
202    frame.iret_rflags = 0x200; // IF=1, clean slate for the new image
203
204    frame.rdi = 0;
205    frame.rsi = 0;
206    frame.rdx = 0;
207    frame.rcx = 0;
208    frame.r8 = 0;
209    frame.r9 = 0;
210    frame.r10 = 0;
211    frame.r11 = 0;
212    frame.rbx = 0;
213    frame.rbp = 0;
214    frame.r12 = 0;
215    frame.r13 = 0;
216    frame.r14 = 0;
217    frame.r15 = 0;
218    frame.rax = 0;
219
220    // Safely drop the old address space now that the new CR3 is loaded
221    drop(old_as);
222
223    Ok(0)
224}

226/// Performs the setup user stack operation.
227fn setup_user_stack(
228    new_as: &AddressSpace,
229    argv_ptr: u64,
230    envp_ptr: u64,
231    elf_info: &LoadedElfInfo,
232    exec_path: &[u8],
233) -> Result<u64, SyscallError> {
234    let args = read_string_array(argv_ptr)?;
235    let envs = read_string_array(envp_ptr)?;
236
237    let mut sp = USER_STACK_TOP;
238    let mut str_ptrs: Vec<u64> = Vec::with_capacity(args.len()); // stores pointers to arguments
239    let mut env_ptrs: Vec<u64> = Vec::with_capacity(envs.len()); // stores pointers to env vars
240
241    // Push strings to stack (highest addresses)
242    // We push them in reverse order so they appear in memory roughly sequentially for cache locality?
243    // Actually standard is to put them at very top. Order doesn't strictly matter as long as pointers are correct.
244    // We'll push ENV strings first (highest), then ARG strings.
245
246    // Push ENV strings
247    for env in envs.iter().rev() {
248        let len = (env.len() + 1) as u64;
249        sp -= len;
250        write_bytes_to_as(new_as, sp, env)?;
251        write_bytes_to_as(new_as, sp + env.len() as u64, &[0])?;
252        env_ptrs.push(sp);
253    }
254    // env_ptrs: [ptr_to_highest_env, ptr_to_second_highest...] which corresponds to [env[last], env[last-1]...]
255    // Userspace expects envp[0] to point to first env string.
256    // So we need to reverse env_ptrs to match original order.
257    env_ptrs.reverse();
258
259    // Push ARG strings
260    for arg in args.iter().rev() {
261        let len = (arg.len() + 1) as u64;
262        sp -= len;
263        write_bytes_to_as(new_as, sp, arg)?;
264        write_bytes_to_as(new_as, sp + arg.len() as u64, &[0])?;
265        str_ptrs.push(sp);
266    }
267    str_ptrs.reverse();
268
269    // Push exec path (for AT_EXECFN).
270    let mut execfn_ptr = 0u64;
271    if !exec_path.is_empty() {
272        let len = (exec_path.len() + 1) as u64;
273        sp -= len;
274        write_bytes_to_as(new_as, sp, exec_path)?;
275        write_bytes_to_as(new_as, sp + exec_path.len() as u64, &[0])?;
276        execfn_ptr = sp;
277    }
278
279    // Push 16 bytes of random seed for AT_RANDOM (deterministic fallback source).
280    sp -= 16;
281    let rand_ptr = sp;
282    let seed = generate_aux_random_seed();
283    write_bytes_to_as(new_as, rand_ptr, &seed)?;
284
285    // Align SP to 16 bytes for System V ABI
286    sp &= !0xF;
287
288    // Phase 2: Push auxv, pointer arrays, then argc.
289    let size_ptr = 8u64;
290
291    // auxv entries end with AT_NULL.
292    let mut auxv: Vec<(u64, u64)> = Vec::with_capacity(10);
293    auxv.push((AT_PHDR, elf_info.phdr_vaddr));
294    auxv.push((AT_PHENT, elf_info.phent as u64));
295    auxv.push((AT_PHNUM, elf_info.phnum as u64));
296    auxv.push((AT_PAGESZ, 4096));
297    if let Some(base) = elf_info.interp_base {
298        auxv.push((AT_BASE, base));
299    }
300    auxv.push((AT_ENTRY, elf_info.program_entry));
301    auxv.push((AT_RANDOM, rand_ptr));
302    if execfn_ptr != 0 {
303        auxv.push((AT_EXECFN, execfn_ptr));
304    }
305
306    // AT_NULL terminator.
307    sp -= size_ptr;
308    write_u64_to_as(new_as, sp, 0)?;
309    sp -= size_ptr;
310    write_u64_to_as(new_as, sp, AT_NULL)?;
311    for &(key, val) in auxv.iter().rev() {
312        sp -= size_ptr;
313        write_u64_to_as(new_as, sp, val)?;
314        sp -= size_ptr;
315        write_u64_to_as(new_as, sp, key)?;
316    }
317
318    // Push ENVP array
319    // [NULL]
320    // [envp[n]]
321    // ...
322    // [envp[0]]
323    sp -= size_ptr;
324    write_u64_to_as(new_as, sp, 0)?; // NULL terminator
325
326    for &ptr in env_ptrs.iter().rev() {
327        sp -= size_ptr;
328        write_u64_to_as(new_as, sp, ptr)?;
329    }
330    // Note: sp now points to envp[0]
331
332    // Push ARGV array
333    // [NULL]
334    // [argv[n]]
335    // ...
336    // [argv[0]]
337    sp -= size_ptr;
338    write_u64_to_as(new_as, sp, 0)?; // NULL terminator
339
340    for &ptr in str_ptrs.iter().rev() {
341        sp -= size_ptr;
342        write_u64_to_as(new_as, sp, ptr)?;
343    }
344    // Note: sp now points to argv[0]
345
346    // Push ARGC
347    sp -= size_ptr;
348    write_u64_to_as(new_as, sp, args.len() as u64)?;
349
350    Ok(sp)
351}

353/// Reads string array.
354fn read_string_array(ptr: u64) -> Result<Vec<Vec<u8>>, SyscallError> {
355    let mut res = Vec::new();
356    if ptr == 0 {
357        return Ok(res);
358    }
359
360    let mut arr_off = 0;
361    loop {
362        // Read string pointer from user memory (current AS)
363        let str_ptr = match UserSliceRead::new(ptr + arr_off, 8) {
364            Ok(slice) => match slice.read_u64(0) {
365                Ok(p) => p,
366                Err(_) => return Err(SyscallError::Fault),
367            },
368            Err(_) => return Err(SyscallError::Fault),
369        };
370
371        if str_ptr == 0 {
372            break;
373        }
374        if res.len() > 1024 {
375            return Err(SyscallError::ArgumentListTooLong);
376        }
377
378        let mut s = Vec::new();
379        let mut i = 0;
380        loop {
381            if i > 4096 {
382                return Err(SyscallError::ArgumentListTooLong);
383            }
384            let b = match UserSliceRead::new(str_ptr + i, 1) {
385                Ok(slice) => match slice.read_u8(0) {
386                    Ok(byte) => byte,
387                    Err(_) => return Err(SyscallError::Fault),
388                },
389                Err(_) => return Err(SyscallError::Fault),
390            };
391            if b == 0 {
392                break;
393            }
394            s.push(b);
395            i += 1;
396        }
397        res.push(s);
398        arr_off += 8;
399    }
400    Ok(res)
401}

403/// Writes bytes to as.
404fn write_bytes_to_as(as_ref: &AddressSpace, vaddr: u64, data: &[u8]) -> Result<(), SyscallError> {
405    use x86_64::VirtAddr;
406    let mut written = 0;
407    // We assume data is small enough or we loop? Using unsafe pointer arithmetic.
408    // The `AddressSpace` methods like `translate` are needed.
409
410    // Since `load_elf_image` in `elf.rs` used `translate`, we should verify visibility.
411    // `AddressSpace` is usually public. `translate` is on `Mapper` trait?
412    // `AddressSpace` in `strat9` likely implements `Mapper` or has it.
413    // `elf.rs` used `user_as.translate(...)`.
414    
415    // I need to import Translate? `AddressSpace` usually has `translate`.
416
417    while written < data.len() {
418        let curr_vaddr = vaddr + written as u64;
419        let page_offset = (curr_vaddr & 0xFFF) as usize;
420        let chunk_size = core::cmp::min(data.len() - written, 4096 - page_offset);
421
422        // translate might fail if page not mapped.
423        // `USER_STACK_BASE`..`USER_STACK_TOP` is mapped.
424        let phys = as_ref
425            .translate(VirtAddr::new(curr_vaddr))
426            .ok_or(SyscallError::Fault)?;
427        let virt = crate::memory::phys_to_virt(phys.as_u64()) as *mut u8;
428
429        unsafe {
430            core::ptr::copy_nonoverlapping(data.as_ptr().add(written), virt, chunk_size);
431        }
432        written += chunk_size;
433    }
434    Ok(())
435}

437/// Writes u64 to as.
438fn write_u64_to_as(as_ref: &AddressSpace, vaddr: u64, val: u64) -> Result<(), SyscallError> {
439    let bytes = val.to_ne_bytes();
440    write_bytes_to_as(as_ref, vaddr, &bytes)
441}

443/// Performs the generate aux random seed operation.
444fn generate_aux_random_seed() -> [u8; 16] {
445    use x86_64::registers::control::Cr3;
446    let mut s = [0u8; 16];
447    let t = crate::process::scheduler::ticks();
448    let (cr3, _) = Cr3::read();
449    let x = t
450        ^ (cr3
451            .start_address()
452            .as_u64()
453            .wrapping_mul(0x9e37_79b9_7f4a_7c15));
454    s[..8].copy_from_slice(&x.to_le_bytes());
455    s[8..].copy_from_slice(&(x.rotate_left(17) ^ 0xa076_1d64_78bd_642f).to_le_bytes());
456    s
