7const POLLIN: i16 = 0x0001;
8const POLLOUT: i16 = 0x0004;
9const POLLNVAL: i16 = 0x0020;
10
11const POLLFD_SIZE: usize = 8; // sizeof(pollfd) = i32 + i16 + i16
12
13fn read_pollfd(buf: &[u8], i: usize) -> (i32, i16) {
14    let off = i * POLLFD_SIZE;
15    let fd = i32::from_le_bytes([buf[off], buf[off + 1], buf[off + 2], buf[off + 3]]);
16    let events = i16::from_le_bytes([buf[off + 4], buf[off + 5]]);
17    (fd, events)
18}
19
20fn write_revents(buf: &mut [u8], i: usize, revents: i16) {
21    let off = i * POLLFD_SIZE + 6;
22    let bytes = revents.to_le_bytes();
23    buf[off] = bytes[0];
24    buf[off + 1] = bytes[1];
25}

27pub fn sys_poll(fds_ptr: u64, nfds: u64, _timeout_ms: u64) -> Result<u64, SyscallError> {
28    if nfds == 0 {
29        return Ok(0);
30    }
31    if nfds > 1024 {
32        return Err(SyscallError::InvalidArgument);
33    }
34    let n = nfds as usize;
35    let byte_len = n * POLLFD_SIZE;
36
37    let task = current_task_clone().ok_or(SyscallError::Fault)?;
38    let fd_table = unsafe { &*task.process.fd_table.get() };
39
40    let reader = UserSliceRead::new(fds_ptr, byte_len)?;
41    let mut buf = alloc::vec![0u8; byte_len];
42    reader.copy_to(&mut buf);
43
44    let mut ready_count = 0u64;
45    for i in 0..n {
46        let (fd, events) = read_pollfd(&buf, i);
47        let revents = if fd < 0 {
48            0i16
49        } else {
50            match fd_table.get(fd as u32) {
51                Ok(_) => {
52                    let mut r = 0i16;
53                    if events & POLLIN != 0 { r |= POLLIN; }
54                    if events & POLLOUT != 0 { r |= POLLOUT; }
55                    r
56                }
57                Err(_) => POLLNVAL,
58            }
59        };
60        write_revents(&mut buf, i, revents);
61        if revents != 0 {
62            ready_count += 1;
63        }
64    }
65
66    let writer = UserSliceWrite::new(fds_ptr, byte_len)?;
67    writer.copy_from(&buf);
68
69    Ok(ready_count)
70}
