35/// Performs the trace dhcp frame operation.
36fn trace_dhcp_frame(tag: &str, frame: &[u8]) {
37    if DHCP_TRACE_LOG_BUDGET.fetch_sub(1, Ordering::Relaxed) == 0 {
38        return;
39    }
40    if frame.len() < 14 {
41        return;
42    }
43    let ethertype = u16::from_be_bytes([frame[12], frame[13]]);
44    if ethertype != 0x0800 {
45        return;
46    }
47    if frame.len() < 34 {
48        return;
49    }
50    let ip_hlen = ((frame[14] & 0x0f) as usize) * 4;
51    if ip_hlen < 20 || frame.len() < 14 + ip_hlen + 8 {
52        return;
53    }
54    if frame[23] != 17 {
55        return;
56    }
57    let udp = 14 + ip_hlen;
58    let src_port = u16::from_be_bytes([frame[udp], frame[udp + 1]]);
59    let dst_port = u16::from_be_bytes([frame[udp + 2], frame[udp + 3]]);
60    let is_dhcp = (src_port == 68 && dst_port == 67) || (src_port == 67 && dst_port == 68);
61    if !is_dhcp {
62        return;
63    }
64    let mut xid: u32 = 0;
65    let bootp = udp + 8;
66    if frame.len() >= bootp + 8 {
67        xid = u32::from_be_bytes([
68            frame[bootp + 4],
69            frame[bootp + 5],
70            frame[bootp + 6],
71            frame[bootp + 7],
72        ]);
73    }
74    crate::serial_println!(
75        "[dhcp-trace] {} src_port={} dst_port={} len={} xid=0x{:08x}",
76        tag,
77        src_port,
78        dst_port,
79        frame.len(),
80        xid
81    );
82}

269        ),
270        _ => {
271            log::warn!("Unknown syscall: {} (0x{:x})", syscall_num, syscall_num);
272            Err(SyscallError::NotImplemented)
273        }
274    };
275
276    match result {
277        Ok(val) => {
278            if syscall_num == SYS_PROC_FORK {
279                crate::serial_println!("[syscall] FORK returning Ok({})", val);
280            } else if syscall_num == SYS_PROC_WAITPID {

289                crate::serial_println!("[syscall] WAITPID returning err {:?}", e);
290            }
291            frame.rax = e.to_raw();
292        }
293    }
294
295    crate::process::signal::deliver_pending_signal(frame);
296
297    frame.rax
298}

360const HANDLE_EVENT_READABLE: u64 = 1 << 0;
361const HANDLE_EVENT_WRITABLE: u64 = 1 << 1;

363/// Performs the poll handle events operation.
364fn poll_handle_events(handle: u64) -> Result<u64, SyscallError> {
365    let task = current_task_clone().ok_or(SyscallError::PermissionDenied)?;
366    let caps = unsafe { &*task.process.capabilities.get() };
367    let cap = caps.get(CapId::from_raw(handle)).ok_or(SyscallError::BadHandle)?;
368
369    match cap.resource_type {
370        ResourceType::Semaphore => {
371            if !cap.permissions.read {
372                return Err(SyscallError::PermissionDenied);
373            }
374            let sem = semaphore::get_semaphore(SemId::from_u64(cap.resource as u64))
375                .ok_or(SyscallError::BadHandle)?;
376            if sem.is_destroyed() {
377                return Err(SyscallError::Pipe);
378            }
379            if sem.count() > 0 {
380                Ok(HANDLE_EVENT_READABLE)
381            } else {
382                Ok(0)
383            }
384        }
385        ResourceType::IpcPort => {
386            let port = port::get_port(PortId::from_u64(cap.resource as u64)).ok_or(SyscallError::BadHandle)?;
387            if port.is_destroyed() {
388                return Err(SyscallError::Pipe);
389            }
390            let mut events = 0u64;
391            if cap.permissions.read && port.has_messages() {
392                events |= HANDLE_EVENT_READABLE;
393            }
394            if cap.permissions.write && port.can_send() {
395                events |= HANDLE_EVENT_WRITABLE;
396            }
397            if events == 0 && !cap.permissions.read && !cap.permissions.write {
398                return Err(SyscallError::PermissionDenied);
399            }
400            Ok(events)
401        }
402        ResourceType::Channel => {
403            let chan =
404                channel::get_channel(ChanId::from_u64(cap.resource as u64)).ok_or(SyscallError::BadHandle)?;
405            if chan.is_destroyed() {
406                return Err(SyscallError::Pipe);
407            }
408            let mut events = 0u64;
409            if cap.permissions.read && !chan.is_empty() {
410                events |= HANDLE_EVENT_READABLE;
411            }
412            if cap.permissions.write && chan.can_send() {
413                events |= HANDLE_EVENT_WRITABLE;
414            }
415            if events == 0 && !cap.permissions.read && !cap.permissions.write {
416                return Err(SyscallError::PermissionDenied);
417            }
418            Ok(events)
419        }
420        ResourceType::SharedRing => {
421            let _ = shared_ring::get_ring(RingId::from_u64(cap.resource as u64)).ok_or(SyscallError::BadHandle)?;
422            let mut events = 0u64;
423            if cap.permissions.read {
424                events |= HANDLE_EVENT_READABLE;
425            }
426            if cap.permissions.write {
427                events |= HANDLE_EVENT_WRITABLE;
428            }
429            if events == 0 {
430                return Err(SyscallError::PermissionDenied);
431            }
432            Ok(events)
433        }
434        _ => Err(SyscallError::NotSupported),
435    }
436}

438/// Performs the sys handle wait operation.
439fn sys_handle_wait(handle: u64, timeout_ns: u64) -> Result<u64, SyscallError> {
440    crate::silo::enforce_cap_for_current_task(handle)?;
441
442    let check_ready = || -> Result<u64, SyscallError> {
443        let events = poll_handle_events(handle)?;
444        if events != 0 {
445            Ok(events)
446        } else {
447            Err(SyscallError::Again)
448        }
449    };
450
451    if timeout_ns == 0 {
452        return check_ready();
453    }
454
455    let deadline = if timeout_ns == u64::MAX {
456        None
457    } else {
458        Some(crate::syscall::time::current_time_ns().saturating_add(timeout_ns))
459    };
460
461    loop {
462        if let Ok(events) = check_ready() {
463            return Ok(events);
464        }
465
466        if crate::process::has_pending_signals() {
467            return Err(SyscallError::Interrupted);
468        }
469
470        let now = crate::syscall::time::current_time_ns();
471        let wake_ns = if let Some(deadline_ns) = deadline {
472            if now >= deadline_ns {
473                return Err(SyscallError::TimedOut);
474            }
475            core::cmp::min(deadline_ns, now.saturating_add(10_000_000))
476        } else {
477            now.saturating_add(10_000_000)
478        };
479
480        if let Some(task) = current_task_clone() {
481            task.wake_deadline_ns.store(wake_ns, Ordering::Relaxed);
482        }
483        crate::process::block_current_task();
484        if let Some(task) = current_task_clone() {
485            task.wake_deadline_ns.store(0, Ordering::Relaxed);
486        }
487    }
488}

491fn sys_handle_grant(handle: u64, target_pid: u64) -> Result<u64, SyscallError> {
492    crate::silo::enforce_cap_for_current_task(handle)?;
493    crate::silo::enforce_silo_may_grant()?;
494    let pid = u32::try_from(target_pid).map_err(|_| SyscallError::InvalidArgument)?;
495
496    let source = current_task_clone().ok_or(SyscallError::PermissionDenied)?;
497    let granted = {
498        let source_caps = unsafe { &*source.process.capabilities.get() };
499        let cap = source_caps
500            .get(CapId::from_raw(handle))
501            .ok_or(SyscallError::BadHandle)?;
502        if !cap.permissions.grant {
503            return Err(SyscallError::PermissionDenied);
504        }
505        let mut dup = cap.clone();
506        dup.id = CapId::new();
507        dup
508    };
509
510    let target = crate::process::get_task_by_pid(pid).ok_or(SyscallError::NotFound)?;
511    let target_caps = unsafe { &mut *target.process.capabilities.get() };
512    let new_id = target_caps.insert(granted);
513    Ok(new_id.as_u64())
514}

612/// SYS_SIGPROCMASK (321): Examine and change blocked signals.
613///
614/// arg1 = how (0=BLOCK, 1=UNBLOCK, 2=SETMASK), arg2 = set_ptr (new mask), arg3 = oldset_ptr (old mask out)
615fn sys_sigprocmask(how: i32, set_ptr: u64, oldset_ptr: u64) -> Result<u64, SyscallError> {
616    use crate::process::current_task_clone;
617
618    const SIG_BLOCK: i32 = 0;
619    const SIG_UNBLOCK: i32 = 1;
620    const SIG_SETMASK: i32 = 2;
621
622    let task = current_task_clone().ok_or(SyscallError::PermissionDenied)?;
623
624    let blocked = &task.blocked_signals;
625
626    if oldset_ptr != 0 {
627        let old_mask = blocked.get_mask();
628        let user = UserSliceWrite::new(oldset_ptr, 8)?;
629        user.copy_from(&old_mask.to_ne_bytes());
630    }
631
632    if set_ptr != 0 {
633        let user = UserSliceRead::new(set_ptr, 8)?;
634        let mut buf = [0u8; 8];
635        user.copy_to(&mut buf);
636        let new_mask = u64::from_ne_bytes(buf);
637
638        let old_mask = blocked.get_mask();
639        let updated_mask = match how {
640            SIG_BLOCK => old_mask | new_mask,
641            SIG_UNBLOCK => old_mask & !new_mask,
642            SIG_SETMASK => new_mask,
643            _ => return Err(SyscallError::InvalidArgument),
644        };
645
646        blocked.set_mask(updated_mask);
647    }
648
649    Ok(0)
650}

783/// Performs the resolve volume device operation.
784fn resolve_volume_device(
785    handle: u64,
786    required: CapPermissions,
787) -> Result<VolumeDeviceRef, SyscallError> {
788    crate::silo::enforce_cap_for_current_task(handle)?;
789    let task = current_task_clone().ok_or(SyscallError::PermissionDenied)?;
790    let caps = unsafe { &*task.process.capabilities.get() };
791    let cap = caps
792        .get_with_permissions(CapId::from_raw(handle), required)
793        .ok_or(SyscallError::PermissionDenied)?;
794    if cap.resource_type != ResourceType::Volume {
795        return Err(SyscallError::BadHandle);
796    }
797
798    if let Some(device) = virtio_block::get_device() {
799        let device_ptr = device as *const virtio_block::VirtioBlockDevice as usize;
800        if cap.resource == device_ptr {
801            return Ok(VolumeDeviceRef::Virtio(device));
802        }
803    }
804    if let Some(device) = ahci::get_device() {
805        let device_ptr = device as *const ahci::AhciController as usize;
806        if cap.resource == device_ptr {
807            return Ok(VolumeDeviceRef::Ahci(device));
808        }
809    }
810    Err(SyscallError::BadHandle)
811}

813/// Performs the sys volume read operation.
814fn sys_volume_read(
815    handle: u64,
816    sector: u64,
817    buf_ptr: u64,
818    sector_count: u64,
819) -> Result<u64, SyscallError> {
820    if sector_count == 0 {
821        return Ok(0);
822    }
823    if sector_count > MAX_SECTORS_PER_CALL {
824        return Err(SyscallError::InvalidArgument);
825    }
826
827    let required = CapPermissions {
828        read: true,
829        write: false,
830        execute: false,
831        grant: false,
832        revoke: false,
833    };
834    let device = resolve_volume_device(handle, required)?;
835    let total_sectors = device.sector_count();
836    if sector >= total_sectors || sector.saturating_add(sector_count) > total_sectors {
837        return Err(SyscallError::InvalidArgument);
838    }
839
840    let mut kbuf = [0u8; SECTOR_SIZE];
841    for i in 0..sector_count {
842        let cur_sector = sector.checked_add(i).ok_or(SyscallError::InvalidArgument)?;
843        device.read_sector(cur_sector, &mut kbuf)?;
844        let offset = (i as usize)
845            .checked_mul(SECTOR_SIZE)
846            .ok_or(SyscallError::InvalidArgument)?;
847        let ptr = buf_ptr
848            .checked_add(offset as u64)
849            .ok_or(SyscallError::Fault)?;
850        let user = UserSliceWrite::new(ptr, SECTOR_SIZE)?;
851        user.copy_from(&kbuf);
852    }
853
854    Ok(sector_count)
855}

1303/// Performs the sys ipc bind port operation.
1304fn sys_ipc_bind_port(port: u64, _path_ptr: u64, _path_len: u64) -> Result<u64, SyscallError> {
1305    crate::silo::enforce_registry_bind_for_current_task()?;
1306    crate::silo::enforce_cap_for_current_task(port)?;
1307    if _path_ptr == 0 || _path_len == 0 {
1308        return Err(SyscallError::Fault);
1309    }
1310    const MAX_PATH_LEN: usize = 4096;
1311    if _path_len as usize > MAX_PATH_LEN {
1312        return Err(SyscallError::InvalidArgument);
1313    }
1314    let user = UserSliceRead::new(_path_ptr, _path_len as usize)?;
1315    let bytes = user.read_to_vec();
1316    let path = core::str::from_utf8(&bytes).map_err(SyscallError::from)?;
1317
1318    let task = current_task_clone().ok_or(SyscallError::PermissionDenied)?;
1319    let caps = unsafe { &*task.process.capabilities.get() };
1320    let cap = caps
1321        .get_with_permissions(
1322            CapId::from_raw(port),
1323            CapPermissions {
1324                read: true,
1325                write: true,
1326                execute: false,
1327                grant: true,
1328                revoke: false,
1329            },
1330        )
1331        .ok_or(SyscallError::PermissionDenied)?;
1332    if cap.resource_type != ResourceType::IpcPort {
1333        return Err(SyscallError::BadHandle);
1334    }
1335
1336    crate::vfs::mount(
1337        path,
1338        Arc::new(crate::vfs::IpcScheme::new(PortId::from_u64(
1339            cap.resource as u64,
1340        ))),
1341    )?;
1342    let _ = crate::namespace::bind(path, cap.resource as u64);
1343    let _ = crate::silo::set_current_silo_label_from_path(path);
1344
1345    // Bootstrap convenience: if a privileged userspace server binds root `/`
1346    // or a strate mountpoint, queue a bootstrap message.
1347    //
1348    // Volume capability seeding is optional (depends on a block device), but
1349    // bootstrap delivery must still happen so strate servers can bind their
1350    // label alias (`/srv/strate-fs-*/<label>`).
1351    let should_bootstrap = path == "/" || path.starts_with("/srv/strate-fs-");
1352    if should_bootstrap {
1353        let mut seeded_handle: u32 = 0;
1354        if let Some(device) = crate::hardware::storage::virtio_block::get_device() {
1355            let volume_resource = device as *const _ as usize;
1356            let volume_perms = CapPermissions {
1357                read: true,
1358                write: true,
1359                execute: false,
1360                grant: true,
1361                revoke: true,
1362            };
1363            let volume_cap = crate::capability::get_capability_manager().create_capability(
1364                ResourceType::Volume,
1365                volume_resource,
1366                volume_perms,
1367            );
1368            let task_caps = unsafe { &mut *task.process.capabilities.get() };
1369            let id = task_caps.insert(volume_cap);
1370            let _ = crate::silo::register_current_task_granted_resource(
1371                ResourceType::Volume,
1372                volume_resource,
1373                volume_perms,
1374            );
1375            if id.as_u64() <= u32::MAX as u64 {
1376                seeded_handle = id.as_u64() as u32;
1377            }
1378            log::info!(
1379                "ipc_bind_port('/'): seeded volume capability handle={} for task {:?}",
1380                id.as_u64(),
1381                task.id
1382            );
1383        }
1384
1385        // Send a bootstrap message to the just-bound filesystem server.
1386        // Message format:
1387        // - msg_type = 0x10
1388        // - flags = optional volume capability handle (0 if none)
1389        // - payload[0] = label length (u8)
1390        // - payload[1..] = UTF-8 label bytes (truncated to fit)
1391        const BOOTSTRAP_MSG_TYPE: u32 = 0x10;
1392        let mut boot_msg = IpcMessage::new(BOOTSTRAP_MSG_TYPE);
1393        // Use the bound task as sender so capability transfer path can
1394        // duplicate `flags` from a valid capability table when present.
1395        boot_msg.sender = task.id.as_u64();
1396        boot_msg.flags = seeded_handle;
1397        let label_owned = crate::silo::current_task_silo_label().unwrap_or_else(|| {
1398            if path == "/" {
1399                alloc::string::String::from("root")
1400            } else {
1401                alloc::string::String::from(
1402                    path.rsplit('/')
1403                        .find(|part| !part.is_empty())
1404                        .unwrap_or("default"),
1405                )
1406            }
1407        });
1408        let label_bytes = label_owned.as_bytes();
1409        let max_len = boot_msg.payload.len().saturating_sub(1);
1410        let copy_len = core::cmp::min(label_bytes.len(), max_len);
1411        boot_msg.payload[0] = copy_len as u8;
1412        if copy_len > 0 {
1413            boot_msg.payload[1..1 + copy_len].copy_from_slice(&label_bytes[..copy_len]);
1414        }
1415
1416        let port_id = PortId::from_u64(cap.resource as u64);
1417        if let Some(p) = port::get_port(port_id) {
1418            if p.send(boot_msg).is_ok() {
1419                log::info!(
1420                    "ipc_bind_port('{}'): queued bootstrap message (handle={}, label={})",
1421                    path,
1422                    seeded_handle,
1423                    label_owned
1424                );
1425            } else {
1426                log::warn!("ipc_bind_port('{}'): failed to queue bootstrap message", path);
1427            }
1428        } else {
1429            log::warn!(
1430                "ipc_bind_port('{}'): bound port disappeared before bootstrap",
1431                path
1432            );
1433        }
1434    }
1435    Ok(0)
1436}

1673/// Reads pci address.
1674fn read_pci_address(addr_ptr: u64) -> Result<pci::PciAddress, SyscallError> {
1675    if addr_ptr == 0 {
1676        return Err(SyscallError::Fault);
1677    }
1678    let user = UserSliceRead::new(addr_ptr, core::mem::size_of::<PciAddressAbi>())?;
1679    let mut raw = [0u8; core::mem::size_of::<PciAddressAbi>()];
1680    user.copy_to(&mut raw);
1681    let abi = unsafe { core::ptr::read_unaligned(raw.as_ptr() as *const PciAddressAbi) };
1682    if abi.device > 31 || abi.function > 7 {
1683        return Err(SyscallError::InvalidArgument);
1684    }
1685    Ok(pci::PciAddress::new(abi.bus, abi.device, abi.function))
1686}

1688/// Performs the sys pci enum operation.
1689fn sys_pci_enum(criteria_ptr: u64, out_ptr: u64, max_entries: u64) -> Result<u64, SyscallError> {
1690    if criteria_ptr == 0 || out_ptr == 0 {
1691        return Err(SyscallError::Fault);
1692    }
1693    if max_entries == 0 {
1694        return Ok(0);
1695    }
1696    let max_entries = core::cmp::min(max_entries as usize, 4096);
1697    let user_criteria = UserSliceRead::new(criteria_ptr, core::mem::size_of::<PciProbeCriteriaAbi>())?;
1698    let mut criteria_bytes = [0u8; core::mem::size_of::<PciProbeCriteriaAbi>()];
1699    user_criteria.copy_to(&mut criteria_bytes);
1700    let criteria_abi =
1701        unsafe { core::ptr::read_unaligned(criteria_bytes.as_ptr() as *const PciProbeCriteriaAbi) };
1702
1703    let criteria = pci::ProbeCriteria {
1704        vendor_id: if (criteria_abi.match_flags & PCI_MATCH_VENDOR_ID) != 0 {
1705            Some(criteria_abi.vendor_id)
1706        } else {
1707            None
1708        },
1709        device_id: if (criteria_abi.match_flags & PCI_MATCH_DEVICE_ID) != 0 {
1710            Some(criteria_abi.device_id)
1711        } else {
1712            None
1713        },
1714        class_code: if (criteria_abi.match_flags & PCI_MATCH_CLASS_CODE) != 0 {
1715            Some(criteria_abi.class_code)
1716        } else {
1717            None
1718        },
1719        subclass: if (criteria_abi.match_flags & PCI_MATCH_SUBCLASS) != 0 {
1720            Some(criteria_abi.subclass)
1721        } else {
1722            None
1723        },
1724        prog_if: if (criteria_abi.match_flags & PCI_MATCH_PROG_IF) != 0 {
1725            Some(criteria_abi.prog_if)
1726        } else {
1727            None
1728        },
1729    };
1730
1731    let devices = pci::probe_all(criteria);
1732    let count = core::cmp::min(devices.len(), max_entries);
1733    let mut out = alloc::vec::Vec::<PciDeviceInfoAbi>::with_capacity(count);
1734    for dev in devices.into_iter().take(count) {
1735        out.push(PciDeviceInfoAbi {
1736            address: PciAddressAbi {
1737                bus: dev.address.bus,
1738                device: dev.address.device,
1739                function: dev.address.function,
1740                _reserved: 0,
1741            },
1742            vendor_id: dev.vendor_id,
1743            device_id: dev.device_id,
1744            class_code: dev.class_code,
1745            subclass: dev.subclass,
1746            prog_if: dev.prog_if,
1747            revision: dev.revision,
1748            header_type: dev.header_type,
1749            interrupt_line: dev.interrupt_line,
1750            interrupt_pin: dev.interrupt_pin,
1751            _reserved: 0,
1752        });
1753    }
1754    let out_bytes_len = out
1755        .len()
1756        .checked_mul(core::mem::size_of::<PciDeviceInfoAbi>())
1757        .ok_or(SyscallError::InvalidArgument)?;
1758    let user_out = UserSliceWrite::new(out_ptr, out_bytes_len)?;
1759    let out_bytes = unsafe {
1760        core::slice::from_raw_parts(out.as_ptr() as *const u8, out_bytes_len)
1761    };
1762    user_out.copy_from(out_bytes);
1763    Ok(out.len() as u64)
1764}

1766/// Performs the sys pci cfg read operation.
1767fn sys_pci_cfg_read(addr_ptr: u64, offset: u64, width: u64) -> Result<u64, SyscallError> {
1768    let dev_addr = read_pci_address(addr_ptr)?;
1769    let offset = u8::try_from(offset).map_err(|_| SyscallError::InvalidArgument)?;
1770    let width = u8::try_from(width).map_err(|_| SyscallError::InvalidArgument)?;
1771    if !matches!(width, 1 | 2 | 4) {
1772        return Err(SyscallError::InvalidArgument);
1773    }
1774    if offset > 0xFC || (offset as u16 + width as u16) > 0x100 {
1775        return Err(SyscallError::InvalidArgument);
1776    }
1777    if (width == 2 && (offset & 1) != 0) || (width == 4 && (offset & 3) != 0) {
1778        return Err(SyscallError::InvalidArgument);
1779    }
1780    let dev = pci::all_devices()
1781        .into_iter()
1782        .find(|d| d.address == dev_addr)
1783        .ok_or(SyscallError::NotFound)?;
1784    let value = match width {
1785        1 => dev.read_config_u8(offset) as u32,
1786        2 => dev.read_config_u16(offset) as u32,
1787        _ => dev.read_config_u32(offset),
1788    };
1789    Ok(value as u64)
1790}

1853/// SYS_CHAN_SEND (221): send one `IpcMessage` to a channel, blocking if full.
1854///
1855/// arg1 = channel handle (CapId), arg2 = user pointer to 64-byte IpcMessage.
1856fn sys_chan_send(handle: u64, msg_ptr: u64) -> Result<u64, SyscallError> {
1857    crate::silo::enforce_cap_for_current_task(handle)?;
1858
1859    // Validate and copy the message from userspace.
1860    let user_slice = UserSliceRead::new(msg_ptr, 64).map_err(SyscallError::from)?;
1861    let mut msg = IpcMessage::new(0);
1862    // SAFETY: IpcMessage is repr(C), 64 bytes, fully initialised above.
1863    let n = user_slice.copy_to(unsafe {
1864        core::slice::from_raw_parts_mut(&mut msg as *mut IpcMessage as *mut u8, 64)
1865    });
1866    if n != 64 {
1867        return Err(SyscallError::Fault);
1868    }
1869
1870    // Fill in the sender task ID.
1871    let task = current_task_clone().ok_or(SyscallError::PermissionDenied)?;
1872    msg.sender = task.id.as_u64();
1873
1874    // Look up the channel capability.
1875    let caps = unsafe { &*task.process.capabilities.get() };
1876    let cap = caps
1877        .get(crate::capability::CapId::from_raw(handle))
1878        .ok_or(SyscallError::BadHandle)?;
1879    if cap.resource_type != ResourceType::Channel || !cap.permissions.write {
1880        return Err(SyscallError::PermissionDenied);
1881    }
1882    let chan_id = ChanId::from_u64(cap.resource as u64);
1883
1884    let chan = channel::get_channel(chan_id).ok_or(SyscallError::BadHandle)?;
1885    chan.send(msg).map_err(SyscallError::from)?;
1886
1887    Ok(0)
1888}
