26#[inline]
27fn local_invlpg(vaddr: u64) {
28    // Local TLB invalidation is sufficient here: this kernel currently runs
29    // one task per user address space (no shared user CR3 across CPUs).
30    unsafe {
31        core::arch::asm!("invlpg [{}]", in(reg) vaddr, options(nostack, preserves_flags));
32    }
33}

79/// Child bootstrap: restore user register snapshot and enter Ring 3.
80extern "C" fn fork_child_start(ctx_ptr: u64) -> ! {
81    let boxed = unsafe { Box::from_raw(ctx_ptr as *mut ForkUserContext) };
82    let ctx = *boxed;
83    unsafe { fork_iret_from_ctx(&ctx as *const ForkUserContext) }
84}

86#[unsafe(naked)]
87unsafe extern "C" fn fork_iret_from_ctx(_ctx: *const ForkUserContext) -> ! {
88    core::arch::naked_asm!(
89        "mov rsi, rdi",
90
91        // ===== Build IRET frame FIRST, using r8 as scratch ===========
92        // (r8 has not been restored yet, so we can clobber it safely)
93        "mov r8, [rsi + {off_user_ss}]",
94        "push r8",                            // SS
95        "mov r8, [rsi + {off_user_rsp}]",
96        "push r8",                            // user RSP
97        "mov r8, [rsi + {off_user_rflags}]",
98        "push r8",                            // user RFLAGS
99        "mov r8, [rsi + {off_user_cs}]",
100        "push r8",                            // CS
101        "mov r8, [rsi + {off_user_rip}]",
102        "push r8",                            // user RIP
103
104        // ===== Now restore ALL general-purpose registers============
105        "mov r15, [rsi + {off_r15}]",
106        "mov r14, [rsi + {off_r14}]",
107        "mov r13, [rsi + {off_r13}]",
108        "mov r12, [rsi + {off_r12}]",
109        "mov rbp, [rsi + {off_rbp}]",
110        "mov rbx, [rsi + {off_rbx}]",
111        "mov r11, [rsi + {off_r11}]",
112        "mov r10, [rsi + {off_r10}]",
113        "mov r9,  [rsi + {off_r9}]",
114        "mov r8,  [rsi + {off_r8}]",          // r8 now gets its correct value
115        "mov rdx, [rsi + {off_rdx}]",
116        "mov rcx, [rsi + {off_rcx}]",
117        "mov rdi, [rsi + {off_rdi}]",
118        "mov rax, 0",                         // child fork() returns 0
119        "mov rsi, [rsi + {off_rsi}]",         // rsi restored last
120        "iretq",
121        off_r15 = const OFF_R15,
122        off_r14 = const OFF_R14,
123        off_r13 = const OFF_R13,
124        off_r12 = const OFF_R12,
125        off_rbp = const OFF_RBP,
126        off_rbx = const OFF_RBX,
127        off_r11 = const OFF_R11,
128        off_r10 = const OFF_R10,
129        off_r9 = const OFF_R9,
130        off_r8 = const OFF_R8,
131        off_rsi = const OFF_RSI,
132        off_rdi = const OFF_RDI,
133        off_rdx = const OFF_RDX,
134        off_rcx = const OFF_RCX,
135        off_user_rip = const OFF_USER_RIP,
136        off_user_cs = const OFF_USER_CS,
137        off_user_rflags = const OFF_USER_RFLAGS,
138        off_user_rsp = const OFF_USER_RSP,
139        off_user_ss = const OFF_USER_SS,
140    );
141}

144fn build_child_task(
145    parent: &Arc<Task>,
146    child_as: Arc<AddressSpace>,
147    bootstrap_ctx: Box<ForkUserContext>,
148) -> Result<Arc<Task>, SyscallError> {
149    let kernel_stack =
150        KernelStack::allocate(Task::DEFAULT_STACK_SIZE).map_err(|_| SyscallError::OutOfMemory)?;
151    let context = CpuContext::new(fork_child_start as *const () as u64, &kernel_stack);
152
153    let parent_caps = unsafe { (&*parent.process.capabilities.get()).clone() };
154    let parent_fd = unsafe { (&*parent.process.fd_table.get()).clone_for_fork() };
155    let parent_blocked = parent.blocked_signals.clone();
156    let parent_actions: [SigActionData; 64] = unsafe { *parent.process.signal_actions.get() };
157    let parent_sigstack: Option<SigStack> = unsafe { *parent.signal_stack.get() };
158
159    let (pid, tid, tgid) = Task::allocate_process_ids();
160    let task = Arc::new(Task {
161        id: TaskId::new(),
162        pid,
163        tid,
164        tgid,
165        pgid: AtomicU32::new(parent.pgid.load(Ordering::Relaxed)),
166        sid: AtomicU32::new(parent.sid.load(Ordering::Relaxed)),
167        uid: AtomicU32::new(parent.uid.load(Ordering::Relaxed)),
168        euid: AtomicU32::new(parent.euid.load(Ordering::Relaxed)),
169        gid: AtomicU32::new(parent.gid.load(Ordering::Relaxed)),
170        egid: AtomicU32::new(parent.egid.load(Ordering::Relaxed)),
171        state: SyncUnsafeCell::new(TaskState::Ready),
172        priority: parent.priority,
173        context: SyncUnsafeCell::new(context),
174        kernel_stack,
175        user_stack: None,
176
177        name: "fork-child",
178        process: alloc::sync::Arc::new(crate::process::process::Process {
179            pid,
180            address_space: crate::process::task::SyncUnsafeCell::new(child_as),
181            fd_table: crate::process::task::SyncUnsafeCell::new(parent_fd),
182            capabilities: crate::process::task::SyncUnsafeCell::new(parent_caps),
183            signal_actions: crate::process::task::SyncUnsafeCell::new(parent_actions),
184            brk: core::sync::atomic::AtomicU64::new(parent.process.brk.load(core::sync::atomic::Ordering::Relaxed)),
185            mmap_hint: core::sync::atomic::AtomicU64::new(parent.process.mmap_hint.load(core::sync::atomic::Ordering::Relaxed)),
186            cwd: crate::process::task::SyncUnsafeCell::new(unsafe { &*parent.process.cwd.get() }.clone()),
187            umask: core::sync::atomic::AtomicU32::new(parent.process.umask.load(core::sync::atomic::Ordering::Relaxed)),
188        }),
189        // POSIX: pending signals are NOT inherited by the child.
190
191        pending_signals: SignalSet::new(),
192        // POSIX: signal mask IS inherited.
193        blocked_signals: parent_blocked,
194        signal_stack: SyncUnsafeCell::new(parent_sigstack),
195        itimers: crate::process::timer::ITimers::new(),
196        wake_pending: AtomicBool::new(false),
197        wake_deadline_ns: AtomicU64::new(0),
198        trampoline_entry: AtomicU64::new(0),
199        trampoline_stack_top: AtomicU64::new(0),
200        trampoline_arg0: AtomicU64::new(0),
201        ticks: AtomicU64::new(0),
202        sched_policy: SyncUnsafeCell::new(parent.sched_policy()),
203        vruntime: AtomicU64::new(parent.vruntime()),
204        // POSIX: clear_child_tid is NOT inherited — child starts with 0.
205        clear_child_tid: AtomicU64::new(0),
206        // POSIX: cwd IS inherited.
207        // POSIX: umask IS inherited.
208        // FS.base: child starts with 0 (its own TLS not yet set up).
209        user_fs_base: AtomicU64::new(0),
210        // FPU state: child inherits parent's FPU state.
211        fpu_state: {
212            let parent_fpu = unsafe { &*parent.fpu_state.get() };
213            let mut child_fpu = crate::process::task::FpuState::new();
214            child_fpu.data.copy_from_slice(&parent_fpu.data);
215            SyncUnsafeCell::new(child_fpu)
216        },
217    });
218
219    // CpuContext initial stack layout: r15, r14, r13(arg), r12(entry), rbp, rbx, ret
220    unsafe {
221        let ctx = &mut *task.context.get();
222        let frame = ctx.saved_rsp as *mut u64;
223        *frame.add(2) = Box::into_raw(bootstrap_ctx) as u64;
224    }
225
226    Ok(task)
227}

229/// SYS_PROC_FORK (302): fork with copy-on-write address-space cloning.
230pub fn sys_fork(frame: &SyscallFrame) -> Result<ForkResult, SyscallError> {
231    let parent = current_task_clone().ok_or(SyscallError::PermissionDenied)?;
232
233    // 1. Sanity check: cannot fork a kernel thread.
234    if parent.is_kernel() {
235        log::warn!("fork: attempt to fork kernel thread '{}'", parent.name);
236        return Err(SyscallError::PermissionDenied);
237    }
238
239    // 2. Capability check: check if task is restricted from forking.
240    // For now, we allow fork for all user processes unless restricted.
241    // TODO: implement ResourceType::Process/Task restricted capabilities.
242
243    let parent_as = unsafe { &*parent.process.address_space.get() };
244
245    // 3. Memory check: ensure parent has actual user-space mappings.
246    if !parent_as.has_user_mappings() {
247        log::warn!(
248            "fork: attempt to fork task '{}' with no user mappings",
249            parent.name
250        );
251        return Err(SyscallError::InvalidArgument);
252    }
253
254    let child_as = parent_as
255        .clone_cow()
256        .map_err(|_| SyscallError::OutOfMemory)?;
257
258    let child_user_ctx = Box::new(ForkUserContext {
259        r15: frame.r15,
260        r14: frame.r14,
261        r13: frame.r13,
262        r12: frame.r12,
263        rbp: frame.rbp,
264        rbx: frame.rbx,
265        r11: frame.r11,
266        r10: frame.r10,
267        r9: frame.r9,
268        r8: frame.r8,
269        rsi: frame.rsi,
270        rdi: frame.rdi,
271        rdx: frame.rdx,
272        rcx: frame.rcx,
273        user_rip: frame.iret_rip,
274        user_cs: frame.iret_cs,
275        user_rflags: frame.iret_rflags,
276        user_rsp: frame.iret_rsp,
277        user_ss: frame.iret_ss,
278    });
279
280    let child_task = build_child_task(&parent, child_as, child_user_ctx)?;
281    let child_pid = child_task.pid;
282    add_task_with_parent(child_task, parent.id);
283
284    Ok(ForkResult { child_pid })
285}

287/// Called from the page fault handler when a write fault occurs on a present page.
288/// Returns Ok(()) if the fault was successfully handled (COW resolution),
289/// or Err if it wasn't a COW fault (real access violation).
290pub fn handle_cow_fault(virt_addr: u64, address_space: &AddressSpace) -> Result<(), &'static str> {
291    use crate::memory::paging::BuddyFrameAllocator;
292    use x86_64::{
293        structures::paging::{Mapper, Page, PageTableFlags, Size4KiB, Translate},
294        VirtAddr,
295    };
296
297    let page = Page::<Size4KiB>::containing_address(VirtAddr::new(virt_addr));
298
299    // SAFETY: we are in an exception handler, address space is active.
300    let mut mapper = unsafe { address_space.mapper() };
301
302    //Check if page is mapped and has COW flag
303    let (phys_frame, flags): (
304        x86_64::structures::paging::PhysFrame<Size4KiB>,
305        PageTableFlags,
306    ) = match mapper.translate(VirtAddr::new(virt_addr)) {
307        TranslateResult::Mapped {
308            frame: x86_64::structures::paging::mapper::MappedFrame::Size4KiB(frame),
309            offset: _,
310            flags,
311        } => (frame, flags),
312        _ => return Err("Page not mapped or huge page"),
313    };
314
315    // We use BIT_9 as software COW flag
316    const COW_BIT: PageTableFlags = PageTableFlags::BIT_9;
317
318    if !flags.contains(COW_BIT) {
319        return Err("Not a COW page");
320    }
321
322    let old_frame = crate::memory::PhysFrame {
323        start_address: phys_frame.start_address(),
324    };
325
326    let refcount = crate::memory::cow::frame_get_refcount(old_frame);
327
328    if refcount == 1 {
329        // Case 1: we are the sole owner. Just make it writable.
330        let new_flags = (flags | PageTableFlags::WRITABLE) & !COW_BIT;
331
332        unsafe {
333            mapper
334                .update_flags(page, new_flags)
335                .map_err(|_| "Failed to update flags")?
336                .flush();
337        }
338        // Only the current CPU can hold this CR3 in the current design.
339        local_invlpg(virt_addr);
340        return Ok(());
341    }
342
343    // Case 2: shared page. Copy to new frame.
344    let mut frame_allocator = BuddyFrameAllocator;
345
346    let new_frame = frame_allocator
347        .allocate_frame()
348        .ok_or("OOM during COW copy")?;
349
350    // Copy content
351    unsafe {
352        let src = crate::memory::phys_to_virt(old_frame.start_address.as_u64()) as *const u8;
353        let dst = crate::memory::phys_to_virt(new_frame.start_address().as_u64()) as *mut u8;
354        core::ptr::copy_nonoverlapping(src, dst, 4096);
355    }
356
357    // Update mapping to new frame, Writable, no COW
358    let new_flags = (flags | PageTableFlags::WRITABLE) & !COW_BIT;
359
360    // Replace existing mapping (present+COW) by the private writable mapping.
361    let old_unmapped = mapper
362        .unmap(page)
363        .map_err(|_| "Failed to unmap old COW frame")?
364        .0;
365    debug_assert_eq!(old_unmapped.start_address(), old_frame.start_address);
366
367    let remap_res = unsafe { mapper.map_to(page, new_frame, new_flags, &mut frame_allocator) };
368    if remap_res.is_err() {
369        unsafe {
370            let _ = mapper.map_to(page, phys_frame, flags, &mut frame_allocator);
371        }
372        let lock = crate::memory::get_allocator();
373        let mut guard = lock.lock();
374        if let Some(allocator) = guard.as_mut() {
375            allocator.free(
376                crate::memory::PhysFrame {
377                    start_address: new_frame.start_address(),
378                },
379                0,
380            );
381        }
382        return Err("Failed to map new COW frame");
383    }
384    match remap_res {
385        Ok(flush) => flush.flush(),
386        Err(_) => unreachable!("checked remap result above"),
387    }
388
389    crate::memory::cow::frame_inc_ref(crate::memory::PhysFrame {
390        start_address: new_frame.start_address(),
391    });
392
393    // Only the current CPU can hold this CR3 in the current design.
394    local_invlpg(virt_addr);
395
396    // Decrement refcount of old frame after the new mapping is installed.
397    crate::memory::cow::frame_dec_ref(old_frame);
398
399    Ok(())
