76const VIRTIO_F_VERSION_1: u64 = 1 << 32;
77const VIRTIO_CONSOLE_F_MULTIPORT: u32 = 1;
78
79const VIRTIO_STATUS_RESET: u8 = 0;
80const VIRTIO_STATUS_ACKNOWLEDGE: u8 = 1;
81const VIRTIO_STATUS_DRIVER: u8 = 2;
82const VIRTIO_STATUS_DRIVER_OK: u8 = 4;
83const VIRTIO_STATUS_FEATURES_OK: u8 = 8;
84
85impl VirtioConsole {
86    /// Creates a new instance.
87    pub unsafe fn new(pci_dev: pci::PciDevice) -> Result<Self, &'static str> {
88        let bar = match pci_dev.read_bar(0) {
89            Some(Bar::Memory64 { addr, .. }) => addr,
90            _ => return Err("Invalid BAR"),
91        };
92
93        let mmio = phys_to_virt(bar) as usize;
94        let mut device = VirtioDevice { mmio };
95
96        device.reset();
97        device.add_status(VIRTIO_STATUS_ACKNOWLEDGE);
98        device.add_status(VIRTIO_STATUS_DRIVER);
99
100        let features = device.read_features();
101        let mut guest_features = VIRTIO_F_VERSION_1;
102        if (features & (1 << VIRTIO_CONSOLE_F_MULTIPORT)) != 0 {
103            guest_features |= 1 << VIRTIO_CONSOLE_F_MULTIPORT;
104        }
105        device.write_features(guest_features);
106        device.add_status(VIRTIO_STATUS_FEATURES_OK);
107
108        if (device.read_status() & VIRTIO_STATUS_FEATURES_OK) == 0 {
109            return Err("Features negotiation failed");
110        }
111
112        let mut ports = Vec::new();
113        let rx_queue = Virtqueue::new(&mut device, 0)?;
114        let tx_queue = Virtqueue::new(&mut device, 1)?;
115
116        ports.push(VirtioConsolePort {
117            id: 0,
118            rx_queue,
119            tx_queue,
120            open: true,
121        });
122
123        device.add_status(VIRTIO_STATUS_DRIVER_OK);
124
125        Ok(Self {
126            device,
127            ports: Mutex::new(ports),
128        })
129    }
130
131    /// Performs the write operation.
132    pub fn write(&self, data: &[u8]) -> Result<usize, &'static str> {
133        let mut ports = self.ports.lock();
134        let port = ports.first_mut().ok_or("No console port")?;
135
136        for chunk in data.chunks(VIRTIO_CONSOLE_PORT_SIZE) {
137            port.tx_queue.write(chunk)?;
138            self.device.notify_queue(1);
139
140            loop {
141                if port.tx_queue.poll_used() {
142                    break;
143                }
144                core::hint::spin_loop();
145            }
146        }
147
148        Ok(data.len())
149    }
150
151    /// Performs the read operation.
152    pub fn read(&self, buf: &mut [u8]) -> Result<usize, &'static str> {
153        let mut ports = self.ports.lock();
154        let port = ports.first_mut().ok_or("No console port")?;
155        port.rx_queue.read(buf)
156    }
157}

159impl VirtioDevice {
160    /// Performs the reset operation.
161    fn reset(&mut self) {
162        unsafe {
163            (self.mmio as *mut u32).write_volatile(0);
164        }
165        core::hint::spin_loop();
166    }
167
168    /// Performs the add status operation.
169    fn add_status(&mut self, status: u8) {
170        unsafe {
171            let current = ((self.mmio + 0x14) as *const u8).read_volatile();
172            ((self.mmio + 0x14) as *mut u8).write_volatile(current | status);
173        }
174    }
175
176    /// Reads status.
177    fn read_status(&self) -> u8 {
178        unsafe { ((self.mmio + 0x14) as *const u8).read_volatile() }
179    }
180
181    /// Reads features.
182    fn read_features(&self) -> u64 {
183        unsafe {
184            let lo = (self.mmio as *const u32).read_volatile() as u64;
185            let hi = ((self.mmio + 4) as *const u32).read_volatile() as u64;
186            (hi << 32) | lo
187        }
188    }
189
190    /// Writes features.
191    fn write_features(&mut self, features: u64) {
192        unsafe {
193            (self.mmio as *mut u32).write_volatile((features & 0xFFFFFFFF) as u32);
194            ((self.mmio + 4) as *mut u32).write_volatile(((features >> 32) & 0xFFFFFFFF) as u32);
195        }
196    }
197
198    /// Performs the notify queue operation.
199    fn notify_queue(&self, queue: u16) {
200        unsafe {
201            let offset = ((self.mmio + 0x20) as *const u16).read_volatile() as usize;
202            let queue_notify = self.mmio + 0x50 + offset * 4;
203            (queue_notify as *mut u32).write_volatile(queue as u32);
204        }
205    }
206}

208impl Virtqueue {
209    /// Creates a new instance.
210    fn new(device: &mut VirtioDevice, queue_idx: u16) -> Result<Self, &'static str> {
211        unsafe {
212            ((device.mmio + 0x16) as *mut u16).write_volatile(queue_idx);
213            let max_size = ((device.mmio + 0x18) as *const u16).read_volatile();
214            if max_size < VIRTIO_RING_SIZE as u16 {
215                return Err("Queue size too small");
216            }
217            ((device.mmio + 0x16) as *mut u16).write_volatile(VIRTIO_RING_SIZE as u16);
218
219            let desc_frame = allocate_dma_frame().ok_or("Failed to allocate desc")?;
220            let avail_frame = allocate_dma_frame().ok_or("Failed to allocate avail")?;
221            let used_frame = allocate_dma_frame().ok_or("Failed to allocate used")?;
222
223            let desc_phys = desc_frame.start_address.as_u64();
224            let avail_phys = avail_frame.start_address.as_u64();
225            let used_phys = used_frame.start_address.as_u64();
226
227            let desc_virt = phys_to_virt(desc_phys) as *mut VirtqDesc;
228            let avail_virt = phys_to_virt(avail_phys) as *mut VirtqAvail;
229            let used_virt = phys_to_virt(used_phys) as *mut VirtqUsed;
230
231            core::ptr::write_bytes(
232                desc_virt,
233                0,
234                VIRTIO_RING_SIZE * core::mem::size_of::<VirtqDesc>(),
235            );
236            core::ptr::write_bytes(avail_virt, 0, core::mem::size_of::<VirtqAvail>());
237            core::ptr::write_bytes(used_virt, 0, core::mem::size_of::<VirtqUsed>());
238
239            ((device.mmio + 0x10) as *mut u32).write_volatile((desc_phys & 0xFFFFFFFF) as u32);
240            ((device.mmio + 0x1A) as *mut u16).write_volatile(0xFFFF);
241
242            let buffer_frame = allocate_dma_frame().ok_or("Failed to allocate buffer")?;
243            let buffer_phys = buffer_frame.start_address.as_u64();
244            let buffer_virt = phys_to_virt(buffer_phys) as *mut u8;
245            core::ptr::write_bytes(buffer_virt, 0, 4096);
246
247            let mut free = Vec::with_capacity(VIRTIO_RING_SIZE);
248            for i in 0..VIRTIO_RING_SIZE {
249                free.push(i as u16);
250            }
251
252            Ok(Self {
253                desc: desc_virt,
254                avail: avail_virt,
255                used: used_virt,
256                desc_phys,
257                avail_phys,
258                used_phys,
259                buffer_phys,
260                buffer_virt,
261                free,
262                last_used_idx: 0,
263            })
264        }
265    }
266
267    /// Performs the write operation.
268    fn write(&mut self, data: &[u8]) -> Result<(), &'static str> {
269        unsafe {
270            if self.free.is_empty() {
271                return Err("No free descriptors");
272            }
273            let desc_idx = self.free.pop().unwrap();
274
275            let desc = &mut *self.desc.add(desc_idx as usize);
276            core::ptr::copy_nonoverlapping(data.as_ptr(), self.buffer_virt, data.len());
277            desc.addr = self.buffer_phys;
278            desc.len = data.len() as u32;
279            desc.flags = 2;
280            desc.next = 0;
281
282            let avail = &mut *self.avail;
283            let idx = avail.idx as usize % VIRTIO_RING_SIZE;
284            avail.ring[idx] = desc_idx;
285            avail.idx = avail.idx.wrapping_add(1);
286        }
287        Ok(())
288    }
289
290    /// Performs the read operation.
291    fn read(&mut self, buf: &mut [u8]) -> Result<usize, &'static str> {
292        unsafe {
293            if self.last_used_idx == (*self.used).idx {
294                return Ok(0);
295            }
296
297            let idx = self.last_used_idx as usize % VIRTIO_RING_SIZE;
298            let elem = (*self.used).ring[idx];
299
300            let len = core::cmp::min(elem.len as usize, buf.len());
301            core::ptr::copy_nonoverlapping(self.buffer_virt, buf.as_mut_ptr(), len);
302
303            self.last_used_idx = self.last_used_idx.wrapping_add(1);
304            Ok(len)
305        }
306    }
307
308    /// Performs the poll used operation.
309    fn poll_used(&mut self) -> bool {
310        unsafe { self.last_used_idx != (*self.used).idx }
311    }
312}

314static CONSOLE_INSTANCE: Once<Arc<VirtioConsole>> = Once::new();
315static CONSOLE_INITIALIZED: AtomicBool = AtomicBool::new(false);
316
317/// Performs the init operation.
318pub fn init() {
319    log::info!("[VirtIO-Console] Scanning for VirtIO Console devices...");
320
321    let candidates = pci::probe_all(ProbeCriteria {
322        vendor_id: Some(pci::vendor::VIRTIO),
323        device_id: Some(pci::device::VIRTIO_CONSOLE),
324        class_code: None,
325        subclass: None,
326        prog_if: None,
327    });
328
329    for pci_dev in candidates.into_iter() {
330        log::info!(
331            "VirtIO-Console: Found device at {:?} (VEN:{:04x} DEV:{:04x})",
332            pci_dev.address,
333            pci_dev.vendor_id,
334            pci_dev.device_id
335        );
336
337        pci_dev.enable_bus_master();
338
339        match unsafe { VirtioConsole::new(pci_dev) } {
340            Ok(console) => {
341                let arc = Arc::new(console);
342                CONSOLE_INSTANCE.call_once(|| arc.clone());
343                CONSOLE_INITIALIZED.store(true, Ordering::SeqCst);
344                log::info!("[VirtIO-Console] Initialized");
345                return;
346            }
347            Err(e) => {
348                log::warn!("VirtIO-Console: Failed to initialize device: {}", e);
349            }
350        }
351    }
352
353    log::info!("[VirtIO-Console] No device found");
354}

356/// Performs the write operation.
357pub fn write(data: &[u8]) -> Result<usize, &'static str> {
358    CONSOLE_INSTANCE
359        .get()
360        .ok_or("Console not initialized")?
361        .write(data)
362}

368        .ok_or("Console not initialized")?
369        .read(buf)
370}

372/// Returns whether available.
373pub fn is_available() -> bool {
374    CONSOLE_INITIALIZED.load(Ordering::Relaxed)
