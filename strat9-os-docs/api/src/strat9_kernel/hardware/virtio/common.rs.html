61/// A VirtIO virtqueue
62///
63/// This structure manages the split virtqueue format as described in
64/// VirtIO spec section 2.6 (Split Virtqueues).
65pub struct Virtqueue {
66    /// Queue size (must be power of 2)
67    queue_size: u16,
68
69    /// Physical address of descriptor table
70    desc_area: PhysFrame,
71
72    /// Physical address of available ring
73    avail_area: PhysFrame,
74
75    /// Physical address of used ring
76    used_area: PhysFrame,
77
78    /// Virtual address of descriptor table
79    desc_ptr: *mut VirtqDesc,
80
81    /// Virtual address of available ring
82    avail_ptr: *mut VirtqAvail,
83
84    /// Virtual address of available ring entries
85    avail_ring_ptr: *mut u16,
86
87    /// Virtual address of used ring
88    used_ptr: *mut VirtqUsed,
89
90    /// Virtual address of used ring entries
91    used_ring_ptr: *mut VirtqUsedElem,
92
93    /// Free descriptor list (indices of free descriptors)
94    free_descriptors: Vec<u16>,
95
96    /// Last seen used index
97    last_used_idx: u16,
98
99    /// Next available index
100    next_avail_idx: u16,
101}

106impl Virtqueue {
107    /// Create a new virtqueue with the specified size
108    ///
109    /// # Safety
110    /// The caller must ensure that the allocated memory is properly mapped
111    /// and accessible.
112    pub unsafe fn new(queue_size: u16) -> Result<Self, &'static str> {
113        if !queue_size.is_power_of_two() {
114            return Err("Queue size must be power of 2");
115        }
116
117        // Allocate descriptor table (16 bytes per descriptor)
118        let desc_size = queue_size as usize * core::mem::size_of::<VirtqDesc>();
119        let desc_pages = (desc_size + 4095) / 4096;
120        let desc_order = desc_pages.next_power_of_two().trailing_zeros() as u8;
121        let desc_area = memory::allocate_frames(desc_order)
122            .map_err(|_| "Failed to allocate descriptor table")?;
123
124        // Allocate available ring (2 + 2 + queue_size * 2 + 2 bytes)
125        let avail_size = 4 + queue_size as usize * 2 + 2;
126        let avail_pages = (avail_size + 4095) / 4096;
127        let avail_order = avail_pages.next_power_of_two().trailing_zeros() as u8;
128        let avail_area = memory::allocate_frames(avail_order)
129            .map_err(|_| "Failed to allocate available ring")?;
130
131        // Allocate used ring (2 + 2 + queue_size * 8 + 2 bytes)
132        let used_size = 4 + queue_size as usize * core::mem::size_of::<VirtqUsedElem>() + 2;
133        let used_pages = (used_size + 4095) / 4096;
134        let used_order = used_pages.next_power_of_two().trailing_zeros() as u8;
135        let used_area =
136            memory::allocate_frames(used_order).map_err(|_| "Failed to allocate used ring")?;
137
138        // SAFETY: we just allocated these frames; convert phys => virt via HHDM
139        // With Limine HHDM, all physical memory is already mapped, so we can
140        // directly use phys_to_virt without additional page table modifications.
141        // DO NOT call ensure_identity_map here - it can corrupt active page tables!
142
143        let desc_virt = crate::memory::phys_to_virt(desc_area.start_address.as_u64());
144        let avail_virt = crate::memory::phys_to_virt(avail_area.start_address.as_u64());
145        let used_virt = crate::memory::phys_to_virt(used_area.start_address.as_u64());
146
147        let desc_ptr = desc_virt as *mut VirtqDesc;
148        let avail_ptr = avail_virt as *mut VirtqAvail;
149        let avail_ring_ptr = (avail_virt + 4) as *mut u16;
150        let used_ptr = used_virt as *mut VirtqUsed;
151        let used_ring_ptr = (used_virt + 4) as *mut VirtqUsedElem;
152
153        // Zero out the memory
154        // SAFETY: we allocated these pages and they're mapped via HHDM
155        // Each descriptor is 16 bytes, so we write queue_size * 16 bytes
156        core::ptr::write_bytes(
157            desc_ptr,
158            0,
159            queue_size as usize * core::mem::size_of::<VirtqDesc>(),
160        );
161        core::ptr::write_bytes(avail_ptr as *mut u8, 0, avail_size);
162        core::ptr::write_bytes(used_ptr as *mut u8, 0, used_size);
163
164        // Initialize free descriptor list
165        let mut free_descriptors = Vec::with_capacity(queue_size as usize);
166        for i in (0..queue_size).rev() {
167            free_descriptors.push(i);
168        }
169
170        Ok(Self {
171            queue_size,
172            desc_area,
173            avail_area,
174            used_area,
175            desc_ptr,
176            avail_ptr,
177            avail_ring_ptr,
178            used_ptr,
179            used_ring_ptr,
180            free_descriptors,
181            last_used_idx: 0,
182            next_avail_idx: 0,
183        })
184    }

213    /// Free a descriptor chain
214    ///
215    /// Walks the chain following NEXT flags and frees all descriptors
216    pub fn free_descriptor(&mut self, head: u16) {
217        let mut current = head;
218
219        loop {
220            // SAFETY: current is a valid descriptor index
221            let desc = unsafe { &*self.desc_ptr.add(current as usize) };
222            let has_next = desc.flags & vring_flags::NEXT != 0;
223            let next = desc.next;
224
225            self.free_descriptors.push(current);
226
227            if !has_next {
228                break;
229            }
230            current = next;
231        }
232    }
233
234    /// Add a buffer to the virtqueue
235    ///
236    /// Returns the descriptor index (token) that can be used to track completion
237    ///
238    /// # Arguments
239    /// * `buffers`: A list of (physical_address, length, is_write_only)
240    pub fn add_buffer(&mut self, buffers: &[(u64, u32, bool)]) -> Result<u16, &'static str> {
241        if buffers.is_empty() {
242            return Err("Empty buffer list");
243        }
244
245        if buffers.len() > self.free_descriptors.len() {
246            return Err("Not enough free descriptors");
247        }
248
249        // Allocate descriptor chain
250        let head = self.alloc_descriptor().ok_or("No free descriptors")?;
251        let mut current = head;
252
253        for (i, &(addr, len, write)) in buffers.iter().enumerate() {
254            let is_last = i == buffers.len() - 1;
255
256            // SAFETY: current is a valid index regulated by alloc_descriptor
257            let desc = unsafe { &mut *self.desc_ptr.add(current as usize) };
258            desc.addr = addr;
259            desc.len = len;
260            desc.flags = if write { vring_flags::WRITE } else { 0 };
261
262            if !is_last {
263                let next = self.alloc_descriptor().ok_or("No free descriptors")?;
264                desc.flags |= vring_flags::NEXT;
265                desc.next = next;
266                current = next;
267            }
268        }
269
270        // Add to available ring
271        // SAFETY: Atomic load
272        let avail_idx = unsafe { (*self.avail_ptr).idx.load(Ordering::Acquire) };
273        let ring_idx = (avail_idx % self.queue_size) as usize;
274
275        // SAFETY: ring_idx is bounded by queue_size
276        unsafe {
277            write_volatile(self.avail_ring_ptr.add(ring_idx), head);
278        }
279
280        // Memory barrier before updating index to ensure device sees the descriptor table updates
281        fence(Ordering::Release);
282
283        // Update available index
284        // SAFETY: Atomic store
285        unsafe {
286            (*self.avail_ptr)
287                .idx
288                .store(avail_idx.wrapping_add(1), Ordering::Release);
289        }
290
291        self.next_avail_idx = avail_idx.wrapping_add(1);
292
293        Ok(head)
294    }

296    /// Check if there are any used buffers
297    pub fn has_used(&self) -> bool {
298        // SAFETY: Atomic load
299        let used_idx = unsafe { (*self.used_ptr).idx.load(Ordering::Acquire) };
300        self.last_used_idx != used_idx
301    }
302
303    /// Get the next used buffer
304    ///
305    /// Returns (descriptor_index, length_written)
306    pub fn get_used(&mut self) -> Option<(u16, u32)> {
307        // SAFETY: Atomic load
308        let used_idx = unsafe { (*self.used_ptr).idx.load(Ordering::Acquire) };
309
310        if self.last_used_idx == used_idx {
311            return None;
312        }
313
314        let ring_idx = (self.last_used_idx % self.queue_size) as usize;
315
316        // SAFETY: ring_idx is bounded by queue_size
317        let elem = unsafe { read_volatile(self.used_ring_ptr.add(ring_idx)) };
318
319        self.last_used_idx = self.last_used_idx.wrapping_add(1);
320
321        // We do NOT free the descriptor here immediately, because the caller might need to read the data.
322        // But in the current design, the caller is responsible for freeing.
323        // Wait, the previous implementation freed it here.
324        // Let's stick to the previous pattern: free the chain, return the Head ID.
325        // The implementation assumes the caller is done with the *descriptors*,
326        // but the data is in the buffers pointed to by the descriptors.
327        self.free_descriptor(elem.id as u16);
328
329        Some((elem.id as u16, elem.len))
330    }

340/// VirtIO device base
341///
342/// Common functionality for all VirtIO devices
343pub struct VirtioDevice {
344    /// PCI device
345    pub pci_dev: PciDevice,
346
347    /// I/O base address (BAR0 for legacy devices)
348    pub io_base: u16,
349}
350
351impl VirtioDevice {
352    /// Create a new VirtIO device from a PCI device
353    ///
354    /// # Safety
355    /// The PCI device must be a valid VirtIO device
356    pub unsafe fn new(pci_dev: PciDevice) -> Result<Self, &'static str> {
357        // Read BAR0 (I/O space for legacy VirtIO devices)
358        let bar0 = pci_dev.read_bar(0).ok_or("BAR0 not present")?;
359
360        let io_base = match bar0 {
361            Bar::Io { port } => port,
362            _ => return Err("BAR0 is not I/O space (legacy VirtIO required)"),
363        };
364
365        // Enable I/O space and bus mastering
366        pci_dev.enable_io_space();
367        pci_dev.enable_bus_master();
368
369        Ok(Self { pci_dev, io_base })
370    }

372    /// Read an 8-bit value from a device register
373    pub fn read_reg_u8(&self, offset: u16) -> u8 {
374        // SAFETY: I/O port access to VirtIO device registers
375        unsafe { crate::arch::x86_64::io::inb(self.io_base + offset) }
376    }

382    }

444    /// Acknowledge interrupt (write 0 to ISR)
445    pub fn ack_interrupt(&self) {
446        // Reading ISR already clears it, but we can also write to acknowledge
447        let _ = self.read_reg_u8(19); // VIRTIO_PCI_ISR
448    }
449
450    /// Setup a virtqueue
451    pub fn setup_queue(&self, queue_index: u16, queue: &Virtqueue) {
452        // Select queue
453        self.write_reg_u16(14, queue_index); // VIRTIO_PCI_QUEUE_SEL
454
455        // Set queue size
456        self.write_reg_u16(12, queue.size()); // VIRTIO_PCI_QUEUE_NUM
457
458        // Set queue addresses (page-aligned physical addresses >> 12)
459        let desc_pfn = (queue.desc_area() >> 12) as u32;
460        self.write_reg_u32(8, desc_pfn); // VIRTIO_PCI_QUEUE_PFN
461    }
462
463    /// Notify a queue
464    pub fn notify_queue(&self, queue_index: u16) {
465        self.write_reg_u16(16, queue_index); // VIRTIO_PCI_QUEUE_NOTIFY
466    }
