1// HPET (High Precision Event Timer) Driver
2// Reference: HPET spec 1.0a
3//
4// Uses the kernel's ACPI subsystem (already initialized) to locate the HPET
5// table, then maps and configures the HPET MMIO registers.
6
7#![allow(dead_code)]
8
9use crate::memory::{self, phys_to_virt};
10use core::ptr;
11use core::sync::atomic::{AtomicBool, Ordering};
12use spin::Mutex;
13
14const HPET_GENERAL_CAP_ID: usize = 0x000;
15const HPET_GENERAL_CONFIG: usize = 0x010;
16const HPET_MAIN_COUNTER: usize = 0x0F0;
17
18const HPET_CAP_NUM_TIMERS_MASK: u64 = 0x0000_0000_0000_1F00;
19const HPET_CAP_COUNTER_SIZE: u64 = 1 << 13;
20const HPET_CAP_COUNTER_CLK_PERIOD_SHIFT: u32 = 32;
21
22const HPET_CONFIG_ENABLE: u64 = 0x0000_0000_0000_0001;
23
24const GAS_ADDR_SPACE_MEMORY: u8 = 0;
25
26/// HPET MMIO register page size (minimum 1 KiB, map one full page).
27const HPET_MMIO_MAP_SIZE: u64 = 0x1000;
28
29pub struct HpetInfo {
30    base_addr: u64,
31    mmio_base: usize,
32    num_timers: u8,
33    tick_period_fs: u32,
34    tick_period_ns: u32,
35    is_64bit: bool,
36}
37
38unsafe impl Send for HpetInfo {}
39unsafe impl Sync for HpetInfo {}
40
41static HPET_INFO: Mutex<Option<HpetInfo>> = Mutex::new(None);
42static HPET_INITIALIZED: AtomicBool = AtomicBool::new(false);
43
44/// Performs the hpet read operation.
45unsafe fn hpet_read(mmio_base: usize, offset: usize) -> u64 {
46    core::ptr::read_volatile((mmio_base + offset) as *const u64)
47}
48
49/// Performs the hpet write operation.
50unsafe fn hpet_write(mmio_base: usize, offset: usize, value: u64) {
51    core::ptr::write_volatile((mmio_base + offset) as *mut u64, value);
52}
53
54/// Initialize HPET using the already-discovered ACPI table.
55pub fn init() -> Result<(), &'static str> {
56    log::info!("[HPET] Searching for HPET...");
57
58    let hpet_acpi = crate::acpi::hpet::HpetAcpiTable::get()
59        .ok_or("HPET ACPI table not found")?;
60
61    // Read packed fields safely using unaligned loads.
62    let gas = unsafe { ptr::read_unaligned(core::ptr::addr_of!(hpet_acpi.gen_addr_struct)) };
63    let base_addr = gas.phys_addr;
64    let address_space = gas.address_space;
65    let comparator_desc = unsafe { ptr::read_unaligned(core::ptr::addr_of!(hpet_acpi.comparator_descriptor)) };
66    let min_tick = unsafe { ptr::read_unaligned(core::ptr::addr_of!(hpet_acpi.min_periodic_clock_tick)) };
67
68    log::info!(
69        "[HPET] Found HPET: base=0x{:x}, comparators={}, min_tick={}",
70        base_addr,
71        (comparator_desc >> 3) & 0x1F,
72        min_tick,
73    );
74
75    if address_space != GAS_ADDR_SPACE_MEMORY {
76        return Err("HPET: address space is not memory-mapped");
77    }
78    if base_addr == 0 || (base_addr & 0x7) != 0 {
79        return Err("HPET: base address invalid or misaligned");
80    }
81
82    // Map the HPET MMIO page(s) into the HHDM before any register access.
83    memory::paging::ensure_identity_map_range(base_addr, HPET_MMIO_MAP_SIZE);
84
85    let mmio_base = phys_to_virt(base_addr) as usize;
86
87    unsafe {
88        let cap_id = hpet_read(mmio_base, HPET_GENERAL_CAP_ID);
89        let num_timers = ((cap_id & HPET_CAP_NUM_TIMERS_MASK) >> 8) as u8 + 1;
90        let is_64bit = (cap_id & HPET_CAP_COUNTER_SIZE) != 0;
91        let tick_period_fs = (cap_id >> HPET_CAP_COUNTER_CLK_PERIOD_SHIFT) as u32;
92        if tick_period_fs == 0 {
93            return Err("HPET counter period is zero");
94        }
95        if tick_period_fs > 1_000_000_000 {
96            return Err("HPET counter period out of range");
97        }
98        let tick_period_ns = core::cmp::max(1, tick_period_fs / 1_000_000);
99
100        let info = HpetInfo {
101            base_addr,
102            mmio_base,
103            num_timers,
104            tick_period_fs,
105            tick_period_ns,
106            is_64bit,
107        };
108
109        *HPET_INFO.lock() = Some(info);
110
111        let mut config = hpet_read(mmio_base, HPET_GENERAL_CONFIG);
112        config &= !0x02;
113        config |= HPET_CONFIG_ENABLE;
114        hpet_write(mmio_base, HPET_GENERAL_CONFIG, config);
115
116        HPET_INITIALIZED.store(true, Ordering::SeqCst);
117
118        log::info!(
119            "[HPET] Initialized: {} timers, {}-bit, {} ns/tick",
120            num_timers,
121            if is_64bit { 64 } else { 32 },
122            tick_period_ns,
123        );
124    }
125
126    Ok(())
127}

129/// Read main counter value
130pub fn read_counter() -> u64 {
131    let info = HPET_INFO.lock();
132    match *info {
133        Some(ref hpet) => unsafe { hpet_read(hpet.mmio_base, HPET_MAIN_COUNTER) },
134        None => 0,
135    }
136}
137
138/// Get tick period in nanoseconds
139pub fn tick_period_ns() -> u32 {
140    let info = HPET_INFO.lock();
141    match *info {
142        Some(ref hpet) => hpet.tick_period_ns,
143        None => 0,
144    }
145}
146
147/// Get HPET frequency in Hz
148pub fn frequency_hz() -> u64 {
149    let info = HPET_INFO.lock();
150    match *info {
151        Some(ref hpet) if hpet.tick_period_fs > 0 => 1_000_000_000_000_000u64 / hpet.tick_period_fs as u64,
152        _ => 0,
153    }
154}

156/// Check if HPET is initialized
157pub fn is_available() -> bool {
158    HPET_INITIALIZED.load(Ordering::Relaxed)
159}
160
161/// Get number of timers
162pub fn num_timers() -> u8 {
163    let info = HPET_INFO.lock();
164    match *info {
165        Some(ref hpet) => hpet.num_timers,
166        None => 0,
167    }
168}

170/// High-precision delay in microseconds
171pub fn delay_us(us: u64) {
172    if !is_available() {
173        // Fallback to busy wait
174        for _ in 0..(us * 100) {
175            core::hint::spin_loop();
176        }
177        return;
178    }
179    
180    let period_ns = tick_period_ns() as u64;
181    if period_ns == 0 {
182        return;
183    }
184    let start = read_counter();
185    let ticks_needed = if us == 0 {
186        0
187    } else {
188        core::cmp::max(1, (us.saturating_mul(1000)) / period_ns)
189    };
190    while read_counter().wrapping_sub(start) < ticks_needed {
191        core::hint::spin_loop();
192    }
193}
194
195/// High-precision delay in milliseconds
196pub fn delay_ms(ms: u64) {
197    delay_us(ms * 1000);
198}
199
200/// Get elapsed time since boot in milliseconds
201pub fn uptime_ms() -> u64 {
202    if !is_available() {
203        return 0;
204    }
205    
206    let counter = read_counter() as u128;
207    let period_ns = tick_period_ns() as u128;
208    ((counter.saturating_mul(period_ns)) / 1_000_000) as u64
209}
210
211/// Get elapsed time since boot in seconds
212pub fn uptime_secs() -> u64 {
213    uptime_ms() / 1000
