37/// VirtIO block request types
38#[allow(dead_code)]
39#[repr(u32)]
40pub enum RequestType {
41    /// Read from device
42    In = 0,
43    /// Write to device
44    Out = 1,
45    /// Flush write cache
46    Flush = 4,
47    /// Get device ID
48    GetId = 8,
49    /// Discard sectors
50    Discard = 11,
51    /// Write zeroes
52    WriteZeroes = 13,
53}

86/// Block device trait (implemented by VirtIO-blk driver)
87pub trait BlockDevice {
88    /// Read sectors from the device
89    fn read_sector(&self, sector: u64, buf: &mut [u8]) -> Result<(), BlockError>;
90    /// Write sectors to the device
91    fn write_sector(&self, sector: u64, buf: &[u8]) -> Result<(), BlockError>;
92    /// Get the total number of sectors
93    fn sector_count(&self) -> u64;
94}

96#[derive(Debug, Clone, Copy, PartialEq, Eq, thiserror::Error)]
97pub enum BlockError {
98    #[error("device I/O error")]
99    IoError,
100    #[error("invalid sector number")]
101    InvalidSector,
102    #[error("buffer too small")]
103    BufferTooSmall,
104    #[error("device not ready")]
105    NotReady,
106}

119impl VirtioBlockDevice {
120    /// Initialize a VirtIO block device from a PCI device
121    ///
122    /// # Safety
123    /// The PCI device must be a valid VirtIO block device
124    pub unsafe fn new(pci_dev: PciDevice) -> Result<Self, &'static str> {
125        log::info!("VirtIO-blk: Initializing device at {:?}", pci_dev.address);
126
127        // Create VirtIO device
128        let device = VirtioDevice::new(pci_dev)?;
129
130        // Reset device
131        device.reset();
132
133        // Acknowledge device
134        device.add_status(status::ACKNOWLEDGE as u8);
135
136        // Indicate we know how to drive it
137        device.add_status(status::DRIVER as u8);
138
139        // Read and negotiate features
140        let device_features = device.read_device_features();
141        log::debug!("VirtIO-blk: Device features: 0x{:08x}", device_features);
142
143        // We don't need any special features for basic operation yet
144        let guest_features = 0;
145        device.write_guest_features(guest_features);
146
147        // Features OK
148        device.add_status(status::FEATURES_OK as u8);
149
150        // Verify features OK
151        if device.get_status() & (status::FEATURES_OK as u8) == 0 {
152            return Err("Device doesn't support our feature set");
153        }
154
155        // Create virtqueue (queue 0 is the request queue)
156        let queue = Virtqueue::new(128)?;
157
158        // Setup queue with device
159        device.setup_queue(0, &queue);
160
161        // Driver ready
162        device.add_status(status::DRIVER_OK as u8);
163
164        // Read device capacity from config space (offset 0 in device-specific config)
165        // For legacy devices, device-specific config starts at offset 20 (after header)
166        // Or strictly speaking, after common config.
167        // Legacy VirtIO Header: 20 bytes.
168        // Block Config starts at offset 20.
169        let capacity_low = device.read_reg_u32(20);
170        let capacity_high = device.read_reg_u32(24);
171        let capacity = ((capacity_high as u64) << 32) | (capacity_low as u64);
172
173        log::info!(
174            "VirtIO-blk: Capacity: {} sectors ({} MB)",
175            capacity,
176            (capacity * SECTOR_SIZE as u64) / (1024 * 1024)
177        );
178
179        log::info!("VirtIO-blk: Device initialized successfully");
180
181        Ok(Self {
182            device,
183            queue: SpinLock::new(queue),
184            capacity,
185        })
186    }
187
188    /// Submit a block request and wait for completion
189    fn do_request(
190        &self,
191        request_type: RequestType,
192        sector: u64,
193        mut data_buf: Option<(&mut [u8], bool)>, // (buffer, is_write)
194    ) -> Result<(), BlockError> {
195        // Allocate request header and status byte
196        // We use a single frame for both if possible, or small allocations?
197        // To be safe and simple with the frame allocator, we'll alloc a frame.
198        // In a real optimized driver, we would have a slab allocator or pre-allocated pool.
199
200        let mut lock = get_allocator().lock();
201        let allocator = lock.as_mut().ok_or(BlockError::NotReady)?;
202        let metadata_frame = allocator.alloc_frame().map_err(|_| BlockError::NotReady)?;
203        drop(lock);
204
205        let metadata_phys = metadata_frame.start_address.as_u64();
206        let metadata_virt = crate::memory::phys_to_virt(metadata_phys);
207        let status_offset = mem::size_of::<BlockRequestHeader>() as u64;
208
209        // Layout: [Header (16 bytes)] ... [Status (1 byte)]
210        let header_ptr = metadata_virt as *mut BlockRequestHeader;
211        let status_ptr = (metadata_virt + status_offset) as *mut u8;
212
213        // Setup request header (CPU access → virtual address)
214        unsafe {
215            ptr::write(
216                header_ptr,
217                BlockRequestHeader {
218                    request_type: request_type as u32,
219                    reserved: 0,
220                    sector,
221                },
222            );
223            ptr::write(status_ptr, 0xFF); // Initialize with invalid status
224        }
225
226        // Build descriptor chain (DMA → physical addresses)
227        let mut buffers = Vec::with_capacity(3);
228
229        // 1. Header (Device Readable) — physical addr for DMA
230        buffers.push((
231            metadata_phys,
232            mem::size_of::<BlockRequestHeader>() as u32,
233            false,
234        ));
235
236        // 2. Data (Device Readable OR Writable)
237        // If data_buf is provided
238        let mut data_frame_info = None;
239
240        if let Some((buf, is_write)) = data_buf.as_mut() {
241            // We need a physically contiguous buffer for DMA
242            // For now, we allocate a bounce buffer.
243            // TODO: Support scatter-gather if the input buffer crosses page boundaries or isn't physical.
244
245            let buf_size = buf.len();
246            let buf_pages = (buf_size + 4095) / 4096;
247            let buf_order = buf_pages.next_power_of_two().trailing_zeros() as u8;
248
249            let mut lock = get_allocator().lock();
250            let allocator = lock.as_mut().ok_or(BlockError::NotReady)?;
251            let buf_frame = allocator.alloc(buf_order).map_err(|_| {
252                allocator.free(metadata_frame, 0);
253                BlockError::NotReady
254            })?;
255            drop(lock);
256
257            let buf_phys = buf_frame.start_address.as_u64();
258            let buf_virt = crate::memory::phys_to_virt(buf_phys);
259            data_frame_info = Some((buf_frame, buf_order));
260
261            // If WRITE (Out): Copy from source buf to DMA bounce buffer (CPU access → virtual)
262            if *is_write {
263                unsafe {
264                    ptr::copy_nonoverlapping(buf.as_ptr(), buf_virt as *mut u8, buf_size);
265                }
266            }
267
268            // `is_write` param tells us if we are writing TO disk.
269            // Write to disk: device reads from memory (flags = 0)
270            // Read from disk: device writes to memory (flags = WRITE)
271            let device_writable = !*is_write;
272
273            // DMA → physical address
274            buffers.push((buf_phys, buf_size as u32, device_writable));
275        }
276
277        // 3. Status (Device Writable) — physical addr for DMA
278        buffers.push((metadata_phys + status_offset, 1, true));
279
280        // Submit request
281        let mut queue = self.queue.lock();
282        let token = match queue.add_buffer(&buffers) {
283            Ok(t) => t,
284            Err(_) => {
285                drop(queue);
286                // Cleanup
287                let mut lock = get_allocator().lock();
288                if let Some(allocator) = lock.as_mut() {
289                    allocator.free(metadata_frame, 0);
290                    if let Some((f, o)) = data_frame_info {
291                        allocator.free(f, o);
292                    }
293                }
294                return Err(BlockError::IoError);
295            }
296        };
297
298        // Notify device
299        if queue.should_notify() {
300            self.device.notify_queue(0);
301        }
302        drop(queue);
303
304        // Wait for completion (busy-poll for now).
305        //
306        // IMPORTANT:
307        // Do not use HLT here. This path can run from syscall context where IF
308        // may be masked, and HLT would deadlock the CPU.
309        // TODO: replace with proper waitqueue + interrupt completion.
310        loop {
311            let mut queue = self.queue.lock();
312            if queue.has_used() {
313                // We don't check the token because we are single-threaded/blocking per device for now
314                // But to be correct we should find OUR token.
315                // virtio::common::Virtqueue::get_used currently pops the *next* used.
316                // If there are multiple in flight, we might pop someone else's.
317                // But here we are blocking, so only one in flight effectively.
318                if let Some((used_token, _len)) = queue.get_used() {
319                    if used_token == token {
320                        break;
321                    } else {
322                        // This shouldn't happen in single-threaded blocking mode
323                        // If it does, we just dropped someone else's completion.
324                        log::warn!("VirtIO-blk: Received unexpected token {}", used_token);
325                    }
326                }
327            }
328            drop(queue);
329            core::hint::spin_loop();
330        }
331
332        // Check status
333        let status_byte = unsafe { ptr::read(status_ptr) };
334
335        // Post-processing
336        if let Some((buf, is_write)) = data_buf {
337            if let Some((buf_frame, buf_order)) = data_frame_info {
338                let buf_virt = crate::memory::phys_to_virt(buf_frame.start_address.as_u64());
339
340                // If Read (In): Copy from DMA buf to destination buf (CPU access → virtual)
341                if !is_write && status_byte == BlockStatus::Ok as u8 {
342                    unsafe {
343                        ptr::copy_nonoverlapping(
344                            buf_virt as *const u8,
345                            buf.as_mut_ptr(),
346                            buf.len(),
347                        );
348                    }
349                }
350
351                // Free DMA buffer
352                let mut lock = get_allocator().lock();
353                if let Some(allocator) = lock.as_mut() {
354                    allocator.free(buf_frame, buf_order);
355                }
356                // drop(lock) implied at end of scope but explicit drop is better if reused
357                drop(lock);
358            }
359        }
360
361        // Free metadata frame
362        let mut lock = get_allocator().lock();
363        if let Some(allocator) = lock.as_mut() {
364            allocator.free(metadata_frame, 0);
365        }
366        drop(lock);
367
368        if status_byte == BlockStatus::Ok as u8 {
369            Ok(())
370        } else {
371            log::error!("VirtIO-blk: Request failed with status {}", status_byte);
372            Err(BlockError::IoError)
373        }
374    }
375}

377impl BlockDevice for VirtioBlockDevice {
378    /// Reads sector.
379    fn read_sector(&self, sector: u64, buf: &mut [u8]) -> Result<(), BlockError> {
380        if sector >= self.capacity {
381            return Err(BlockError::InvalidSector);
382        }
383
384        if buf.len() < SECTOR_SIZE {
385            return Err(BlockError::BufferTooSmall);
386        }
387
388        self.do_request(RequestType::In, sector, Some((buf, false)))
389    }
390
391    /// Writes sector.
392    fn write_sector(&self, sector: u64, buf: &[u8]) -> Result<(), BlockError> {
393        if sector >= self.capacity {
394            return Err(BlockError::InvalidSector);
395        }
396
397        if buf.len() < SECTOR_SIZE {
398            return Err(BlockError::BufferTooSmall);
399        }
400
401        // Need mutable buffer for internal DMA operations signature (though we won't modify it if is_write=true)
402        // Our do_request takes &mut [u8], so we need to either change do_request or cast.
403        // It's safer to copy the input slice to a temp buffer if we needed to, but here do_request copies it to DMA anyway.
404        // But do_request signature expects &mut [u8] because it handles both read and write.
405        // We can just cast const to mut since we know we won't write to it if is_write=true.
406        // Or better, let's fix do_request to take Option<(&mut [u8], Direction)>.
407        // For now, let's do a safe copy to avoid unsafe hacks.
408
409        let mut buf_copy = [0u8; SECTOR_SIZE];
410        buf_copy[..SECTOR_SIZE].copy_from_slice(&buf[..SECTOR_SIZE]);
411
412        self.do_request(RequestType::Out, sector, Some((&mut buf_copy, true)))
413    }
414
415    /// Performs the sector count operation.
416    fn sector_count(&self) -> u64 {
417        self.capacity
418    }
419}

421/// Global VirtIO block device
422static VIRTIO_BLOCK: SpinLock<Option<Box<VirtioBlockDevice>>> = SpinLock::new(None);
423
424/// VirtIO block IRQ line (will be set during init)
425static mut VIRTIO_BLOCK_IRQ: u8 = 0;
426
427/// Initialize VirtIO block device
428///
429/// Scans PCI bus for VirtIO block devices and initializes the first one found.
430pub fn init() {
431    log::info!("VirtIO-blk: Scanning for devices...");
432
433    // Prefer strict class-based probe (mass storage), with fallback to
434    // vendor+device for odd firmware/virtual setups.
435    let pci_dev = match pci::probe_first(pci::ProbeCriteria {
436        vendor_id: Some(pci::vendor::VIRTIO),
437        device_id: Some(pci::device::VIRTIO_BLOCK),
438        class_code: Some(pci::class::MASS_STORAGE),
439        subclass: None,
440        prog_if: None,
441    })
442    .or_else(|| pci::find_virtio_device(pci::device::VIRTIO_BLOCK))
443    {
444        Some(dev) => dev,
445        None => {
446            log::warn!("VirtIO-blk: No block device found");
447            return;
448        }
449    };
450
451    // Read interrupt line from PCI config
452    let irq_line = pci_dev.read_config_u8(pci::config::INTERRUPT_LINE);
453
454    // Initialize device
455    match unsafe { VirtioBlockDevice::new(pci_dev) } {
456        Ok(device) => {
457            // Store IRQ line for interrupt handler
458            unsafe {
459                VIRTIO_BLOCK_IRQ = irq_line;
460            }
461
462            // Register device
463            *VIRTIO_BLOCK.lock() = Some(Box::new(device));
464
465            // Register IRQ handler in IDT
466            crate::arch::x86_64::idt::register_virtio_block_irq(irq_line);
467
468            log::info!("VirtIO-blk: Device initialized on IRQ {}", irq_line);
469        }
470        Err(e) => {
471            log::error!("VirtIO-blk: Failed to initialize device: {}", e);
472        }
473    }
474}

476/// Handle VirtIO block device interrupt
477///
478/// Called from the IDT IRQ handler when the VirtIO device signals completion.
479/// Acknowledges the interrupt and processes completed requests.
480pub fn handle_interrupt() {
481    // Acknowledge the interrupt at the device level
482    let lock = VIRTIO_BLOCK.lock();
483    if let Some(device) = lock.as_ref() {
484        // Read ISR status to check if interrupt is for us
485        let isr_status = device.device.read_isr_status();
486        if isr_status != 0 {
487            // Acknowledge the interrupt
488            device.device.ack_interrupt();
489
490            // Process completed requests (wake up waiting tasks)
491            // For now, just log the interrupt
492            log::trace!("VirtIO-blk: Interrupt handled (ISR={})", isr_status);
493        }
494    }
495}

497/// Get the global VirtIO block device
498pub fn get_device() -> Option<&'static VirtioBlockDevice> {
499    unsafe {
500        let lock = VIRTIO_BLOCK.lock();
501        if lock.is_some() {
502            // This is slightly unsafe if the lock is dropped and the box is moved,
503            // but the static Option is never cleared in this kernel.
504            // A safer way is needed for production.
505            let ptr = &**lock.as_ref().unwrap() as *const VirtioBlockDevice;
506            Some(&*ptr)
507        } else {
508            None
509        }
510    }
511}
