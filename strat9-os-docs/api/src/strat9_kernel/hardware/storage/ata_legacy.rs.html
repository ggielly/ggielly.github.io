13const ATA_PRIMARY_IO: u16 = 0x1F0;
14const ATA_SECONDARY_IO: u16 = 0x170;
15
16const ATA_REG_DATA: usize = 0;
17const ATA_REG_ERROR: usize = 1;
18const ATA_REG_SECCOUNT: usize = 2;
19const ATA_REG_LBA_LOW: usize = 3;
20const ATA_REG_LBA_MID: usize = 4;
21const ATA_REG_LBA_HIGH: usize = 5;
22const ATA_REG_DEVICE: usize = 6;
23const ATA_REG_STATUS: usize = 7;
24const ATA_REG_COMMAND: usize = 7;
25
26const ATA_SR_BSY: u8 = 0x80;
27const ATA_SR_DRDY: u8 = 0x40;
28const ATA_SR_DRQ: u8 = 0x08;
29const ATA_SR_ERR: u8 = 0x01;
30
31const ATA_CMD_IDENTIFY: u8 = 0xEC;
32
33const ATA_DEVICE_MASTER: u8 = 0xA0;
34const ATA_DEVICE_SLAVE: u8 = 0xB0;
35const ATA_DEVICE_LBA: u8 = 0x40;

37#[derive(Clone, Copy)]
38pub struct AtaChannel {
39    io_base: u16,
40    control_base: u16,
41    bus: u8,
42}
43
44impl AtaChannel {
45    /// Creates a new instance.
46    fn new(io_base: u16, bus: u8) -> Self {
47        Self {
48            io_base,
49            control_base: io_base + 0x206,
50            bus,
51        }
52    }
53
54    /// Performs the read8 operation.
55    fn read8(&self, offset: usize) -> u8 {
56        unsafe { x86_64::instructions::port::Port::new(self.io_base + offset as u16).read() }
57    }
58
59    /// Performs the write8 operation.
60    fn write8(&self, offset: usize, value: u8) {
61        unsafe { x86_64::instructions::port::Port::new(self.io_base + offset as u16).write(value) }
62    }
63
64    /// Performs the read16 operation.
65    fn read16(&self) -> u16 {
66        unsafe { x86_64::instructions::port::Port::new(self.io_base).read() }
67    }
68
69    /// Performs the write16 operation.
70    fn write16(&self, value: u16) {
71        unsafe { x86_64::instructions::port::Port::new(self.io_base).write(value) }
72    }
73
74    /// Performs the wait ready operation.
75    fn wait_ready(&self) -> Result<(), &'static str> {
76        for _ in 0..100000 {
77            let status = self.read8(ATA_REG_STATUS);
78            if (status & ATA_SR_BSY) == 0 {
79                return Ok(());
80            }
81            core::hint::spin_loop();
82        }
83        Err("ATA timeout")
84    }
85
86    /// Performs the wait drq operation.
87    fn wait_drq(&self) -> Result<(), &'static str> {
88        for _ in 0..100000 {
89            let status = self.read8(ATA_REG_STATUS);
90            if (status & ATA_SR_DRQ) != 0 {
91                return Ok(());
92            }
93            if (status & ATA_SR_ERR) != 0 {
94                return Err("ATA error");
95            }
96            core::hint::spin_loop();
97        }
98        Err("ATA timeout")
99    }
100
101    /// Performs the select device operation.
102    fn select_device(&self, _device: u8, lba: u64) {
103        let device_reg = ATA_DEVICE_MASTER | ATA_DEVICE_LBA | ((lba >> 24) & 0x0F) as u8;
104        self.write8(ATA_REG_DEVICE, device_reg);
105        self.read8(ATA_REG_STATUS);
106        for _ in 0..4 {
107            core::hint::spin_loop();
108        }
109    }
110
111    /// Performs the identify operation.
112    fn identify(&self, device: u8) -> Option<AtaDriveInfo> {
113        self.select_device(device, 0);
114        self.write8(ATA_REG_SECCOUNT, 0);
115        self.write8(ATA_REG_LBA_LOW, 0);
116        self.write8(ATA_REG_LBA_MID, 0);
117        self.write8(ATA_REG_LBA_HIGH, 0);
118        self.write8(ATA_REG_COMMAND, ATA_CMD_IDENTIFY);
119
120        let status = self.read8(ATA_REG_STATUS);
121        if status == 0 {
122            return None;
123        }
124
125        if let Err(_) = self.wait_ready() {
126            return None;
127        }
128
129        if let Err(_) = self.wait_drq() {
130            return None;
131        }
132
133        let mut buffer = [0u16; 256];
134        for i in 0..256 {
135            buffer[i] = self.read16();
136        }
137
138        let serial = Self::decode_identify_string(&buffer, 10, 20);
139        let model = Self::decode_identify_string(&buffer, 27, 54);
140        let capacity = (buffer[60] as u64) | ((buffer[61] as u64) << 16);
141
142        Some(AtaDriveInfo { model, serial, capacity })
143    }
144
145    /// Performs the decode identify string operation.
146    fn decode_identify_string(buffer: &[u16], start: usize, end: usize) -> String {
147        use alloc::string::String;
148        let mut s = String::new();
149        for i in start..end {
150            if i < buffer.len() {
151                let c = buffer[i];
152                let hi = (c >> 8) as u8;
153                let lo = (c & 0xFF) as u8;
154                if hi != 0 { s.push(hi as char); }
155                if lo != 0 { s.push(lo as char); }
156            }
157        }
158        s
159    }
160
161    /// Reads sector pio.
162    fn read_sector_pio(&self, device: u8, lba: u64, buffer: &mut [u8]) -> Result<(), &'static str> {
163        if buffer.len() < SECTOR_SIZE {
164            return Err("Buffer too small");
165        }
166
167        self.wait_ready()?;
168        self.select_device(device, lba);
169
170        self.write8(ATA_REG_SECCOUNT, 1);
171        self.write8(ATA_REG_LBA_LOW, (lba & 0xFF) as u8);
172        self.write8(ATA_REG_LBA_MID, ((lba >> 8) & 0xFF) as u8);
173        self.write8(ATA_REG_LBA_HIGH, ((lba >> 16) & 0xFF) as u8);
174
175        self.write8(ATA_REG_COMMAND, 0x24);
176        self.wait_drq()?;
177
178        let buf_ptr = buffer.as_mut_ptr() as *mut u16;
179        for i in 0..256 {
180            unsafe {
181                core::ptr::write_volatile(buf_ptr.add(i), self.read16());
182            }
183        }
184
185        self.wait_ready()?;
186        Ok(())
187    }
188
189    /// Writes sector pio.
190    fn write_sector_pio(&self, device: u8, lba: u64, buffer: &[u8]) -> Result<(), &'static str> {
191        if buffer.len() < SECTOR_SIZE {
192            return Err("Buffer too small");
193        }
194
195        self.wait_ready()?;
196        self.select_device(device, lba);
197
198        self.write8(ATA_REG_SECCOUNT, 1);
199        self.write8(ATA_REG_LBA_LOW, (lba & 0xFF) as u8);
200        self.write8(ATA_REG_LBA_MID, ((lba >> 8) & 0xFF) as u8);
201        self.write8(ATA_REG_LBA_HIGH, ((lba >> 16) & 0xFF) as u8);
202
203        self.write8(ATA_REG_COMMAND, 0x34);
204        self.wait_drq()?;
205
206        let buf_ptr = buffer.as_ptr() as *const u16;
207        for i in 0..256 {
208            unsafe {
209                self.write16(core::ptr::read_volatile(buf_ptr.add(i)));
210            }
211        }
212
213        self.wait_ready()?;
214        Ok(())
215    }
216}

218#[derive(Clone)]
219pub struct AtaDriveInfo {
220    pub model: String,
221    pub serial: String,
222    pub capacity: u64,
223}
224
225pub struct AtaDrive {
226    channel: AtaChannel,
227    device: u8,
228    info: AtaDriveInfo,
229    name: String,
230}
231
232unsafe impl Send for AtaDrive {}
233unsafe impl Sync for AtaDrive {}
234
235impl AtaDrive {
236    /// Creates a new instance.
237    pub fn new(channel: AtaChannel, device: u8) -> Option<Self> {
238        let info = channel.identify(device)?;
239        let name = format!("ata{}_{}", channel.bus, if device == ATA_DEVICE_MASTER { "master" } else { "slave" });
240        Some(Self { channel, device, info, name })
241    }
242
243    /// Performs the info operation.
244    pub fn info(&self) -> &AtaDriveInfo {
245        &self.info
246    }
247}
248
249impl BlockDevice for AtaDrive {
250    /// Reads sector.
251    fn read_sector(&self, sector: u64, buf: &mut [u8]) -> Result<(), BlockError> {
252        self.channel.read_sector_pio(self.device, sector, buf).map_err(|_| BlockError::IoError)
253    }
254
255    /// Writes sector.
256    fn write_sector(&self, sector: u64, buf: &[u8]) -> Result<(), BlockError> {
257        self.channel.write_sector_pio(self.device, sector, buf).map_err(|_| BlockError::IoError)
258    }
259
260    /// Performs the sector count operation.
261    fn sector_count(&self) -> u64 {
262        self.info.capacity
263    }
264}

266static ATA_DRIVES: Mutex<Vec<Arc<AtaDrive>>> = Mutex::new(Vec::new());
267static ATA_INITIALIZED: AtomicBool = AtomicBool::new(false);
268
269/// Performs the init operation.
270pub fn init() {
271    log::info!("[ATA] Scanning for legacy ATA/IDE devices...");
272
273    let channels = [
274        AtaChannel::new(ATA_PRIMARY_IO, 0),
275        AtaChannel::new(ATA_SECONDARY_IO, 1),
276    ];
277
278    for channel in &channels {
279        for device in [ATA_DEVICE_MASTER, ATA_DEVICE_SLAVE] {
280            if let Some(drive) = AtaDrive::new(channel.clone(), device) {
281                log::info!(
282                    "ATA: Found drive on bus{} device{}: {} ({} sectors)",
283                    channel.bus,
284                    if device == ATA_DEVICE_MASTER { "master" } else { "slave" },
285                    drive.info().model,
286                    drive.info().capacity
287                );
288                ATA_DRIVES.lock().push(Arc::new(drive));
289            }
290        }
291    }
292
293    ATA_INITIALIZED.store(true, Ordering::SeqCst);
294    log::info!("[ATA] Found {} drive(s)", ATA_DRIVES.lock().len());
295}
296
297/// Returns drive.
298pub fn get_drive(index: usize) -> Option<Arc<AtaDrive>> {
299    ATA_DRIVES.lock().get(index).cloned()
300}
301
302/// Returns first drive.
303pub fn get_first_drive() -> Option<Arc<AtaDrive>> {
304    ATA_DRIVES.lock().first().cloned()
305}
306
307/// Returns whether available.
308pub fn is_available() -> bool {
309    ATA_INITIALIZED.load(Ordering::Relaxed)
