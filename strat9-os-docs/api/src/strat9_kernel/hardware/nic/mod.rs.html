19struct NetDeviceEntry {
20    iface: String,
21    device: Arc<dyn NetworkDevice>,
22}
23
24static NET_DEVICES: RwLock<Vec<NetDeviceEntry>> = RwLock::new(Vec::new());

26/// Map a driver name to a FreeBSD-style interface prefix.
27///
28/// | Driver          | Prefix   | Example |
29/// |-----------------|----------|---------|
30/// | e1000 / Intel   | `em`     | `em0`   |
31/// | VirtIO-net      | `vtnet`  | `vtnet0`|
32/// | (other)         | `net`    | `net0`  |
33fn bsd_prefix(driver_name: &str) -> &'static str {
34    let lower = driver_name.as_bytes();
35    // Match common patterns without pulling in a full lowercase comparison
36    if lower.len() >= 4
37        && (lower[0] | 0x20) == b'e'
38        && (lower[1] | 0x20) == b'1'
39        && lower[2] == b'0'
40        && lower[3] == b'0'
41    {
42        return "em"; // Intel PRO/1000 family
43    }
44    if lower.len() >= 6
45        && (lower[0] | 0x20) == b'v'
46        && (lower[1] | 0x20) == b'i'
47        && (lower[2] | 0x20) == b'r'
48        && (lower[3] | 0x20) == b't'
49        && (lower[4] | 0x20) == b'i'
50        && (lower[5] | 0x20) == b'o'
51    {
52        return "vtnet"; // VirtIO
53    }
54    "net" // fallback
55}
56
57/// Counters per-prefix so that `em0`, `em1`, `vtnet0` are independent.
58static PREFIX_COUNTERS: RwLock<Vec<(String, usize)>> = RwLock::new(Vec::new());
59
60fn next_index_for(prefix: &str) -> usize {
61    let mut counters = PREFIX_COUNTERS.write();
62    for entry in counters.iter_mut() {
63        if entry.0 == prefix {
64            let idx = entry.1;
65            entry.1 += 1;
66            return idx;
67        }
68    }
69    counters.push((String::from(prefix), 1));
70    0
71}

73pub fn register_device(device: Arc<dyn NetworkDevice>) -> String {
74    let prefix = bsd_prefix(device.name());
75    let idx = next_index_for(prefix);
76    let iface = format!("{}{}", prefix, idx);
77    let mac = device.mac_address();
78    log::info!(
79        "[net] {} -> {} (MAC {:02x}:{:02x}:{:02x}:{:02x}:{:02x}:{:02x})",
80        device.name(),
81        iface,
82        mac[0],
83        mac[1],
84        mac[2],
85        mac[3],
86        mac[4],
87        mac[5],
88    );
89    let mut devs = NET_DEVICES.write();
90    devs.push(NetDeviceEntry {
91        iface: iface.clone(),
92        device,
93    });
94    iface
95}
96
97pub fn get_device(name: &str) -> Option<Arc<dyn NetworkDevice>> {
98    NET_DEVICES
99        .read()
100        .iter()
101        .find(|e| e.iface == name)
102        .map(|e| e.device.clone())
103}
104
105pub fn get_default_device() -> Option<Arc<dyn NetworkDevice>> {
106    NET_DEVICES.read().first().map(|e| e.device.clone())
107}
108
109pub fn list_interfaces() -> Vec<String> {
110    NET_DEVICES.read().iter().map(|e| e.iface.clone()).collect()
111}

113pub fn init() {
114    log::info!("[net] Scanning for network devices...");
115    // Probe modern Intel first, then legacy fallback.
116    e1000e_drv::init();
117    igc_drv::init();
118    e1000_drv::init();
119    pcnet_drv::init();
120    rtl8139_drv::init();
121    virtio_net::init();
122    if let Err(e) = scheme::register_net_scheme() {
123        log::warn!("[net] Failed to register net scheme: {:?}", e);
124    }
125}
