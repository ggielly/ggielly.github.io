71/// VirtIO Network Device driver
72pub struct VirtioNetDevice {
73    device: VirtioDevice,
74    rx_queue: SpinLock<Virtqueue>,
75    tx_queue: SpinLock<Virtqueue>,
76    mac_address: [u8; 6],
77    pub rx_frames: SpinLock<VecDeque<(PhysFrame, u8)>>, // Track allocated RX frames
78}

80// Send and Sync are safe because we use SpinLocks
81unsafe impl Send for VirtioNetDevice {}
82unsafe impl Sync for VirtioNetDevice {}

84impl VirtioNetDevice {
85    /// Initialize a VirtIO network device from a PCI device
86    pub unsafe fn new(pci_dev: PciDevice) -> Result<Self, &'static str> {
87        log::info!("VirtIO-net: Initializing device at {:?}", pci_dev.address);
88
89        // Create VirtIO device
90        let device = VirtioDevice::new(pci_dev)?;
91
92        // Reset device
93        device.reset();
94
95        // Acknowledge device
96        device.add_status(status::ACKNOWLEDGE as u8);
97
98        // Indicate we know how to drive it
99        device.add_status(status::DRIVER as u8);
100
101        // Read and negotiate features
102        let _device_features = device.read_device_features();
103
104        // Request MAC address feature
105        let guest_features = features::VIRTIO_NET_F_MAC | features::VIRTIO_NET_F_STATUS;
106        device.write_guest_features(guest_features);
107
108        // Features OK
109        device.add_status(status::FEATURES_OK as u8);
110
111        // Verify features OK
112        if device.get_status() & (status::FEATURES_OK as u8) == 0 {
113            return Err("Device doesn't support our feature set");
114        }
115
116        // Create virtqueues
117        // Queue 0: RX (receive)
118        // Queue 1: TX (transmit)
119        let rx_queue = Virtqueue::new(128)?;
120        let tx_queue = Virtqueue::new(128)?;
121
122        // Setup queues with device
123        device.setup_queue(0, &rx_queue);
124        device.setup_queue(1, &tx_queue);
125
126        // Read MAC address from device config space
127        // For legacy devices, MAC is at offset 20 + 0
128        let mut mac_address = [0u8; 6];
129        for i in 0..6 {
130            mac_address[i] = device.read_reg_u8(20 + i as u16);
131        }
132
133        log::info!(
134            "VirtIO-net: MAC address: {:02x}:{:02x}:{:02x}:{:02x}:{:02x}:{:02x}",
135            mac_address[0],
136            mac_address[1],
137            mac_address[2],
138            mac_address[3],
139            mac_address[4],
140            mac_address[5]
141        );
142
143        // Driver ready
144        device.add_status(status::DRIVER_OK as u8);
145
146        let net_device = Self {
147            device,
148            rx_queue: SpinLock::new(rx_queue),
149            tx_queue: SpinLock::new(tx_queue),
150            mac_address,
151            rx_frames: SpinLock::new(VecDeque::new()),
152        };
153
154        // Fill RX queue with buffers
155        net_device.refill_rx_queue()?;
156
157        Ok(net_device)
158    }
159
160    /// Fill the RX queue with receive buffers
161    fn refill_rx_queue(&self) -> Result<(), &'static str> {
162        let mut rx_queue = self.rx_queue.lock();
163        let mut rx_frames = self.rx_frames.lock();
164
165        // We want to keep some buffers in the RX queue
166        let current_filled = rx_frames.len();
167        let target_filled = 64;
168        let mut added = 0usize;
169
170        if current_filled >= target_filled {
171            return Ok(());
172        }
173
174        for _ in 0..(target_filled - current_filled) {
175            // Allocate buffer for header + MTU
176            let buf_size = NET_HDR_SIZE + net::MTU;
177            let buf_pages = (buf_size + 4095) / 4096;
178            let buf_order = buf_pages.next_power_of_two().trailing_zeros() as u8;
179
180            let buf_frame = match memory::allocate_frames(buf_order) {
181                Ok(frame) => frame,
182                Err(_) => break, // No more memory available
183            };
184
185            let buf_addr = buf_frame.start_address.as_u64();
186            let virt_addr = crate::memory::phys_to_virt(buf_addr);
187
188            // Zero the buffer (header needs to be zeroed mostly)
189            unsafe {
190                ptr::write_bytes(virt_addr as *mut u8, 0, buf_size);
191            }
192
193            // Add buffer to RX queue (device Writable)
194            match rx_queue.add_buffer(&[(buf_addr, buf_size as u32, true)]) {
195                Ok(_) => {
196                    rx_frames.push_back((buf_frame, buf_order));
197                    added += 1;
198                }
199                Err(_) => {
200                    // Queue full, free the buffer
201                    memory::free_frames(buf_frame, buf_order);
202                    break;
203                }
204            }
205        }
206
207        // Notify device about new RX buffers
208        if rx_queue.should_notify() {
209            self.device.notify_queue(0);
210        }
211
212        if rx_frames.is_empty() && current_filled == 0 && added == 0 {
213            return Err("Failed to allocate RX buffers");
214        }
215
216        Ok(())
217    }
218
219    /// Read link status from device
220    fn read_link_status(&self) -> u16 {
221        // Status is at offset 6 in device-specific config (offset 20 + 6 = 26)
222        self.device.read_reg_u16(26)
223    }
224}

226impl NetworkDevice for VirtioNetDevice {
227    /// Performs the name operation.
228    fn name(&self) -> &str {
229        "virtio-net"
230    }
231
232    /// Performs the receive operation.
233    fn receive(&self, buf: &mut [u8]) -> Result<usize, NetError> {
234        let mut rx_queue = self.rx_queue.lock();
235
236        // Check if there's a used buffer
237        if !rx_queue.has_used() {
238            return Err(NetError::NoPacket);
239        }
240
241        let (token, len) = rx_queue.get_used().ok_or(NetError::NoPacket)?;
242
243        let _desc_index = token as usize;
244        let _desc_table = rx_queue.desc_area(); // Physical address
245
246        let (frame, order) = self
247            .rx_frames
248            .lock()
249            .pop_front()
250            .ok_or(NetError::NotReady)?;
251
252        let buf_addr = frame.start_address.as_u64();
253        let virt_addr = crate::memory::phys_to_virt(buf_addr);
254
255        // Check if token matches what we expect?
256        // We can't easily without reading the descriptor.
257
258        let header_ptr = virt_addr as *const VirtioNetHeader;
259        let data_ptr = (virt_addr + NET_HDR_SIZE as u64) as *const u8;
260
261        let _header = unsafe { ptr::read(header_ptr) };
262        let packet_len = (len as usize).saturating_sub(NET_HDR_SIZE);
263
264        if buf.len() < packet_len {
265            // Buffer too small, packet lost
266            memory::free_frames(frame, order);
267            drop(rx_queue);
268            // We still need to refill.
269            let _ = self.refill_rx_queue();
270            return Err(NetError::BufferTooSmall);
271        }
272
273        // Copy packet data
274        if packet_len > 0 {
275            unsafe {
276                ptr::copy_nonoverlapping(data_ptr, buf.as_mut_ptr(), packet_len);
277            }
278        }
279
280        // Free the frame
281        memory::free_frames(frame, order);
282        drop(rx_queue);
283
284        // Refill RX queue
285        let _ = self.refill_rx_queue();
286
287        Ok(packet_len)
288    }
289
290    /// Performs the transmit operation.
291    fn transmit(&self, buf: &[u8]) -> Result<(), NetError> {
292        if buf.len() > net::MTU {
293            return Err(NetError::BufferTooSmall);
294        }
295
296        // Allocate TX buffer (header + data)
297        let buf_size = NET_HDR_SIZE + buf.len();
298        let buf_pages = (buf_size + 4095) / 4096;
299        let buf_order = buf_pages.next_power_of_two().trailing_zeros() as u8;
300
301        let buf_frame = memory::allocate_frames(buf_order).map_err(|_| NetError::NotReady)?;
302
303        let buf_addr = buf_frame.start_address.as_u64();
304        let virt_addr = crate::memory::phys_to_virt(buf_addr);
305
306        let header_ptr = virt_addr as *mut VirtioNetHeader;
307        let data_ptr = (virt_addr + NET_HDR_SIZE as u64) as *mut u8;
308
309        // Write header
310        unsafe {
311            ptr::write(header_ptr, VirtioNetHeader::default());
312            ptr::copy_nonoverlapping(buf.as_ptr(), data_ptr, buf.len());
313        }
314
315        // Submit to TX queue
316        let mut tx_queue = self.tx_queue.lock();
317        let token = tx_queue
318            .add_buffer(&[(buf_addr, buf_size as u32, false)]) // Device Readable
319            .map_err(|_| {
320                // Free buffer if failed
321                memory::free_frames(buf_frame, buf_order);
322                NetError::TxQueueFull
323            })?;
324
325        if tx_queue.should_notify() {
326            self.device.notify_queue(1);
327        }
328        drop(tx_queue);
329
330        // Wait for completion (simple spin for now)
331        loop {
332            let mut tx_queue = self.tx_queue.lock();
333            if tx_queue.has_used() {
334                if let Some((used_token, _)) = tx_queue.get_used() {
335                    // Assuming correct order for now or just waiting for *any* completion which matches ours
336                    if used_token == token {
337                        break;
338                    }
339                }
340            }
341            drop(tx_queue);
342            core::hint::spin_loop();
343        }
344
345        // Free TX buffer
346        memory::free_frames(buf_frame, buf_order);
347
348        Ok(())
349    }
350
351    /// Performs the mac address operation.
352    fn mac_address(&self) -> [u8; 6] {
353        self.mac_address
354    }
355
356    /// Performs the link up operation.
357    fn link_up(&self) -> bool {
358        let status = self.read_link_status();
359        status & net_status::VIRTIO_NET_S_LINK_UP != 0
360    }
361}

363/// Global VirtIO network device
364static VIRTIO_NET: SpinRwLock<Option<Arc<VirtioNetDevice>>> = SpinRwLock::new(None);
365
366/// Initialize VirtIO network device and register it in the global net registry.
367pub fn init() {
368    log::info!("VirtIO-net: Scanning for devices...");
369
370    // Prefer strict class-based probe (network/ethernet), with fallback to
371    // vendor+device for odd firmware/virtual setups.
372    let pci_dev = match pci::probe_first(pci::ProbeCriteria {
373        vendor_id: Some(pci::vendor::VIRTIO),
374        device_id: Some(pci::device::VIRTIO_NET),
375        class_code: Some(pci::class::NETWORK),
376        subclass: Some(pci::net_subclass::ETHERNET),
377        prog_if: None,
378    })
379    .or_else(|| pci::find_virtio_device(pci::device::VIRTIO_NET))
380    {
381        Some(dev) => dev,
382        None => {
383            log::warn!("VirtIO-net: No network device found");
384            return;
385        }
386    };
387
388    match unsafe { VirtioNetDevice::new(pci_dev) } {
389        Ok(device) => {
390            let arc = Arc::new(device);
391            *VIRTIO_NET.write() = Some(arc.clone());
392            net::register_device(arc);
393        }
394        Err(e) => {
395            log::error!("VirtIO-net: Failed to initialize device: {}", e);
396        }
397    }
398}
