1//! Virtual Memory Management (Paging) for Strat9-OS
2//!
3//! Uses the `x86_64` crate's `OffsetPageTable` which is designed for the HHDM
4//! (Higher Half Direct Map) pattern — exactly what Limine provides.
5//!
6//! Provides map/unmap/translate operations on the active page table.
7
8use x86_64::{
9    registers::control::Cr3,
10    structures::paging::{
11        FrameAllocator as X86FrameAllocator, Mapper, OffsetPageTable, Page, PageTable,
12        PageTableFlags, PhysFrame as X86PhysFrame, Size4KiB, Translate,
13    },
14    PhysAddr, VirtAddr,
15};

17/// Wrapper around our buddy allocator implementing the x86_64 crate's FrameAllocator trait.
18///
19/// This is used by `OffsetPageTable` when it needs to allocate intermediate page tables.
20pub struct BuddyFrameAllocator;
21
22// SAFETY: allocate_frame returns valid, unused, 4KiB-aligned physical frames
23// from the buddy allocator.
24unsafe impl X86FrameAllocator<Size4KiB> for BuddyFrameAllocator {
25    /// Performs the allocate frame operation.
26    fn allocate_frame(&mut self) -> Option<X86PhysFrame<Size4KiB>> {
27        let frame = crate::memory::allocate_frame().ok()?;
28        X86PhysFrame::from_start_address(frame.start_address).ok()
29    }
30}

32/// Paging initialization flag.
33static mut PAGING_READY: bool = false;
34
35/// Physical address of the kernel's level-4 page table (set at init, never changes).
36static mut KERNEL_CR3: PhysAddr = PhysAddr::new_truncate(0);
37
38/// Returns whether initialized.
39pub fn is_initialized() -> bool {
40    unsafe { *(&raw const PAGING_READY) }
41}
42
43/// Initialize the paging subsystem.
44///
45/// Reads the active CR3 (level-4 page table) and creates an `OffsetPageTable`
46/// mapper using the HHDM offset for physical-to-virtual translation.
47///
48/// Must be called after the buddy allocator and HHDM offset are initialized.
49pub fn init(hhdm_offset: u64) {
50    let phys_offset = VirtAddr::new(hhdm_offset);
51    let (level_4_frame, _flags) = Cr3::read();
52    let level_4_phys = level_4_frame.start_address().as_u64();
53    let level_4_virt = phys_offset + level_4_phys;
54
55    // SAFETY: Called once during single-threaded init. The HHDM offset correctly
56    // maps all physical RAM to virtual addresses. CR3 points to a valid page table
57    // set up by Limine.
58    unsafe {
59        let kcr3 = &raw mut KERNEL_CR3;
60        *kcr3 = level_4_frame.start_address();
61        let ready = &raw mut PAGING_READY;
62        *ready = true;
63    }
64
65    log::info!(
66        "Paging initialized: CR3={:#x}, HHDM={:#x}, L4 table @ {:#x}",
67        level_4_phys,
68        hhdm_offset,
69        level_4_virt.as_u64(),
70    );
71}

73/// Map a virtual page to a physical frame with the given flags.
74///
75/// Intermediate page tables are allocated from the buddy allocator as needed.
76pub fn map_page(
77    page: Page<Size4KiB>,
78    frame: X86PhysFrame<Size4KiB>,
79    flags: PageTableFlags,
80) -> Result<(), &'static str> {
81    if !is_initialized() {
82        return Err("Paging not initialized");
83    }
84    let phys_offset = VirtAddr::new(crate::memory::hhdm_offset());
85    let (level_4_frame, _) = Cr3::read();
86    let level_4_virt = phys_offset + level_4_frame.start_address().as_u64();
87    // SAFETY: level_4_virt points to the active CR3 PML4 via HHDM.
88    let mapper = unsafe { &mut *level_4_virt.as_mut_ptr::<PageTable>() };
89    let mut mapper = unsafe { OffsetPageTable::new(mapper, phys_offset) };
90    let mut allocator = BuddyFrameAllocator;
91
92    unsafe {
93        mapper
94            .map_to(page, frame, flags, &mut allocator)
95            .map_err(|_| "Failed to map page")?
96            .flush();
97    }
98    Ok(())
99}

101/// Unmap a virtual page, returning the physical frame it was mapped to.
102pub fn unmap_page(page: Page<Size4KiB>) -> Result<X86PhysFrame<Size4KiB>, &'static str> {
103    if !is_initialized() {
104        return Err("Paging not initialized");
105    }
106    let phys_offset = VirtAddr::new(crate::memory::hhdm_offset());
107    let (level_4_frame, _) = Cr3::read();
108    let level_4_virt = phys_offset + level_4_frame.start_address().as_u64();
109    // SAFETY: level_4_virt points to the active CR3 PML4 via HHDM.
110    let mapper = unsafe { &mut *level_4_virt.as_mut_ptr::<PageTable>() };
111    let mut mapper = unsafe { OffsetPageTable::new(mapper, phys_offset) };
112    let (frame, flush) = mapper.unmap(page).map_err(|_| "Failed to unmap page")?;
113    flush.flush();
114    Ok(frame)
115}

117/// Translate a virtual address to its mapped physical address.
118///
119/// Returns `None` if the address is not mapped.
120pub fn translate(addr: VirtAddr) -> Option<PhysAddr> {
121    if !is_initialized() {
122        return None;
123    }
124    let phys_offset = VirtAddr::new(crate::memory::hhdm_offset());
125    let (level_4_frame, _) = Cr3::read();
126    let level_4_virt = phys_offset + level_4_frame.start_address().as_u64();
127    // SAFETY: level_4_virt points to the active CR3 PML4 via HHDM.
128    let mapper = unsafe { &mut *level_4_virt.as_mut_ptr::<PageTable>() };
129    let mapper = unsafe { OffsetPageTable::new(mapper, phys_offset) };
130    mapper.translate_addr(addr)
131}

133/// Read the current CR3 value (physical address of the active level-4 page table).
134pub fn active_page_table() -> PhysAddr {
135    let (frame, _) = Cr3::read();
136    frame.start_address()
137}

139/// Return the physical address of the kernel's level-4 page table.
140///
141/// This is the CR3 value captured at init time — used by `AddressSpace::new_user()`
142/// to clone kernel mappings (PML4 entries 256..512) into new address spaces.
143pub fn kernel_l4_phys() -> PhysAddr {
144    // SAFETY: Written once during single-threaded init, read-only after that.
145    unsafe { *(&raw const KERNEL_CR3) }
146}

148/// Ensure a physical address is identity-mapped in the HHDM region.
149///
150/// If the page is not present, it is mapped with Read/Write permissions.
151/// This is used to lazily map MMIO or legacy BIOS regions (like ACPI tables)
152/// that might have been skipped by the bootloader's initial map.
153pub fn ensure_identity_map(phys_addr: u64) {
154    let virt_addr = crate::memory::phys_to_virt(phys_addr);
155    let page = Page::<Size4KiB>::containing_address(VirtAddr::new(virt_addr));
156    let frame = X86PhysFrame::containing_address(PhysAddr::new(phys_addr));
157
158    if translate(VirtAddr::new(virt_addr)).is_none() {
159        log::debug!(
160            "Identity mapping missing page: {:#x} -> {:#x}",
161            phys_addr,
162            virt_addr
163        );
164        // Map as Present | Writable (generic safe default for MMIO/BIOS)
165        let flags = PageTableFlags::PRESENT | PageTableFlags::WRITABLE;
166        if let Err(e) = map_page(page, frame, flags) {
167            log::error!("Failed to identity map {:#x}: {}", phys_addr, e);
168        }
169    }
170}

172/// Ensure a physical range is mapped in the HHDM region.
173pub fn ensure_identity_map_range(phys_base: u64, size: u64) {
174    if size == 0 {
175        return;
176    }
177    let page_size = 4096u64;
178    let start = phys_base & !(page_size - 1);
179    let end = phys_base
180        .saturating_add(size.saturating_sub(1))
181        .saturating_add(page_size)
182        & !(page_size - 1);
183    let mut p = start;
184    while p < end {
185        ensure_identity_map(p);
186        p = p.saturating_add(page_size);
187    }
188}
