1//! Boot-time Bump Allocator
2//!
3//! Provides a simple, lock-free (single-threaded boot) allocator for
4//! critical kernel structures like AP stacks before the full buddy
5//! allocator and scheduler are active.
6//!
7//! The pool lives in the kernel's .bss section (1 MiB).  All allocations
8//! are permanent — there is no deallocation.  This is intentional: boot
9//! structures (AP kernel stacks) must outlive the boot phase.
10//!
11//! NOTE: stacks allocated here do NOT have a guard page.  A stack overflow
12//! on an AP will silently corrupt adjacent boot-pool memory.  Once the
13//! scheduler creates tasks through the normal path (KernelStack::allocate),
14//! those stacks do get proper guard-page protection.
15
16use core::{
17    cell::UnsafeCell,
18    sync::atomic::{AtomicUsize, Ordering},
19};
20
21/// Size of the static boot allocation pool (1 MiB).
22const BOOT_POOL_SIZE: usize = 1024 * 1024;
23
24/// Wrapper to hold the boot pool in a `UnsafeCell` instead of `static mut`.
25///
26/// `static mut` is deprecated in Rust 2024 edition and any aliased reference
27/// to it is UB.  `UnsafeCell` is the correct primitive for interior mutability.
28#[repr(C, align(4096))]
29struct BootPool(UnsafeCell<[u8; BOOT_POOL_SIZE]>);
30
31// SAFETY: The boot pool is only accessed during the single-threaded BSP boot
32// phase (before APs are started).  The atomic bump pointer serializes access
33// if it were ever used concurrently.
34unsafe impl Sync for BootPool {}
35
36static BOOT_POOL: BootPool = BootPool(UnsafeCell::new([0; BOOT_POOL_SIZE]));
37
38/// Current offset in the boot pool (bump pointer).
39static BOOT_PTR: AtomicUsize = AtomicUsize::new(0);
40
41/// Allocate a block of memory from the boot pool.
42///
43/// Returns a pointer to a 16-byte-aligned region of `size` bytes, or `None`
44/// if the pool is exhausted.  This is intended for use during the
45/// single-threaded boot phase only.
46pub fn alloc(size: usize) -> Option<*mut u8> {
47    let aligned_size = (size + 15) & !15;
48
49    let old_ptr = BOOT_PTR.fetch_add(aligned_size, Ordering::Relaxed);
50    if old_ptr + aligned_size > BOOT_POOL_SIZE {
51        // Roll back so the counter doesn't drift (best-effort).
52        BOOT_PTR.fetch_sub(aligned_size, Ordering::Relaxed);
53        return None;
54    }
55
56    // SAFETY: `old_ptr` is within [0, BOOT_POOL_SIZE - aligned_size].
57    // The boot pool is only accessed during the single-threaded BSP phase.
58    let base = BOOT_POOL.0.get() as *mut u8;
59    unsafe { Some(base.add(old_ptr)) }
60}
61
62/// Helper to allocate a stack and return its top (highest address).
63pub fn alloc_stack(size: usize) -> Option<u64> {
64    let base = alloc(size)?;
65    Some(base as u64 + size as u64)
66}
