1// Heap allocator: slab sub-allocator + buddy fallback for large objects.
2//
3// Small allocations (effective size ≤ 2048 B) come from per-size-class slab
4// free lists.  Each slab class draws whole pages from the buddy allocator and
5// carves them into fixed-size blocks.  Freed blocks return to the slab free
6// list, so the buddy's page counter stabilises after warm-up instead of
7// growing on every tiny allocation.
8//
9// Large allocations (> 2048 B) go directly to the buddy allocator, exactly
10// as before.
11//
12// Lock ordering: SLAB_ALLOC (outer) → buddy allocator (inner, only when a slab
13// class needs a fresh page).  No code path holds the buddy lock and then tries
14// to acquire SLAB_ALLOC, so there is no deadlock.
15

21    },
22    sync::SpinLock,

30// ---------------------------------------------------------------------------
31// Slab size classes
32// ---------------------------------------------------------------------------
33
34/// Power-of-two block sizes handled by the slab allocator.
35const SLAB_SIZES: [usize; 9] = [8, 16, 32, 64, 128, 256, 512, 1024, 2048];
36const NUM_SLABS: usize = SLAB_SIZES.len();
37/// Allocations with effective size above this threshold bypass the slab.
38const MAX_SLAB_SIZE: usize = 2048;
39
40// ---------------------------------------------------------------------------
41// SlabState
42// ---------------------------------------------------------------------------
43
44/// Per-size-class free lists.  Each element is the head of an intrusive
45/// singly-linked list stored *in* the free blocks themselves.
46struct SlabState {
47    /// `free_lists[i]` is the head of the free list for `SLAB_SIZES[i]`.
48    free_lists: [*mut u8; NUM_SLABS],
49}
50
51// SAFETY: protected exclusively through `SLAB_ALLOC: SpinLock<SlabState>`.
52unsafe impl Send for SlabState {}
53unsafe impl Sync for SlabState {}
54
55impl SlabState {
56    /// Creates a new instance.
57    const fn new() -> Self {
58        SlabState {
59            free_lists: [ptr::null_mut(); NUM_SLABS],
60        }
61    }
62
63    /// Return the slab class index for `effective` bytes (already rounded up
64    /// via `max(size, align)`).  Panics if called with `effective > MAX_SLAB_SIZE`.
65    #[inline]
66    fn class_index(effective: usize) -> usize {
67        for (i, &s) in SLAB_SIZES.iter().enumerate() {
68            if effective <= s {
69                return i;
70            }
71        }
72        unreachable!("class_index called with effective > MAX_SLAB_SIZE")
73    }
74
75    /// Carve one buddy page into blocks of size `SLAB_SIZES[ci]` and prepend
76    /// them all to the free list for that class.
77    ///
78    /// Acquires the buddy allocator lock *inside* the caller's SLAB_ALLOC
79    /// lock — this is safe because the buddy allocator never calls back into
80    /// the heap allocator.
81    unsafe fn refill(&mut self, ci: usize) {
82        let slab_size = SLAB_SIZES[ci];
83
84        // Allocate one page (order-0) from the buddy allocator.
85        let frame = {
86            let buddy_lock = get_allocator();
87            let mut guard = buddy_lock.lock();
88            match guard.as_mut() {
89                Some(buddy) => match buddy.alloc(0) {
90                    Ok(f) => f,
91                    Err(_) => return, // OOM — caller will return null
92                },
93                None => return,
94            }
95            // `guard` (buddy lock) is released here
96        };
97
98        let page_virt = super::phys_to_virt(frame.start_address.as_u64()) as *mut u8;
99        let num_blocks = 4096 / slab_size;
100
101        // Link all blocks into the free list (highest address first so the
102        // first block handed out is at the lowest address — irrelevant for
103        // correctness but tidy).
104        let mut head = self.free_lists[ci];
105        for i in (0..num_blocks).rev() {
106            let block = page_virt.add(i * slab_size);
107            // Store the next-free pointer in the first word of the free block.
108            *(block as *mut *mut u8) = head;
109            head = block;
110        }
111        self.free_lists[ci] = head;
112    }
113
114    /// Pop a block from the free list for class `ci`, refilling from a buddy
115    /// page if the list is empty.  Returns null on OOM.
116    unsafe fn alloc_block(&mut self, ci: usize) -> *mut u8 {
117        if self.free_lists[ci].is_null() {
118            self.refill(ci);
119        }
120        let head = self.free_lists[ci];
121        if head.is_null() {
122            return ptr::null_mut();
123        }
124        // Read the next pointer stored at the start of the block.
125        let next = *(head as *const *mut u8);
126        self.free_lists[ci] = next;
127        head
128    }
129
130    /// Push a block back onto the free list for class `ci`.
131    unsafe fn dealloc_block(&mut self, ptr: *mut u8, ci: usize) {
132        // Overwrite the first word of the freed block with the current head.
133        *(ptr as *mut *mut u8) = self.free_lists[ci];
134        self.free_lists[ci] = ptr;
135    }
136}

138static SLAB_ALLOC: SpinLock<SlabState> = SpinLock::new(SlabState::new());

140// ---------------------------------------------------------------------------
141// GlobalAlloc implementation
142// ---------------------------------------------------------------------------
143
144pub struct LockedHeap;
145
146unsafe impl GlobalAlloc for LockedHeap {
147    /// Performs the alloc operation.
148    unsafe fn alloc(&self, layout: Layout) -> *mut u8 {
149        // Effective size must satisfy both the size and alignment requirements.
150        let effective = layout.size().max(layout.align());
151
152        if effective <= MAX_SLAB_SIZE {
153            // --- slab path ---
154            let ci = SlabState::class_index(effective);
155            let mut slab = SLAB_ALLOC.lock();
156            slab.alloc_block(ci)
157        } else {
158            // --- buddy path (large allocation) ---
159            let pages_needed = (effective + 4095) / 4096;
160            let max_pages = 1usize << MAX_ORDER;
161            if pages_needed > max_pages {
162                return ptr::null_mut();
163            }
164            let order = pages_needed.next_power_of_two().trailing_zeros() as u8;
165
166            let buddy_lock = get_allocator();
167            let mut guard = buddy_lock.lock();
168            if let Some(ref mut buddy) = *guard {
169                match buddy.alloc(order) {
170                    Ok(frame) => super::phys_to_virt(frame.start_address.as_u64()) as *mut u8,
171                    Err(_) => ptr::null_mut(),
172                }
173            } else {
174                ptr::null_mut()
175            }
176        }
177    }
178
179    /// Performs the dealloc operation.
180    unsafe fn dealloc(&self, ptr: *mut u8, layout: Layout) {
181        let effective = layout.size().max(layout.align());
182
183        if effective <= MAX_SLAB_SIZE {
184            // --- slab path: return block to free list ---
185            let ci = SlabState::class_index(effective);
186            let mut slab = SLAB_ALLOC.lock();
187            slab.dealloc_block(ptr, ci);
188        } else {
189            // --- buddy path: return page(s) to buddy ---
190            let pages_needed = (effective + 4095) / 4096;
191            let max_pages = 1usize << MAX_ORDER;
192            if pages_needed > max_pages {
193                return;
194            }
195            let order = pages_needed.next_power_of_two().trailing_zeros() as u8;
196
197            let hhdm = super::HHDM_OFFSET.load(core::sync::atomic::Ordering::Relaxed);
198            let phys_addr = (ptr as u64).wrapping_sub(hhdm);
199
200            let buddy_lock = get_allocator();
201            let mut guard = buddy_lock.lock();
202            if let Some(ref mut buddy) = *guard {
203                if let Ok(frame) = PhysFrame::from_start_address(PhysAddr::new(phys_addr)) {
204                    buddy.free(frame, order);
205                }
206            }
207        }
208    }
209}
