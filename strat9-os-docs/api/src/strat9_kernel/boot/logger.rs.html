1use log::{Level, LevelFilter, Metadata, Record};
2
3/// Simple serial port logger
4struct SerialLogger;
5
6static LOGGER: SerialLogger = SerialLogger;
7
8impl log::Log for SerialLogger {
9    fn enabled(&self, metadata: &Metadata) -> bool {
10        metadata.level() <= Level::Trace
11    }
12
13    fn log(&self, record: &Record) {
14        if self.enabled(record.metadata()) {
15            let (level_str, msg_color) = match record.level() {
16                Level::Error => ("\x1b[31mERROR\x1b[0m", "\x1b[31m"),
17                Level::Warn => ("\x1b[33mWARN\x1b[0m", "\x1b[33m"),
18                Level::Info => ("\x1b[32mINFO\x1b[0m", "\x1b[37m"),
19                Level::Debug => ("\x1b[90mDEBUG\x1b[0m", "\x1b[90m"),
20                Level::Trace => ("\x1b[90mTRACE\x1b[0m", "\x1b[90m"),
21            };
22
23            crate::arch::x86_64::serial::_print(format_args!(
24                "[{}] {}{}\x1b[0m\n",
25                level_str,
26                msg_color,
27                record.args()
28            ));
29        }
30    }
31
32    fn flush(&self) {}
33}
34
35/// Initialize the logger
36pub fn init() {
37    log::set_logger(&LOGGER)
38        .map(|()| log::set_max_level(LevelFilter::Trace))
39        .expect("Failed to set logger");
40}
