1//! SMP (Symmetric Multi-Processing) boot for x86_64.
2//!
3//! Boots Application Processors (APs) using the legacy INIT+SIPI sequence
4//! and parks them in an idle loop. Per-CPU data is initialized but no
5//! per-CPU scheduler is active yet.

132/// Busy-wait for the given number of microseconds (very rough).
133fn udelay(us: u32) {
134    for _ in 0..us {
135        io_wait();
136    }
137}

139/// Performs the ensure identity mapping operation.
140fn ensure_identity_mapping(phys_start: u64, length: usize) {
141    let start = phys_start & !0xFFFu64;
142    let end = (phys_start + length as u64 + 0xFFF) & !0xFFFu64;
143    let flags = PageTableFlags::PRESENT | PageTableFlags::WRITABLE;
144
145    let mut addr = start;
146    while addr < end {
147        let virt = VirtAddr::new(addr);
148        if let Some(mapped) = crate::memory::paging::translate(virt) {
149            if mapped.as_u64() != addr {
150                log::warn!(
151                    "SMP: identity map collision at {:#x} -> {:#x}",
152                    addr,
153                    mapped.as_u64()
154                );
155            }
156        } else {
157            let page = Page::<Size4KiB>::containing_address(virt);
158            let frame = PhysFrame::<Size4KiB>::containing_address(PhysAddr::new(addr));
159            if let Err(e) = crate::memory::paging::map_page(page, frame, flags) {
160                log::error!("SMP: failed to identity map {:#x}: {}", addr, e);
161            }
162        }
163        addr += 0x1000;
164    }
165}

167/// Performs the copy trampoline operation.
168fn copy_trampoline(cr3_phys: u64, stacks_ptr: *const u64) {
169    let tramp_len = (smp_trampoline_end as *const u8 as usize)
170        .saturating_sub(smp_trampoline as *const u8 as usize);
171
172    ensure_identity_mapping(TRAMPOLINE_PHYS_ADDR, tramp_len + 16);
173
174    let tramp_virt = memory::phys_to_virt(TRAMPOLINE_PHYS_ADDR) as *mut u8;
175
176    // SAFETY: trampoline destination is mapped and writable in HHDM.
177    unsafe {
178        core::ptr::copy_nonoverlapping(smp_trampoline as *const u8, tramp_virt, tramp_len);
179        let ptrs = tramp_virt.add(tramp_len) as *mut u64;
180        core::ptr::write_volatile(ptrs, cr3_phys);
181        core::ptr::write_volatile(ptrs.add(1), stacks_ptr as u64);
182    }
183}

185/// Performs the wait delivery operation.
186fn wait_delivery() {
187    const DELIVERY_STATUS: u32 = 1 << 12;
188    for _ in 0..1_000_000 {
189        // SAFETY: APIC initialized, ICR low is readable.
190        let val = unsafe { apic::read_reg(apic::REG_ICR_LOW) };
191        if val & DELIVERY_STATUS == 0 {
192            return;
193        }
194        core::hint::spin_loop();
195    }
196    log::warn!("SMP: IPI delivery timeout");
197}

199/// Performs the send ipi operation.
200fn send_ipi(apic_id: u32, value: u32) {
201    unsafe {
202        apic::write_reg(apic::REG_ESR, 0);
203        apic::write_reg(apic::REG_ESR, 0);
204        apic::write_reg(apic::REG_ICR_HIGH, apic_id << 24);
205        apic::write_reg(apic::REG_ICR_LOW, value);
206    }
207    wait_delivery();
208}
209
210/// Performs the send init sipi operation.
211fn send_init_sipi(apic_id: u32) {
212    // INIT IPI (assert)
213    send_ipi(apic_id, 0x0000_c500);
214    udelay(10_000);
215
216    // INIT de-assert
217    send_ipi(apic_id, 0x0000_8500);
218    udelay(200);
219
220    // SIPI twice, vector = 0x8 (0x8000 >> 12)
221    for _ in 0..2 {
222        send_ipi(apic_id, 0x0000_0608);
223        udelay(200);
224    }
225}

227/// Boot Application Processors.
228pub fn init() -> Result<usize, &'static str> {
229    if !apic::is_initialized() {
230        return Err("APIC not initialized");
231    }
232
233    BOOTED_CORES.store(1, Ordering::Release);
234
235    let madt_info = madt::parse_madt().ok_or("MADT not available")?;
236    let bsp_apic_id = apic::lapic_id();
237
238    if madt_info.local_apic_count <= 1 {
239        log::info!("SMP: single CPU system");
240        return Ok(1);
241    }
242
243    let mut max_apic_id: usize = 0;
244    for i in 0..madt_info.local_apic_count {
245        if let Some(ref entry) = madt_info.local_apics[i] {
246            max_apic_id = max_apic_id.max(entry.apic_id as usize);
247        }
248    }
249
250    let mut stacks: Vec<u64> = vec![0; max_apic_id + 1];
251    let cr3_phys = crate::memory::paging::kernel_l4_phys().as_u64();
252    let mut targets: Vec<u32> = Vec::new();
253    let mut expected: usize = 1;
254
255    for i in 0..madt_info.local_apic_count {
256        let Some(ref entry) = madt_info.local_apics[i] else {
257            continue;
258        };
259
260        let apic_id = entry.apic_id as u32;
261        if apic_id == bsp_apic_id {
262            continue;
263        }
264
265        let kernel_stack = KernelStack::allocate(crate::process::task::Task::DEFAULT_STACK_SIZE)?;
266        let stack_top = kernel_stack.virt_base.as_u64() + kernel_stack.size as u64;
267
268        if apic_id as usize >= stacks.len() {
269            log::warn!("SMP: APIC id {} out of stack array range", apic_id);
270            continue;
271        }
272
273        stacks[apic_id as usize] = stack_top;
274
275        let cpu_index =
276            percpu::register_cpu(apic_id).ok_or("SMP: exceeded MAX_CPUS for per-CPU data")?;
277        percpu::set_kernel_stack_top(cpu_index, stack_top);
278
279        AP_KERNEL_STACKS.lock().push(kernel_stack);
280        targets.push(apic_id);
281        expected += 1;
282    }
283
284    copy_trampoline(cr3_phys, stacks.as_ptr());
285
286    for apic_id in targets {
287        send_init_sipi(apic_id);
288    }
289
290    while BOOTED_CORES.load(Ordering::Acquire) < expected {
291        core::hint::spin_loop();
292    }
293
294    log::info!("SMP: {} cores online", expected);
295    Ok(expected)
296}

298/// First Rust function executed on APs after the trampoline.
299#[unsafe(no_mangle)]
300pub extern "C" fn smp_main() -> ! {
301    // Load IDT for this core (shared IDT is fine for now).
302    idt::init();
303
304    // Re-initialize Local APIC for this core (per-core registers).
305    apic::init_ap();
306
307    let apic_id = apic::lapic_id();
308    let cpu_index = match percpu::cpu_index_by_apic(apic_id) {
309        Some(idx) => idx,
310        None => {
311            // If we fall back to 0, this AP would share the BSP's
312            // CPU index and double-increment TICK_COUNT 2× timer speed.
313            log::error!("SMP AP: APIC id {} not registered — halting core", apic_id);
314            loop {
315                core::hint::spin_loop();
316            }
317        }
318    };
319
320    // Initialize per-CPU TSS/GDT.
321    crate::arch::x86_64::tss::init_cpu(cpu_index);
322    crate::arch::x86_64::gdt::init_cpu(cpu_index);
323    crate::arch::x86_64::percpu::init_gs_base(cpu_index);
324    crate::arch::x86_64::syscall::init();
325    crate::arch::x86_64::init_cpu_extensions();
326
327    if let Some(stack_top) = percpu::kernel_stack_top(cpu_index) {
328        crate::arch::x86_64::tss::set_kernel_stack_for(cpu_index, x86_64::VirtAddr::new(stack_top));
329    }
330
331    let _ = percpu::mark_online_by_apic(apic_id);
332    BOOTED_CORES.fetch_add(1, Ordering::Release);
333
334    // Start APIC timer on this CPU (uses cached calibration from BSP).
335    timer::start_apic_timer_cached();
336
337    // Start per-CPU scheduler (never returns).
338    crate::process::scheduler::schedule_on_cpu(cpu_index)
339}

342pub fn cpu_count() -> usize {
