1//! Global Descriptor Table (GDT) for Strat9-OS
2//!
3//! Sets up the GDT with kernel code/data segments, user segments for SYSRET,
4//! and a TSS descriptor. The TSS must be initialized before this module.
5//!
6//! ## Segment layout (SYSRET-compatible)
7//!
8//! ```text
9//! Index 0: Null
10//! Index 1: Kernel Code 64-bit  (CS=0x08)     ← SYSCALL loads CS from STAR[47:32]
11//! Index 2: Kernel Data          (SS=0x10)     ← SYSCALL loads SS
12//! Index 3: User Code 32-bit    (dummy, 0x18)  ← STAR[63:48] base
13//! Index 4: User Data            (0x20)        ← SYSRETQ: SS = base+8 | RPL3 = 0x23
14//! Index 5: User Code 64-bit    (0x28)        ← SYSRETQ: CS = base+16 | RPL3 = 0x2B
15//! Index 6-7: TSS (16-byte descriptor)
16//! ```
17//!
18//! STAR MSR: `[47:32]=0x08` (kernel CS), `[63:48]=0x18` (index 3).
19//! SYSRETQ: CS = 0x18+16 = 0x28 | RPL3 = 0x2B, SS = 0x18+8 = 0x20 | RPL3 = 0x23.
20
21use core::{
22    mem::MaybeUninit,

25use x86_64::{

30    structures::gdt::{Descriptor, GlobalDescriptorTable, SegmentSelector},

33/// GDT storage — per CPU
34static mut GDT: [MaybeUninit<GlobalDescriptorTable>; crate::arch::x86_64::percpu::MAX_CPUS] =
35    [const { MaybeUninit::uninit() }; crate::arch::x86_64::percpu::MAX_CPUS];
36
37/// Cached segment selectors after GDT is loaded (per CPU)
38static mut SELECTORS: [MaybeUninit<Selectors>; crate::arch::x86_64::percpu::MAX_CPUS] =
39    [const { MaybeUninit::uninit() }; crate::arch::x86_64::percpu::MAX_CPUS];
40static SELECTORS_INIT: [AtomicBool; crate::arch::x86_64::percpu::MAX_CPUS] =
41    [const { AtomicBool::new(false) }; crate::arch::x86_64::percpu::MAX_CPUS];
42
43#[derive(Copy, Clone)]
44struct Selectors {
45    kernel_code: SegmentSelector,
46    kernel_data: SegmentSelector,
47    #[allow(dead_code)]
48    user_code32: SegmentSelector,
49    user_data: SegmentSelector,
50    user_code64: SegmentSelector,
51    #[allow(dead_code)]
52    tss: SegmentSelector,
53}
54
55/// Initialize the GDT with kernel segments, user segments, and TSS.
56///
57/// **Prerequisite**: `tss::init()` must be called first.
58///
59/// The segment ordering is critical for SYSRET to work correctly.
60/// SYSRET expects: STAR[63:48] points to a base where base+0 = user_code32,
61/// base+8 = user_data, base+16 = user_code64.
62pub fn init() {
63    init_cpu(0);
64}
65
66/// Initialize the GDT for a given CPU index.
67pub fn init_cpu(cpu_index: usize) {
68    // SAFETY: Called during init (BSP) or AP bring-up before interrupts are enabled on that CPU.
69    unsafe {
70        let gdt = &mut *GDT[cpu_index].as_mut_ptr();
71        *gdt = GlobalDescriptorTable::new();
72
73        // Index 1: Kernel Code 64-bit (0x08)
74        let kernel_code = gdt.append(Descriptor::kernel_code_segment());
75        // Index 2: Kernel Data (0x10)
76        let kernel_data = gdt.append(Descriptor::kernel_data_segment());
77
78        // Index 3: User Code 32-bit dummy (0x18)
79        // SYSRET requires this slot to exist. We create a 32-bit code segment
80        // with DPL=3 that is never actually used for execution.
81        // Descriptor bits: Present | DPL=3 | Code segment | Readable | 32-bit
82        let user_code32_bits: u64 = (1 << 47)       // Present
83            | (3 << 45)                               // DPL = 3
84            | (1 << 44)                               // S = 1 (code/data)
85            | (1 << 43)                               // Executable
86            | (1 << 41)                               // Readable
87            | (1 << 54); // D = 1 (32-bit default)
88        let user_code32 = gdt.append(Descriptor::UserSegment(user_code32_bits));
89
90        // Index 4: User Data (0x20)
91        let user_data = gdt.append(Descriptor::user_data_segment());
92        // Index 5: User Code 64-bit (0x28)
93        let user_code64 = gdt.append(Descriptor::user_code_segment());
94
95        // Index 6-7: TSS (16-byte descriptor)
96        let tss_sel = gdt.append(Descriptor::tss_segment(super::tss::tss_for(cpu_index)));
97
98        gdt.load_unsafe();
99
100        // Reload segment registers with new selectors
101        CS::set_reg(kernel_code);
102        DS::set_reg(kernel_data);
103        SS::set_reg(kernel_data);
104
105        // Load TSS into task register
106        load_tss(tss_sel);
107
108        SELECTORS[cpu_index].write(Selectors {
109            kernel_code,
110            kernel_data,
111            user_code32,
112            user_data,
113            user_code64,
114            tss: tss_sel,
115        });
116        SELECTORS_INIT[cpu_index].store(true, Ordering::Release);
117
118        log::info!(
119            "GDT[CPU{}] loaded: CS={:#x} DS/SS={:#x} user32={:#x} user_data={:#x} user64={:#x} TSS={:#x}",
120            cpu_index,
121            kernel_code.0,
122            kernel_data.0,
123            user_code32.0,
124            user_data.0,
125            user_code64.0,
126            tss_sel.0,
127        );
128    }
129}

131/// Get the kernel code segment selector
132pub fn kernel_code_selector() -> SegmentSelector {
133    selectors_for(current_cpu_index()).kernel_code
134}
135
136/// Get the kernel data segment selector
137pub fn kernel_data_selector() -> SegmentSelector {
138    selectors_for(current_cpu_index()).kernel_data
139}
140
141/// Get the user code 64-bit segment selector (with RPL=3)
142pub fn user_code_selector() -> SegmentSelector {
143    let sel = selectors_for(current_cpu_index()).user_code64;
144    SegmentSelector(sel.0 | 3) // Set RPL=3
145}
146
147/// Get the user data segment selector (with RPL=3)
148pub fn user_data_selector() -> SegmentSelector {
149    let sel = selectors_for(current_cpu_index()).user_data;
150    SegmentSelector(sel.0 | 3) // Set RPL=3
151}
152
153/// Get the raw STAR MSR value for SYSCALL/SYSRET.
154///
155/// STAR[47:32] = kernel CS selector (for SYSCALL entry)
156/// STAR[63:48] = user code 32 selector base (for SYSRET calculation)
157pub fn star_msr_value() -> u64 {
158    let sels = selectors_for(current_cpu_index());
159    let kernel_cs = sels.kernel_code.0 as u64;
160    let user_base = sels.user_code32.0 as u64; // 0x18 (without RPL)
161    (kernel_cs << 32) | (user_base << 48)
162}
163
164fn selectors_for(cpu_index: usize) -> Selectors {
165    if !SELECTORS_INIT[cpu_index].load(Ordering::Acquire) {
166        panic!("GDT selectors for CPU{} not initialized", cpu_index);
167    }
168    // SAFETY: Initialized in init_cpu and stored for 'static lifetime.
169    unsafe { *SELECTORS[cpu_index].as_ptr() }
170}

172fn current_cpu_index() -> usize {
173    let apic_id = super::apic::lapic_id();
174    crate::arch::x86_64::percpu::cpu_index_by_apic(apic_id).unwrap_or(0)
175}
