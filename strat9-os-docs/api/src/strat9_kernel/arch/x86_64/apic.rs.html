1//! Local APIC (Advanced Programmable Interrupt Controller) driver
2//!
3//! Provides MMIO-based access to the Local APIC registers.
4//! The Local APIC base address is discovered via ACPI MADT and
5//! accessed through the HHDM (Higher Half Direct Map).
6
7use crate::memory;
8use core::sync::atomic::{AtomicBool, AtomicU64, Ordering};
9
10/// Whether the Local APIC has been initialized
11static APIC_INITIALIZED: AtomicBool = AtomicBool::new(false);
12
13/// Virtual base address of the Local APIC MMIO registers
14static APIC_BASE_VIRT: AtomicU64 = AtomicU64::new(0);
15
16/// Physical base address of the Local APIC MMIO registers (used by address-space init)
17static APIC_BASE_PHYS: AtomicU64 = AtomicU64::new(0);
18
19/// Return the physical base address of the LAPIC MMIO region, or 0 if not yet initialized.
20pub fn lapic_phys() -> u64 {
21    APIC_BASE_PHYS.load(Ordering::Relaxed)
22}

24// ===== Local APIC Register Offsets =====
25
26/// Local APIC ID Register
27const REG_ID: u32 = 0x020;
28/// Local APIC Version Register
29#[allow(dead_code)]
30const REG_VERSION: u32 = 0x030;
31/// Task Priority Register
32const REG_TPR: u32 = 0x080;
33/// End of Interrupt Register
34const REG_EOI: u32 = 0x0B0;
35/// Spurious Interrupt Vector Register
36const REG_SPURIOUS: u32 = 0x0F0;
37/// Error Status Register
38pub const REG_ESR: u32 = 0x280;
39/// Interrupt Command Register (low)
40pub const REG_ICR_LOW: u32 = 0x300;
41/// Interrupt Command Register (high)
42pub const REG_ICR_HIGH: u32 = 0x310;
43/// LVT Timer Register
44pub const REG_LVT_TIMER: u32 = 0x320;
45/// Timer Initial Count Register
46pub const REG_TIMER_INIT: u32 = 0x380;
47/// Timer Current Count Register
48pub const REG_TIMER_CURRENT: u32 = 0x390;
49/// Timer Divide Configuration Register
50pub const REG_TIMER_DIVIDE: u32 = 0x3E0;
51
52// ===== LVT Timer Modes =====
53
54/// LVT Timer: periodic mode (bit 17)
55pub const LVT_TIMER_PERIODIC: u32 = 1 << 17;
56/// LVT Timer: masked (bit 16)
57#[allow(dead_code)]
58pub const LVT_TIMER_MASKED: u32 = 1 << 16;
59/// Dedicated Local APIC timer interrupt vector (aligned with Theseus-style setup).
60pub const LVT_TIMER_VECTOR: u8 = 0x22;
61
62/// MSR address for APIC base
63const IA32_APIC_BASE_MSR: u32 = 0x1B;
64/// APIC global enable bit in IA32_APIC_BASE MSR
65const APIC_BASE_ENABLE: u64 = 1 << 11;
66
67/// Spurious interrupt vector number
68const SPURIOUS_VECTOR: u8 = 0xFF;
69
70/// Vector used for cross-CPU reschedule IPIs.
71pub const IPI_RESCHED_VECTOR: u8 = 0xE0;
72
73/// Vector used for TLB shootdown IPIs.
74pub const IPI_TLB_SHOOTDOWN_VECTOR: u8 = 0xF0;

76/// Check if APIC is present via CPUID
77pub fn is_present() -> bool {
78    let (_eax, _ebx, _ecx, edx) = super::cpuid(1, 0);
79    // CPUID.01H:EDX bit 9 = APIC
80    edx & (1 << 9) != 0
81}
82
83/// Check if x2APIC is supported via CPUID
84pub fn is_x2apic_supported() -> bool {
85    let (_eax, _ebx, ecx, _edx) = super::cpuid(1, 0);
86    // CPUID.01H:ECX bit 21 = x2APIC
87    ecx & (1 << 21) != 0
88}

90/// Get the current Local APIC ID.
91///
92/// When APIC is initialized, read the LAPIC ID register (authoritative at runtime).
93/// Before APIC init, fall back to CPUID initial APIC ID.
94pub fn lapic_id() -> u32 {
95    if APIC_INITIALIZED.load(Ordering::Relaxed) {
96        // SAFETY: guarded by APIC_INITIALIZED.
97        return unsafe { read_reg(REG_ID) >> 24 };
98    }
99
100    let (_eax, ebx, _ecx, _edx) = super::cpuid(1, 0);
101    (ebx >> 24) & 0xFF
102}
103
104/// Check if the Local APIC has been initialized
105pub fn is_initialized() -> bool {
106    APIC_INITIALIZED.load(Ordering::Relaxed)
107}

109/// Read a Local APIC register
110///
111/// # Safety
112/// APIC must be initialized (base address valid and mapped).
113pub unsafe fn read_reg(offset: u32) -> u32 {
114    let addr = APIC_BASE_VIRT.load(Ordering::Relaxed) + offset as u64;
115    // SAFETY: APIC MMIO is mapped via HHDM, volatile read required for MMIO
116    unsafe { core::ptr::read_volatile(addr as *const u32) }
117}
118
119/// Write a Local APIC register
120///
121/// # Safety
122/// APIC must be initialized (base address valid and mapped).
123pub unsafe fn write_reg(offset: u32, value: u32) {
124    let addr = APIC_BASE_VIRT.load(Ordering::Relaxed) + offset as u64;
125    // SAFETY: APIC MMIO is mapped via HHDM, volatile write required for MMIO
126    unsafe { core::ptr::write_volatile(addr as *mut u32, value) }
127}

129/// Initialize the Local APIC.
130///
131/// `madt_lapic_addr` is the LAPIC base physical address from the MADT.
132/// We use the MSR value as authoritative but log if it differs from MADT.
133pub fn init(madt_lapic_addr: u32) {
134    // Read the APIC base MSR to get the actual physical address
135    let apic_base_msr = super::rdmsr(IA32_APIC_BASE_MSR);
136    let apic_phys = apic_base_msr & 0xFFFF_F000; // Mask to get 4K-aligned base
137
138    if apic_phys != madt_lapic_addr as u64 {
139        log::warn!(
140            "LAPIC: MSR base 0x{:X} differs from MADT 0x{:X}, using MSR",
141            apic_phys,
142            madt_lapic_addr
143        );
144    }
145
146    // Enable the APIC globally via MSR if not already enabled
147    if apic_base_msr & APIC_BASE_ENABLE == 0 {
148        super::wrmsr(IA32_APIC_BASE_MSR, apic_base_msr | APIC_BASE_ENABLE);
149    }
150
151    // Convert physical base to virtual via HHDM
152    let apic_virt = memory::phys_to_virt(apic_phys);
153    APIC_BASE_VIRT.store(apic_virt, Ordering::Relaxed);
154    APIC_BASE_PHYS.store(apic_phys, Ordering::Relaxed);
155
156    // Diagnostic: log HHDM and computed addresses so we can detect HHDM=0 issues.
157    crate::serial_println!(
158        "[apic] init: hhdm={:#x} lapic_phys={:#x} lapic_virt={:#x}",
159        memory::hhdm_offset(),
160        apic_phys,
161        apic_virt
162    );
163    if apic_virt == apic_phys {
164        crate::serial_println!(
165            "[apic] WARN: lapic_virt == lapic_phys (HHDM offset is 0!) \
166             The LAPIC MMIO is identity-mapped at a low address. \
167             Kernel MMIO entries will be propagated to user page tables."
168        );
169    }
170
171    // SAFETY: APIC base is now set and mapped via HHDM
172    unsafe {
173        // Clear the Error Status Register (write twice per Intel SDM)
174        write_reg(REG_ESR, 0);
175        write_reg(REG_ESR, 0);
176
177        // Set Task Priority Register to 0 (accept all interrupts)
178        write_reg(REG_TPR, 0);
179
180        // Enable the APIC: set bit 8 (APIC Software Enable) + spurious vector
181        write_reg(REG_SPURIOUS, 0x100 | SPURIOUS_VECTOR as u32);
182    }
183
184    APIC_INITIALIZED.store(true, Ordering::Relaxed);
185
186    // SAFETY: APIC is initialized
187    let id = unsafe { read_reg(REG_ID) >> 24 };
188    log::info!(
189        "LAPIC: initialized at phys=0x{:X} virt=0x{:X} (ID={})",
190        apic_phys,
191        apic_virt,
192        id
193    );
194
195    // Check for x2APIC support
196    if is_x2apic_supported() {
197        log::info!("x2APIC: supported by CPU (will use xAPIC mode for now)");
198        // Note: We could switch to x2APIC mode here, but for compatibility
199        // we continue using xAPIC mode unless explicitly configured otherwise
200    }
201}

203/// Initialize per-core Local APIC state on Application Processors.
204///
205/// This assumes the APIC base is already mapped and `APIC_BASE_VIRT`
206/// has been set by the BSP during `init()`.
207pub fn init_ap() {
208    if !APIC_INITIALIZED.load(Ordering::Relaxed) {
209        log::warn!("LAPIC: init_ap called before init");
210        return;
211    }
212
213    // SAFETY: APIC base is mapped via HHDM and APIC is enabled.
214    unsafe {
215        // Clear ESR (twice per SDM)
216        write_reg(REG_ESR, 0);
217        write_reg(REG_ESR, 0);
218        // Accept all interrupts
219        write_reg(REG_TPR, 0);
220        // Software enable APIC + spurious vector
221        write_reg(REG_SPURIOUS, 0x100 | SPURIOUS_VECTOR as u32);
222    }
223}

225/// Send End-of-Interrupt to the Local APIC
226#[inline]
227pub fn eoi() {
228    // SAFETY: APIC is initialized when this is called from interrupt handlers
229    unsafe {
230        write_reg(REG_EOI, 0);
231    }
232}

234/// Send a "reschedule now" IPI to the CPU identified by `target_apic_id`.
235///
236/// Uses Fixed delivery mode (vector `IPI_RESCHED_VECTOR`), physical destination.
237/// This is fire-and-forget — no delivery status wait is needed for reschedule
238/// IPIs because a missed IPI only delays preemption by at most one timer tick.
239///
240/// # Safety
241/// APIC must be initialized (`APIC_INITIALIZED == true`).
242pub fn send_resched_ipi(target_apic_id: u32) {
243    // SAFETY: APIC base is valid and mapped; ICR MMIO is 32-bit aligned.
244    unsafe {
245        // Clear ESR before sending (Intel SDM §10.12.7).
246        write_reg(REG_ESR, 0);
247        // Set destination APIC ID in ICR high word.
248        write_reg(REG_ICR_HIGH, target_apic_id << 24);
249        // ICR low: bits[7:0] = vector, bits[10:8] = 000 (Fixed delivery),
250        //          bit 11 = 0 (Physical destination), bit 14 = 1 (Assert level),
251        //          bits[15] = 0 (Edge trigger).
252        // Writing ICR low triggers the IPI.
253        write_reg(REG_ICR_LOW, IPI_RESCHED_VECTOR as u32 | (1 << 14));
254    }
