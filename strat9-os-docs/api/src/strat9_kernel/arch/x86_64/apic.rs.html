1//! Local APIC (Advanced Programmable Interrupt Controller) driver
2//!
3//! Provides MMIO-based access to the Local APIC registers.
4//! The Local APIC base address is discovered via ACPI MADT and
5//! accessed through the HHDM (Higher Half Direct Map).
6
7use crate::memory;
8use core::sync::atomic::{AtomicBool, AtomicU64, Ordering};
9
10/// Whether the Local APIC has been initialized
11static APIC_INITIALIZED: AtomicBool = AtomicBool::new(false);
12
13/// Virtual base address of the Local APIC MMIO registers
14static APIC_BASE_VIRT: AtomicU64 = AtomicU64::new(0);
15
16// ===== Local APIC Register Offsets =====
17
18/// Local APIC ID Register
19const REG_ID: u32 = 0x020;
20/// Local APIC Version Register
21#[allow(dead_code)]
22const REG_VERSION: u32 = 0x030;
23/// Task Priority Register
24const REG_TPR: u32 = 0x080;
25/// End of Interrupt Register
26const REG_EOI: u32 = 0x0B0;
27/// Spurious Interrupt Vector Register
28const REG_SPURIOUS: u32 = 0x0F0;
29/// Error Status Register
30pub const REG_ESR: u32 = 0x280;
31/// Interrupt Command Register (low)
32pub const REG_ICR_LOW: u32 = 0x300;
33/// Interrupt Command Register (high)
34pub const REG_ICR_HIGH: u32 = 0x310;
35/// LVT Timer Register
36pub const REG_LVT_TIMER: u32 = 0x320;
37/// Timer Initial Count Register
38pub const REG_TIMER_INIT: u32 = 0x380;
39/// Timer Current Count Register
40pub const REG_TIMER_CURRENT: u32 = 0x390;
41/// Timer Divide Configuration Register
42pub const REG_TIMER_DIVIDE: u32 = 0x3E0;
43
44// ===== LVT Timer Modes =====
45
46/// LVT Timer: periodic mode (bit 17)
47pub const LVT_TIMER_PERIODIC: u32 = 1 << 17;
48/// LVT Timer: masked (bit 16)
49#[allow(dead_code)]
50pub const LVT_TIMER_MASKED: u32 = 1 << 16;
51/// Dedicated Local APIC timer interrupt vector (aligned with Theseus-style setup).
52pub const LVT_TIMER_VECTOR: u8 = 0x22;
53
54/// MSR address for APIC base
55const IA32_APIC_BASE_MSR: u32 = 0x1B;
56/// APIC global enable bit in IA32_APIC_BASE MSR
57const APIC_BASE_ENABLE: u64 = 1 << 11;
58
59/// Spurious interrupt vector number
60const SPURIOUS_VECTOR: u8 = 0xFF;
61
62/// Vector used for cross-CPU reschedule IPIs.
63pub const IPI_RESCHED_VECTOR: u8 = 0xE0;
64
65/// Vector used for TLB shootdown IPIs.
66pub const IPI_TLB_SHOOTDOWN_VECTOR: u8 = 0xF0;
67
68/// Check if APIC is present via CPUID
69pub fn is_present() -> bool {
70    let (_eax, _ebx, _ecx, edx) = super::cpuid(1, 0);
71    // CPUID.01H:EDX bit 9 = APIC
72    edx & (1 << 9) != 0
73}
74
75/// Get the current Local APIC ID.
76///
77/// When APIC is initialized, read the LAPIC ID register (authoritative at runtime).
78/// Before APIC init, fall back to CPUID initial APIC ID.
79pub fn lapic_id() -> u32 {
80    if APIC_INITIALIZED.load(Ordering::Relaxed) {
81        // SAFETY: guarded by APIC_INITIALIZED.
82        return unsafe { read_reg(REG_ID) >> 24 };
83    }
84
85    let (_eax, ebx, _ecx, _edx) = super::cpuid(1, 0);
86    (ebx >> 24) & 0xFF
87}
88
89/// Check if the Local APIC has been initialized
90pub fn is_initialized() -> bool {
91    APIC_INITIALIZED.load(Ordering::Relaxed)
92}

94/// Read a Local APIC register
95///
96/// # Safety
97/// APIC must be initialized (base address valid and mapped).
98pub unsafe fn read_reg(offset: u32) -> u32 {
99    let addr = APIC_BASE_VIRT.load(Ordering::Relaxed) + offset as u64;
100    // SAFETY: APIC MMIO is mapped via HHDM, volatile read required for MMIO
101    unsafe { core::ptr::read_volatile(addr as *const u32) }
102}
103
104/// Write a Local APIC register
105///
106/// # Safety
107/// APIC must be initialized (base address valid and mapped).
108pub unsafe fn write_reg(offset: u32, value: u32) {
109    let addr = APIC_BASE_VIRT.load(Ordering::Relaxed) + offset as u64;
110    // SAFETY: APIC MMIO is mapped via HHDM, volatile write required for MMIO
111    unsafe { core::ptr::write_volatile(addr as *mut u32, value) }
112}

114/// Initialize the Local APIC.
115///
116/// `madt_lapic_addr` is the LAPIC base physical address from the MADT.
117/// We use the MSR value as authoritative but log if it differs from MADT.
118pub fn init(madt_lapic_addr: u32) {
119    // Read the APIC base MSR to get the actual physical address
120    let apic_base_msr = super::rdmsr(IA32_APIC_BASE_MSR);
121    let apic_phys = apic_base_msr & 0xFFFF_F000; // Mask to get 4K-aligned base
122
123    if apic_phys != madt_lapic_addr as u64 {
124        log::warn!(
125            "LAPIC: MSR base 0x{:X} differs from MADT 0x{:X}, using MSR",
126            apic_phys,
127            madt_lapic_addr
128        );
129    }
130
131    // Enable the APIC globally via MSR if not already enabled
132    if apic_base_msr & APIC_BASE_ENABLE == 0 {
133        super::wrmsr(IA32_APIC_BASE_MSR, apic_base_msr | APIC_BASE_ENABLE);
134    }
135
136    // Convert physical base to virtual via HHDM
137    let apic_virt = memory::phys_to_virt(apic_phys);
138    APIC_BASE_VIRT.store(apic_virt, Ordering::Relaxed);
139
140    // SAFETY: APIC base is now set and mapped via HHDM
141    unsafe {
142        // Clear the Error Status Register (write twice per Intel SDM)
143        write_reg(REG_ESR, 0);
144        write_reg(REG_ESR, 0);
145
146        // Set Task Priority Register to 0 (accept all interrupts)
147        write_reg(REG_TPR, 0);
148
149        // Enable the APIC: set bit 8 (APIC Software Enable) + spurious vector
150        write_reg(REG_SPURIOUS, 0x100 | SPURIOUS_VECTOR as u32);
151    }
152
153    APIC_INITIALIZED.store(true, Ordering::Relaxed);
154
155    // SAFETY: APIC is initialized
156    let id = unsafe { read_reg(REG_ID) >> 24 };
157    log::info!(
158        "LAPIC: initialized at phys=0x{:X} virt=0x{:X} (ID={})",
159        apic_phys,
160        apic_virt,
161        id
162    );
163}

165/// Initialize per-core Local APIC state on Application Processors.
166///
167/// This assumes the APIC base is already mapped and `APIC_BASE_VIRT`
168/// has been set by the BSP during `init()`.
169pub fn init_ap() {
170    if !APIC_INITIALIZED.load(Ordering::Relaxed) {
171        log::warn!("LAPIC: init_ap called before init");
172        return;
173    }
174
175    // SAFETY: APIC base is mapped via HHDM and APIC is enabled.
176    unsafe {
177        // Clear ESR (twice per SDM)
178        write_reg(REG_ESR, 0);
179        write_reg(REG_ESR, 0);
180        // Accept all interrupts
181        write_reg(REG_TPR, 0);
182        // Software enable APIC + spurious vector
183        write_reg(REG_SPURIOUS, 0x100 | SPURIOUS_VECTOR as u32);
184    }
185}

187/// Send End-of-Interrupt to the Local APIC
188#[inline]
189pub fn eoi() {
190    // SAFETY: APIC is initialized when this is called from interrupt handlers
191    unsafe {
192        write_reg(REG_EOI, 0);
193    }
194}

196/// Send a "reschedule now" IPI to the CPU identified by `target_apic_id`.
197///
198/// Uses Fixed delivery mode (vector `IPI_RESCHED_VECTOR`), physical destination.
199/// This is fire-and-forget — no delivery status wait is needed for reschedule
200/// IPIs because a missed IPI only delays preemption by at most one timer tick.
201///
202/// # Safety
203/// APIC must be initialized (`APIC_INITIALIZED == true`).
204pub fn send_resched_ipi(target_apic_id: u32) {
205    // SAFETY: APIC base is valid and mapped; ICR MMIO is 32-bit aligned.
206    unsafe {
207        // Clear ESR before sending (Intel SDM §10.12.7).
208        write_reg(REG_ESR, 0);
209        // Set destination APIC ID in ICR high word.
210        write_reg(REG_ICR_HIGH, target_apic_id << 24);
211        // ICR low: bits[7:0] = vector, bits[10:8] = 000 (Fixed delivery),
212        //          bit 11 = 0 (Physical destination), bit 14 = 1 (Assert level),
213        //          bits[15] = 0 (Edge trigger).
214        // Writing ICR low triggers the IPI.
215        write_reg(REG_ICR_LOW, IPI_RESCHED_VECTOR as u32 | (1 << 14));
216    }
