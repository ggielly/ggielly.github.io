1//! Per-CPU data (x86_64)
2//!
3//! Minimal per-CPU tracking for SMP bring-up. This keeps CPU identity,
4//! online state, and per-CPU kernel stack top pointers.
5
6use core::sync::atomic::{AtomicBool, AtomicU32, AtomicU64, AtomicUsize, Ordering};
7
8/// Maximum number of CPUs supported for now.
9pub const MAX_CPUS: usize = 32;
10
11/// Offsets used by the SYSCALL entry (must match `PerCpuArch` layout).
12/// Note: cpu_index is at offset 0 (8 bytes).
13pub const USER_RSP_OFFSET: usize = 8;
14pub const KERNEL_RSP_OFFSET: usize = 16;
15
16/// Minimal per-CPU block accessed from assembly via GS base.
17///
18/// **Layout invariant** (`#[repr(C)]`):
19/// - `cpu_index` at offset  0 (8 bytes) — read by assembly as `gs:[0]`
20/// - `user_rsp`  at offset  8 (8 bytes)
21/// - `kernel_rsp` at offset 16 (8 bytes)
22///
23/// `AtomicU64` is `#[repr(C, align(8))]` and has the same size/alignment as
24/// `u64`, so the offsets declared in `USER_RSP_OFFSET`/`KERNEL_RSP_OFFSET` are
25/// preserved. The inline assembly in `current_cpu_index()` reads the raw 8
26/// bytes from `gs:[0]`, which are the inner `u64` of `AtomicU64`.
27#[repr(C)]
28pub struct PerCpuArch {
29    /// CPU index for this block. Written once during init via `AtomicU64::store`
30    /// (no raw-pointer cast needed), then only ever read. `AtomicU64` provides
31    /// proper `UnsafeCell` interior mutability so reads through `&PerCpuArch`
32    /// are not UB even though the field was written after the struct was placed
33    /// in a `static`.
34    pub cpu_index: AtomicU64, // offset 0 — read by assembly: `mov rax, gs:[0]`
35    pub user_rsp: AtomicU64,   // offset 8
36    pub kernel_rsp: AtomicU64, // offset 16
37}
38
39/// Per-CPU state.
40#[repr(C)]
41pub struct PerCpu {
42    pub arch: PerCpuArch,
43    present: AtomicBool,
44    online: AtomicBool,
45    apic_id: AtomicU32,
46    kernel_stack_top: AtomicU64,
47    tlb_ready: AtomicBool,
48    /// Preemption-disable depth counter.
49    /// When > 0, `maybe_preempt()` and `yield_task()` are no-ops on this CPU.
50    pub preempt_count: AtomicU32,
51}
52
53impl PerCpu {
54    /// Creates a new instance.
55    pub const fn new() -> Self {
56        Self {
57            arch: PerCpuArch {
58                cpu_index: AtomicU64::new(0),
59                user_rsp: AtomicU64::new(0),
60                kernel_rsp: AtomicU64::new(0),
61            },
62            present: AtomicBool::new(false),
63            online: AtomicBool::new(false),
64            apic_id: AtomicU32::new(0),
65            kernel_stack_top: AtomicU64::new(0),
66            tlb_ready: AtomicBool::new(false),
67            preempt_count: AtomicU32::new(0),
68        }
69    }
70
71    /// Performs the apic id operation.
72    pub fn apic_id(&self) -> u32 {
73        self.apic_id.load(Ordering::Acquire)
74    }
75
76    /// Performs the online operation.
77    pub fn online(&self) -> bool {
78        self.online.load(Ordering::Acquire)
79    }
80}

82static CPU_COUNT: AtomicUsize = AtomicUsize::new(0);
83static PERCPU: [PerCpu; MAX_CPUS] = [const { PerCpu::new() }; MAX_CPUS];
84
85/// Set to `true` once the BSP has called `init_gs_base`.
86///
87/// Used by `current_cpu_index()` to skip the serialising `rdmsr` null-guard
88/// on every hot-path call once GS is permanently initialised.
89///
90/// **Invariant**: APs always call `init_gs_base` *before* the first call to
91/// `current_cpu_index()` on that AP (see `smp_main` in `smp.rs`).  Once
92/// `BSP_GS_INITIALIZED` is true, any CPU that could possibly call
93/// `current_cpu_index()` must already have a valid GS base, so the direct
94/// `gs:[0]` read on the fast path is safe.
95static BSP_GS_INITIALIZED: AtomicBool = AtomicBool::new(false);
96
97/// Initialize the boot CPU (BSP) entry.
98pub fn init_boot_cpu(apic_id: u32) -> usize {
99    let cpu = &PERCPU[0];
100    cpu.present.store(true, Ordering::Release);
101    cpu.online.store(true, Ordering::Release);
102    cpu.apic_id.store(apic_id, Ordering::Release);
103    // AtomicU64::store through shared ref — no unsafe needed, UnsafeCell provides
104    // interior mutability so this write is well-defined.
105    cpu.arch.cpu_index.store(0, Ordering::Release);
106    cpu.tlb_ready.store(false, Ordering::Release);
107    CPU_COUNT.store(1, Ordering::Release);
108    0
109}

111/// Register a new CPU by APIC ID, returning its CPU index.
112pub fn register_cpu(apic_id: u32) -> Option<usize> {
113    for (idx, cpu) in PERCPU.iter().enumerate() {
114        if cpu
115            .present
116            .compare_exchange(false, true, Ordering::AcqRel, Ordering::Acquire)
117            .is_ok()
118        {
119            cpu.online.store(false, Ordering::Release);
120            cpu.apic_id.store(apic_id, Ordering::Release);
121            cpu.arch.cpu_index.store(idx as u64, Ordering::Release);
122            cpu.tlb_ready.store(false, Ordering::Release);
123            CPU_COUNT.fetch_add(1, Ordering::AcqRel);
124            return Some(idx);
125        }
126    }
127    None
128}

130/// Mark a CPU as online by APIC ID.
131pub fn mark_online_by_apic(apic_id: u32) -> Option<usize> {
132    for (idx, cpu) in PERCPU.iter().enumerate() {
133        if cpu.present.load(Ordering::Acquire) && cpu.apic_id.load(Ordering::Acquire) == apic_id {
134            cpu.online.store(true, Ordering::Release);
135            cpu.tlb_ready.store(false, Ordering::Release);
136            // Re-confirm index in arch block (no-op if already set, safe to repeat).
137            cpu.arch.cpu_index.store(idx as u64, Ordering::Release);
138            return Some(idx);
139        }
140    }
141    None
142}

171/// Initialize GS base for this CPU to point at its per-CPU block.
172///
173/// Sets both `IA32_GS_BASE` (0xC000_0101, current GS base) and
174/// `IA32_KERNEL_GS_BASE` (0xC000_0102) to `&PERCPU[cpu_index].arch`.
175///
176/// For the BSP (cpu_index == 0) this also sets `BSP_GS_INITIALIZED`, enabling
177/// the fast (non-serialising) path in `current_cpu_index()`.
178///
179/// **Ordering for APs**: in `smp_main`, `init_gs_base` is called before the
180/// first `current_cpu_index()` can execute on that AP, so it is safe to take
181/// the fast path on the AP after the BSP flag is visible.
182pub fn init_gs_base(cpu_index: usize) {
183    let base = &PERCPU[cpu_index].arch as *const PerCpuArch as u64;
184    // IA32_GS_BASE = 0xC000_0101, IA32_KERNEL_GS_BASE = 0xC000_0102.
185    // Both are set to the kernel per-CPU block.  When user-space TLS is
186    // added, only KERNEL_GS_BASE should be set here; GS_BASE will be
187    // managed by the context switcher.
188    crate::arch::x86_64::wrmsr(0xC000_0101, base);
189    crate::arch::x86_64::wrmsr(0xC000_0102, base);
190
191    if cpu_index == 0 {
192        // Release ordering: all prior init writes must be visible before any
193        // CPU sees the flag and takes the fast path in current_cpu_index().
194        BSP_GS_INITIALIZED.store(true, Ordering::Release);
195    }
196}

220/// Increment the preemption-disable depth for the current CPU.
221/// When depth > 0, the scheduler will not preempt this CPU.
222///
223/// Safe to call from any Ring-0 context **after** `init_gs_base` has run on
224/// this CPU.  If called before GS is initialised (early boot), `current_cpu_index`
225/// returns 0, which is always the BSP slot.  On the BSP itself that is correct;
226/// on an AP before its GS base is set the call is a no-op because the AP's
227/// scheduler is not yet running — incrementing slot 0 would be wrong, so we
228/// verify the GS index matches the slot we are about to touch.
229#[inline]
230pub fn preempt_disable() {
231    let idx = current_cpu_index();
232    // SAFETY: idx is clamped to [0, MAX_CPUS-1] by current_cpu_index().
233    // Additional guard: if GS is not yet set on this CPU, cpu_index will be
234    // the BSP's slot (0).  Skip if the slot's stored cpu_index disagrees with
235    // idx — that means GS isn't set here yet.
236    if PERCPU[idx].arch.cpu_index.load(Ordering::Relaxed) as usize == idx {
237        PERCPU[idx].preempt_count.fetch_add(1, Ordering::Relaxed);
238    }
239}
240
241/// Decrement the preemption-disable depth for the current CPU.
242/// Must be paired with exactly one prior call to `preempt_disable`.
243///
244/// Includes an underflow guard: if the counter is already 0 (mismatched call),
245/// the decrement is skipped and a warning is emitted rather than letting u32
246/// wrap to u32::MAX and disabling the scheduler permanently.
247#[inline]
248pub fn preempt_enable() {
249    let idx = current_cpu_index();
250    if PERCPU[idx].arch.cpu_index.load(Ordering::Relaxed) as usize == idx {
251        // Load first: prevent wrapping from 0 to u32::MAX.
252        let prev = PERCPU[idx].preempt_count.load(Ordering::Relaxed);
253        if prev > 0 {
254            PERCPU[idx].preempt_count.fetch_sub(1, Ordering::Relaxed);
255        } else {
256            // Mismatched preempt_enable — log and leave count at 0.
257            log::warn!("preempt_enable: underflow on CPU {} (mismatched call)", idx);
258        }
259    }
260}

262/// Returns `true` if preemption is currently allowed on this CPU
263/// (preempt_count == 0).
264#[inline]
265pub fn is_preemptible() -> bool {
266    let idx = current_cpu_index();
267    if PERCPU[idx].arch.cpu_index.load(Ordering::Relaxed) as usize != idx {
268        return true; // Early boot: no scheduler active, preemption is allowed.
269    }
270    PERCPU[idx].preempt_count.load(Ordering::Relaxed) == 0
271}

281/// Resolve current CPU index via a GS-relative load from offset 0.
282///
283/// **Hot path** (after `init_gs_base` has run on the BSP): a single
284/// non-serialising `mov rax, gs:[0]` — no MSR, no pipeline stall.
285///
286/// **Slow/early-boot path** (before `init_gs_base` is called the first time):
287/// reads `IA32_GS_BASE` via `rdmsr` as a null-guard; if the GS base is 0
288/// (hardware reset value, before our `wrmsr`) the function returns 0 (BSP
289/// slot) without touching GS.  Once the BSP calls `init_gs_base`, the slow
290/// path is never taken again.
291///
292/// **Corrupt-GS defence**: the returned index is clamped to [0, MAX_CPUS-1]
293/// so a bogus GS value can never produce an out-of-bounds array access.
294///
295/// **Invariant for APs**: `smp_main` always calls `init_gs_base` before the
296/// first `current_cpu_index()` on each AP, so the hot path is safe for APs.
297#[inline]
298pub fn current_cpu_index() -> usize {
299    // SAFETY:
300    // Hot path — `gs:[0]` reads `PerCpuArch::cpu_index` (AtomicU64, repr(C),
301    // offset 0).  Valid because `BSP_GS_INITIALIZED` is only set after the BSP's
302    // GS base has been written, and APs always write their GS base before
303    // executing any code that calls this function.
304    //
305    // Slow path — `rdmsr` is a privileged Ring-0 instruction, always valid here.
306    // The derference of `gs_base` is guarded by the != 0 check.
307    unsafe {
308        if BSP_GS_INITIALIZED.load(Ordering::Acquire) {
309            // Fast path: GS is valid on every CPU that can run kernel code now.
310            let idx: u64;
311            core::arch::asm!(
312                "mov {idx}, gs:[0]",
313                idx = out(reg) idx,
314                options(nostack, preserves_flags, readonly),
315            );
316            return (idx as usize).min(MAX_CPUS - 1);
317        }
318
319        // Slow path: early boot — GS not yet set on any CPU.
320        // `rdmsr` reads IA32_GS_BASE (0xC000_0101) as a null-guard.
321        let lo: u32;
322        let hi: u32;
323        core::arch::asm!(
324            "rdmsr",
325            in("ecx")  0xC000_0101u32,
326            out("eax") lo,
327            out("edx") hi,
328            options(nostack, preserves_flags),
329        );
330        let gs_base = (lo as u64) | ((hi as u64) << 32);
331        if gs_base == 0 {
332            return 0; // GS not yet initialised — early boot, return BSP slot.
333        }
334
335        // GS is set but BSP_GS_INITIALIZED wasn't yet visible (narrow race
336        // between wrmsr and the store; take the GS-relative read here too).
337        let idx: u64;
338        core::arch::asm!(
339            "mov {idx}, gs:[0]",
340            idx = out(reg) idx,
341            options(nostack, preserves_flags, readonly),
342        );
343        (idx as usize).min(MAX_CPUS - 1)
344    }
345}

369/// Returns true iff CPU `index` is online and ready for TLB shootdown.
370pub fn tlb_ready(index: usize) -> bool {
371    PERCPU
372        .get(index)
373        .map(|cpu| {
374            cpu.present.load(Ordering::Acquire)
375                && cpu.online.load(Ordering::Acquire)
376                && cpu.tlb_ready.load(Ordering::Acquire)
377        })
378        .unwrap_or(false)
