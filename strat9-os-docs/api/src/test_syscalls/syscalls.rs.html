42/// Implements log u64.
43fn log_u64(mut value: u64) {
44    let mut buf = [0u8; 21];
45    if value == 0 {
46        log("0");
47        return;
48    }
49    let mut i = buf.len();
50    while value > 0 {
51        i -= 1;
52        buf[i] = b'0' + (value % 10) as u8;
53        value /= 10;
54    }
55    let s = unsafe { core::str::from_utf8_unchecked(&buf[i..]) };
56    log(s);
57}

59/// Implements log hex u64.
60fn log_hex_u64(mut value: u64) {
61    let mut buf = [0u8; 16];
62    for i in (0..16).rev() {
63        let nibble = (value & 0xF) as u8;
64        buf[i] = if nibble < 10 {
65            b'0' + nibble
66        } else {
67            b'a' + (nibble - 10)
68        };
69        value >>= 4;
70    }
71    log("0x");
72    let s = unsafe { core::str::from_utf8_unchecked(&buf) };
73    log(s);
74}

106/// Implements check ok.
107fn check_ok(ctx: &mut Ctx, label: &str, res: core::result::Result<usize, Error>) -> Option<usize> {
108    match res {
109        Ok(v) => {
110            ok(ctx, label, v);
111            Some(v)
112        }
113        Err(e) => {
114            fail(ctx, label, e);
115            None
116        }
117    }
118}

120/// Implements check expect one of.
121fn check_expect_one_of(
122    ctx: &mut Ctx,
123    label: &str,
124    res: core::result::Result<usize, Error>,
125    e1: Error,
126    e2: Error,
127) {
128    match res {
129        Ok(v) => {
130            if e1 == Error::Unknown(0) || e2 == Error::Unknown(0) {
131                ok(ctx, label, v);
132            } else {
133                ctx.fail += 1;
134                log_err("[FAIL] ");
135                log_err(label);
136                log_err(" -> expected error but got OK=");
137                log_u64(v as u64);
138                log_err("\n");
139            }
140        }
141        Err(e) => {
142            if e == e1 || e == e2 {
143                ok(ctx, label, 0);
144            } else {
145                fail(ctx, label, e);
146            }
147        }
148    }
149}

151/// Implements check expect err.
152fn check_expect_err(ctx: &mut Ctx, label: &str, res: core::result::Result<usize, Error>, expected: Error) {
153    match res {
154        Ok(v) => {
155            ctx.fail += 1;
156            log_err("[FAIL] ");
157            log_err(label);
158            log_err(" -> expected ");
159            log_err(expected.name());
160            log_err(" got OK=");
161            log_u64(v as u64);
162            log_err("\n");
163        }
164        Err(e) => {
165            if e == expected {
166                ok(ctx, label, 0);
167            } else {
168                fail(ctx, label, e);
169            }
170        }
171    }

175fn test_process_and_ids(ctx: &mut Ctx) {
176    section("Process IDs / Session / Group / Credentials");
177
178    let _pid = check_ok(ctx, "getpid()", call::getpid()).unwrap_or(0);

191    let cur_gid = unsafe { syscall1(SYS_GETGID, 0) }.unwrap_or(0);
192    let _ = check_ok(ctx, "raw SYS_SETUID(current uid)", unsafe { syscall1(SYS_SETUID, cur_uid) });

196/// Implements test memory.
197fn test_memory(ctx: &mut Ctx) {
198    section("Memory: brk / mmap / mprotect / mremap / munmap");
199
200    let base = check_ok(ctx, "brk(0)", call::brk(0)).unwrap_or(0);
201    let grow = base + PAGE_SIZE * 2;
202    let _ = check_ok(ctx, "brk(grow)", call::brk(grow));
203    let _ = check_ok(ctx, "brk(shrink)", call::brk(base));
204
205    let mapped = check_ok(
206        ctx,
207        "SYS_MMAP anon private RW 2 pages",
208        unsafe {
209            syscall6(
210                number::SYS_MMAP,
211                0,
212                PAGE_SIZE * 2,
213                PROT_READ | PROT_WRITE,
214                MAP_PRIVATE | MAP_ANON,
215                0,
216                0,
217            )
218        },
219    )
220    .unwrap_or(0);
221
222    if mapped != 0 {
223        let ptr = mapped as *mut u8;
224        unsafe {
225            core::ptr::write_volatile(ptr, 0xAA);
226            core::ptr::write_volatile(ptr.add(PAGE_SIZE), 0xBB);
227        }
228    }
229
230    let _ = check_ok(
231        ctx,
232        "SYS_MPROTECT RO",
233        unsafe { syscall3(number::SYS_MPROTECT, mapped, PAGE_SIZE * 2, PROT_READ) },
234    );
235    let _ = check_ok(
236        ctx,
237        "SYS_MPROTECT RW",
238        unsafe {
239            syscall3(
240                number::SYS_MPROTECT,
241                mapped,
242                PAGE_SIZE * 2,
243                PROT_READ | PROT_WRITE,
244            )
245        },
246    );
247    let remapped = check_ok(
248        ctx,
249        "SYS_MREMAP grow to 3 pages (MAYMOVE)",
250        unsafe { syscall4(number::SYS_MREMAP, mapped, PAGE_SIZE * 2, PAGE_SIZE * 3, MREMAP_MAYMOVE) },
251    )
252    .unwrap_or(mapped);
253    let _ = check_ok(
254        ctx,
255        "SYS_MREMAP shrink back to 2 pages",
256        unsafe { syscall4(number::SYS_MREMAP, remapped, PAGE_SIZE * 3, PAGE_SIZE * 2, MREMAP_MAYMOVE) },
257    );
258    let _ = check_ok(
259        ctx,
260        "SYS_MUNMAP final",
261        unsafe { syscall2(number::SYS_MUNMAP, remapped, PAGE_SIZE * 2) },
262    );
263}

438/// Implements test handles.
439fn test_handles(ctx: &mut Ctx) {
440    section("Handle syscalls via semaphore handle");
441
442    let sem = check_ok(
443        ctx,
444        "SYS_SEM_CREATE initial=1",
445        unsafe { syscall1(number::SYS_SEM_CREATE, 1) },
446    )
447    .unwrap_or(usize::MAX);
448    if sem == usize::MAX {
449        return;
450    }
451
452    let mut info = data::HandleInfo {
453        resource_type: 0,
454        permissions: 0,
455        resource: 0,
456    };
457    let _ = check_ok(ctx, "handle_info(sem)", call::handle_info(sem, &mut info));
458    log("[info] resource_type=");
459    log_u64(info.resource_type as u64);
460    log(" perms=");
461    log_hex_u64(info.permissions as u64);
462    log(" resource=");
463    log_hex_u64(info.resource);
464    log("\n");
465
466    let _ = check_ok(ctx, "handle_wait(sem immediate)", call::handle_wait_timeout(sem, 0));
467    let _ = check_ok(ctx, "SYS_SEM_WAIT consume token", unsafe { syscall1(number::SYS_SEM_WAIT, sem) });
468    check_expect_err(
469        ctx,
470        "handle_wait_timeout(sem empty, 1ms) -> ETIMEDOUT",
471        call::handle_wait_timeout(sem, 1_000_000),
472        Error::TimedOut,
473    );
474    let _ = check_ok(ctx, "SYS_SEM_POST produce token", unsafe { syscall1(number::SYS_SEM_POST, sem) });
475    let _ = check_ok(ctx, "handle_wait(sem after post)", call::handle_wait_timeout(sem, 1_000_000));
476
477    let self_pid = call::getpid().unwrap_or(0);
478    let dup = check_ok(ctx, "handle_grant(self_pid)", call::handle_grant(sem, self_pid)).unwrap_or(usize::MAX);
479    if dup != usize::MAX {
480        let _ = check_ok(ctx, "handle_revoke(dup)", call::handle_revoke(dup));
481    }
482    match unsafe { syscall1(number::SYS_SEM_CLOSE, sem) } {
483        Ok(v) => ok(ctx, "SYS_SEM_CLOSE(original)", v),
484        Err(Error::NotFound) | Err(Error::BadHandle) => ok(ctx, "SYS_SEM_CLOSE(original already cleaned)", 0),
485        Err(e) => fail(ctx, "SYS_SEM_CLOSE(original)", e),
486    }
487}
