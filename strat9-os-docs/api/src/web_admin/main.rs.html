31#[alloc_error_handler]
32/// Implements alloc error.
33fn alloc_error(layout: Layout) -> ! {
34    let mut buf = [0u8; 80];
35    let n = {
36        let mut w = BufWriter::new(&mut buf);
37        let _ = write!(w, "[web-admin] OOM: {} bytes align {}\n", layout.size(), layout.align());
38        w.len()
39    };
40    let _ = call::debug_log(&buf[..n]);
41    call::exit(12)
42}

57struct BufWriter<'a> {
58    buf: &'a mut [u8],
59    pos: usize,
60}
61
62impl<'a> BufWriter<'a> {
63    /// Creates a new instance.
64    fn new(buf: &'a mut [u8]) -> Self {
65        Self { buf, pos: 0 }
66    }
67    /// Implements len.
68    fn len(&self) -> usize {
69        self.pos
70    }
71}
72
73impl core::fmt::Write for BufWriter<'_> {
74    /// Writes str.
75    fn write_str(&mut self, s: &str) -> core::fmt::Result {
76        let bytes = s.as_bytes();
77        let avail = self.buf.len().saturating_sub(self.pos);
78        let n = bytes.len().min(avail);
79        self.buf[self.pos..self.pos + n].copy_from_slice(&bytes[..n]);
80        self.pos += n;
81        Ok(())
82    }
83}

90static CONFIG: picoserve::Config = picoserve::Config::const_default().close_connection_after_response();

92#[unsafe(no_mangle)]
93/// Implements start.
94pub extern "C" fn _start() -> ! {
95    log("[web-admin] Strat9 Web Admin starting on port 8080\n");
96
97    let app = routes::build_router();
98    let mut conn_count: u64 = 0;
99
100    loop {
101        let fd = net::open_listener(LISTEN_PORT);
102        conn_count += 1;
103
104        executor::block_on(async {
105            let socket = io::TcpSocket::new(fd);
106            let timer = io::Strat9Timer;
107            let mut http_buf = [0u8; HTTP_BUF_SIZE];
108
109            let server = picoserve::Server::custom(&app, timer, &CONFIG, &mut http_buf);
110
111            match server.serve(socket).await {
112                Ok(info) => {
113                    let msg = alloc::format!(
114                        "[web-admin] conn #{}: {} reqs\n",
115                        conn_count,
116                        info.handled_requests_count
117                    );
118                    log(&msg);
119                }
120                Err(_) => {
121                    log("[web-admin] connection error\n");
122                }
123            }
124        });
125    }
