1use alloc::{string::String, vec::Vec};
2use core::sync::atomic::{AtomicU64, Ordering};
3use crate::{BusChild, BusDriver, BusError, PowerState};

6const L3_AGENT_CONTROL: usize = 0x020;
7const L3_AGENT_STATUS: usize = 0x028;
8const L3_ERROR_LOG: usize = 0x058;
9const L3_ERROR_LOG_ADDR: usize = 0x060;
10const L3_SI_FLAG_STATUS_0: usize = 0x510;
11const L3_SI_FLAG_STATUS_1: usize = 0x530;
12
13const L3_AGENT_STATUS_CLEAR_IA: u32 = 0x1000_0000;
14const L3_AGENT_STATUS_CLEAR_TA: u32 = 0x0100_0000;
15
16const L3_ERROR_LOG_MULTI: u32 = 1 << 31;
17const L3_ERROR_LOG_SECONDARY: u32 = 1 << 30;
18const L3_ERROR_LOG_CODE_MASK: u32 = 0x0F00_0000;
19const L3_ERROR_LOG_CODE_SHIFT: u32 = 24;
20const L3_ERROR_LOG_INITID_MASK: u32 = 0x0000_FF00;
21const L3_ERROR_LOG_INITID_SHIFT: u32 = 8;
22const L3_ERROR_LOG_CMD_MASK: u32 = 0x0000_0007;
23
24const COMPATIBLE: &[&str] = &["ti,omap3-l3-smx"];
25
26const L3_ERROR_CODES: &[&str] = &[
27    "no error",
28    "unsupported command",
29    "address hole",
30    "protection violation",
31    "in-band error",
32    "request timeout (not accepted)",
33    "request timeout (no response)",
34];
35
36const L3_APP_BASES: &[usize] = &[
37    0x1400, 0x1800, 0x1C00, 0x4400, 0x4000,
38    0x5800, 0x5400, 0x4C00, 0x5000, 0x3000,
39];
40
41const L3_DEBUG_BASES: &[usize] = &[0x1400, 0x5C00, 0x1800];
42
43pub struct Omap3L3ErrorInfo {
44    pub code: u32,
45    pub initiator: u32,
46    pub cmd: u32,
47    pub address: u32,
48    pub is_multi: bool,
49    pub is_secondary: bool,
50}
51
52impl Omap3L3ErrorInfo {
53    pub fn code_str(&self) -> &'static str {
54        L3_ERROR_CODES.get(self.code as usize).unwrap_or(&"unknown")
55    }
56}
57
58pub struct OmapL3Smx {
59    regs: MmioRegion,
60    error_count: AtomicU64,
61    power_state: PowerState,
62}
63
64impl OmapL3Smx {
65    pub fn new() -> Self {
66        Self {
67            regs: MmioRegion::new(),
68            error_count: AtomicU64::new(0),
69            power_state: PowerState::Off,
70        }
71    }
72
73    pub fn read_error_at(&self, base_offset: usize) -> Omap3L3ErrorInfo {
74        let err_log = self.regs.read32(base_offset + L3_ERROR_LOG);
75        let address = self.regs.read32(base_offset + L3_ERROR_LOG_ADDR);
76
77        let code = (err_log & L3_ERROR_LOG_CODE_MASK) >> L3_ERROR_LOG_CODE_SHIFT;
78        let initiator = (err_log & L3_ERROR_LOG_INITID_MASK) >> L3_ERROR_LOG_INITID_SHIFT;
79        let cmd = err_log & L3_ERROR_LOG_CMD_MASK;
80
81        Omap3L3ErrorInfo {
82            code,
83            initiator,
84            cmd,
85            address,
86            is_multi: (err_log & L3_ERROR_LOG_MULTI) != 0,
87            is_secondary: (err_log & L3_ERROR_LOG_SECONDARY) != 0,
88        }
89    }
90
91    pub fn clear_error_at(&self, base_offset: usize) {
92        self.regs.write32(
93            base_offset + L3_AGENT_STATUS,
94            L3_AGENT_STATUS_CLEAR_IA | L3_AGENT_STATUS_CLEAR_TA,
95        );
96        self.regs.write32(base_offset + L3_ERROR_LOG, 0);
97    }
98
99    pub fn read_flag_status(&self, irq_type: usize) -> u32 {
100        let offset = if irq_type == 0 { L3_SI_FLAG_STATUS_0 } else { L3_SI_FLAG_STATUS_1 };
101        self.regs.read32(offset)
102    }
103
104    pub fn handle_error_irq(&mut self, irq_type: usize) -> Option<Omap3L3ErrorInfo> {
105        let status = self.read_flag_status(irq_type);
106        if status == 0 {
107            return None;
108        }
109
110        let source = status.trailing_zeros() as usize;
111        let bases = if irq_type == 0 { L3_APP_BASES } else { L3_DEBUG_BASES };
112
113        if source >= bases.len() {
114            return None;
115        }
116
117        let base = bases[source];
118        let info = self.read_error_at(base);
119        self.clear_error_at(base);
120        self.error_count.fetch_add(1, Ordering::Relaxed);
121        Some(info)
122    }
123}
124
125impl BusDriver for OmapL3Smx {
126    fn name(&self) -> &str { "omap-l3-smx" }
127
128    fn compatible(&self) -> &[&str] { COMPATIBLE }
129
130    fn init(&mut self, base: usize) -> Result<(), BusError> {
131        self.regs.init(base, 0x10000);
132        self.power_state = PowerState::On;
133        Ok(())
134    }
135
136    fn shutdown(&mut self) -> Result<(), BusError> {
137        self.power_state = PowerState::Off;
138        Ok(())
139    }
140
141    fn read_reg(&self, offset: usize) -> Result<u32, BusError> {
142        if !self.regs.is_valid() { return Err(BusError::InitFailed); }
143        Ok(self.regs.read32(offset))
144    }
145
146    fn write_reg(&mut self, offset: usize, value: u32) -> Result<(), BusError> {
147        if !self.regs.is_valid() { return Err(BusError::InitFailed); }
148        self.regs.write32(offset, value);
149        Ok(())
150    }
151
152    fn error_count(&self) -> u64 {
153        self.error_count.load(Ordering::Relaxed)
154    }
155
156    fn handle_irq(&mut self) -> bool {
157        let app = self.handle_error_irq(0);
158        let dbg = self.handle_error_irq(1);
159        app.is_some() || dbg.is_some()
160    }
161}
